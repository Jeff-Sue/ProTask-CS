from __future__ import annotations

import argparse
import json
from collections import defaultdict
from typing import Any, Callable, Dict, List, Set

from .io import write_json
from .schema import extract_case_ids


KNOWLEDGE_LABELS = {"ExplainedResponse", "CaseRecommendation"}
QUERY_POSITIVE_MODES = {"rewrite", "refine", "reuse"}


def safe_div(a: float, b: float) -> float:
    return float(a) / float(b) if b else 0.0


def f1_score(precision: float, recall: float) -> float:
    return safe_div(2 * precision * recall, precision + recall)


def normalize_case_id(value: Any) -> str:
    return str(value or "").strip().upper()


def unique_keep_order(values: List[str]) -> List[str]:
    seen: Set[str] = set()
    output: List[str] = []
    for value in values:
        value = normalize_case_id(value)
        if value and value not in seen:
            seen.add(value)
            output.append(value)
    return output


def flatten_predictions(data: Any) -> List[Dict[str, Any]]:
    if not isinstance(data, list):
        raise ValueError(f"Prediction file should contain a list, got {type(data)}")

    rows: List[Dict[str, Any]] = []
    for item in data:
        if not isinstance(item, dict):
            continue

        if isinstance(item.get("turns"), list):
            dialog_id = str(item.get("dialog_id") or "")
            for turn in item["turns"]:
                if not isinstance(turn, dict):
                    continue
                row = dict(turn)
                row.setdefault("dialog_id", dialog_id)
                rows.append(row)
        else:
            rows.append(item)

    return rows


def get_query(row: Dict[str, Any]) -> Dict[str, Any]:
    value = row.get("q")
    return value if isinstance(value, dict) else {}


def get_trigger(row: Dict[str, Any]) -> Dict[str, Any]:
    value = row.get("t")
    return value if isinstance(value, dict) else {}


def get_policy(row: Dict[str, Any]) -> Dict[str, Any]:
    value = row.get("p")
    return value if isinstance(value, dict) else {}


def get_case_id(case: Any) -> str:
    if not isinstance(case, dict):
        return ""
    return normalize_case_id(
        case.get("case_id")
        or case.get("caseId")
        or case.get("id")
        or case.get("doc_id")
        or case.get("knowledge_id")
    )


def collect_case_ids_from_evidence(evidence: Any) -> List[str]:
    if not isinstance(evidence, dict):
        return []

    case_ids: List[str] = []
    selected = evidence.get("selected_case_ids") or evidence.get("case_ids") or []
    if isinstance(selected, list):
        case_ids.extend(
            normalize_case_id(case_id)
            for case_id in selected
            if normalize_case_id(case_id)
        )

    cases = (
        evidence.get("cases")
        or evidence.get("retrieved_cases")
        or evidence.get("evidence")
        or []
    )
    if isinstance(cases, list):
        for case in cases:
            case_id = get_case_id(case)
            if case_id:
                case_ids.append(case_id)

    return unique_keep_order(case_ids)


def get_gold_case_ids(row: Dict[str, Any]) -> List[str]:
    explicit_ids = row.get("gold_case_ids")
    if isinstance(explicit_ids, list):
        return unique_keep_order(
            [normalize_case_id(case_id) for case_id in explicit_ids if normalize_case_id(case_id)]
        )

    golden_response = str(row.get("golden_response") or "")
    return unique_keep_order(
        [normalize_case_id(case_id) for case_id in extract_case_ids(golden_response)]
    )


def get_retrieved_case_ids(row: Dict[str, Any]) -> List[str]:
    active_ids = collect_case_ids_from_evidence(
        row.get("active_evidence_for_policy_response")
    )
    if active_ids:
        return active_ids
    return collect_case_ids_from_evidence(row.get("e"))


def get_predicted_case_id(row: Dict[str, Any]) -> str:
    return normalize_case_id(get_policy(row).get("case_id"))


def knowledge_link_called(row: Dict[str, Any]) -> bool:
    query = get_query(row)
    trigger = get_trigger(row)

    query_called = (
        query.get("query_mode") in QUERY_POSITIVE_MODES
        and bool(str(query.get("query_text") or "").strip())
    )
    evidence_called = trigger.get("evidence_mode") in {
        "retrieve_new_evidence",
        "reuse_existing_evidence",
    }
    return query_called or evidence_called


def knowledge_policy_selected(row: Dict[str, Any]) -> bool:
    return get_policy(row).get("label") in KNOWLEDGE_LABELS


def recommendation_made(row: Dict[str, Any]) -> bool:
    return knowledge_policy_selected(row) and bool(get_predicted_case_id(row))


def case_hit(row: Dict[str, Any]) -> bool:
    predicted_case_id = get_predicted_case_id(row)
    return bool(predicted_case_id) and predicted_case_id in set(get_gold_case_ids(row))


def topk_hit(row: Dict[str, Any], k: int) -> bool:
    gold_case_ids = set(get_gold_case_ids(row))
    retrieved_case_ids = get_retrieved_case_ids(row)[:k]
    return bool(gold_case_ids.intersection(retrieved_case_ids))


def binary_metrics(
    rows: List[Dict[str, Any]],
    pred_fn: Callable[[Dict[str, Any]], bool],
    gold_fn: Callable[[Dict[str, Any]], bool],
) -> Dict[str, Any]:
    tp = fp = fn = tn = 0

    for row in rows:
        pred = bool(pred_fn(row))
        gold = bool(gold_fn(row))

        if pred and gold:
            tp += 1
        elif pred and not gold:
            fp += 1
        elif not pred and gold:
            fn += 1
        else:
            tn += 1

    precision = safe_div(tp, tp + fp)
    recall = safe_div(tp, tp + fn)

    return {
        "precision": precision,
        "recall": recall,
        "f1": f1_score(precision, recall),
        "tp": tp,
        "fp": fp,
        "fn": fn,
        "tn": tn,
    }


def evaluate_case_turns(rows: List[Dict[str, Any]]) -> Dict[str, Any]:
    case_rows = [row for row in rows if get_gold_case_ids(row)]
    linked_rows = [row for row in case_rows if knowledge_link_called(row)]
    top5_hit_rows = [row for row in linked_rows if topk_hit(row, 5)]
    knowledge_policy_rows = [
        row for row in top5_hit_rows if knowledge_policy_selected(row)
    ]
    concrete_selection_rows = [
        row for row in knowledge_policy_rows if get_predicted_case_id(row)
    ]

    return {
        "case_turn_count": len(case_rows),
        "knowledge_link_call_rate": safe_div(len(linked_rows), len(case_rows)),
        "top1_recall_given_link": safe_div(
            sum(topk_hit(row, 1) for row in linked_rows), len(linked_rows)
        ),
        "top3_recall_given_link": safe_div(
            sum(topk_hit(row, 3) for row in linked_rows), len(linked_rows)
        ),
        "top5_recall_given_link": safe_div(
            sum(topk_hit(row, 5) for row in linked_rows), len(linked_rows)
        ),
        "case_policy_accuracy_given_top5_hit": safe_div(
            len(knowledge_policy_rows), len(top5_hit_rows)
        ),
        "case_selection_accuracy_given_policy": safe_div(
            sum(case_hit(row) for row in concrete_selection_rows),
            len(concrete_selection_rows),
        ),
        "final_recommendation_success_rate": safe_div(
            sum(case_hit(row) for row in case_rows), len(case_rows)
        ),
        "linked_case_turn_count": len(linked_rows),
        "top5_hit_count": len(top5_hit_rows),
        "knowledge_policy_count": len(knowledge_policy_rows),
        "concrete_selection_count": len(concrete_selection_rows),
    }


def evaluate_recommendation_timing(rows: List[Dict[str, Any]]) -> Dict[str, Any]:
    metrics = binary_metrics(
        rows=rows,
        pred_fn=recommendation_made,
        gold_fn=lambda row: bool(get_gold_case_ids(row)),
    )
    metrics["recommendation_rate"] = safe_div(
        sum(recommendation_made(row) for row in rows), len(rows)
    )
    metrics["gold_case_turn_rate"] = safe_div(
        sum(bool(get_gold_case_ids(row)) for row in rows), len(rows)
    )
    metrics["early_recommendation_rate"] = safe_div(
        metrics["fp"], metrics["fp"] + metrics["tn"]
    )
    metrics["missed_recommendation_rate"] = safe_div(
        metrics["fn"], metrics["tp"] + metrics["fn"]
    )
    return metrics


def evaluate_dialogues(rows: List[Dict[str, Any]]) -> Dict[str, Any]:
    by_dialog: Dict[str, List[Dict[str, Any]]] = defaultdict(list)
    for row in rows:
        by_dialog[str(row.get("dialog_id") or "")].append(row)

    dialogue_count_with_gold = 0
    successful_dialogue_count = 0
    offsets: List[int] = []

    for dialogue_rows in by_dialog.values():
        dialogue_rows = sorted(
            dialogue_rows,
            key=lambda row: int(row.get("turn_id") or 0),
        )

        gold_first_turn: Dict[str, int] = {}
        for row in dialogue_rows:
            turn_id = int(row.get("turn_id") or 0)
            for case_id in get_gold_case_ids(row):
                gold_first_turn.setdefault(case_id, turn_id)

        if not gold_first_turn:
            continue

        dialogue_count_with_gold += 1
        successful_events = []

        for row in dialogue_rows:
            if not recommendation_made(row):
                continue

            predicted_case_id = get_predicted_case_id(row)
            if predicted_case_id in gold_first_turn:
                successful_events.append(
                    (int(row.get("turn_id") or 0), predicted_case_id)
                )

        if not successful_events:
            continue

        successful_dialogue_count += 1
        predicted_turn, predicted_case_id = min(
            successful_events,
            key=lambda item: item[0],
        )
        offsets.append(predicted_turn - gold_first_turn[predicted_case_id])

    early_count = sum(offset < 0 for offset in offsets)
    on_time_count = sum(offset == 0 for offset in offsets)
    late_count = sum(offset > 0 for offset in offsets)

    return {
        "dialog_count": len(by_dialog),
        "dialog_with_gold_count": dialogue_count_with_gold,
        "successful_dialog_count": successful_dialogue_count,
        "dialog_hit_rate": safe_div(
            successful_dialogue_count, dialogue_count_with_gold
        ),
        "avg_signed_hit_offset": safe_div(sum(offsets), len(offsets)),
        "avg_absolute_hit_offset": safe_div(
            sum(abs(offset) for offset in offsets), len(offsets)
        ),
        "early_hit_rate_among_success": safe_div(early_count, len(offsets)),
        "on_time_hit_rate_among_success": safe_div(on_time_count, len(offsets)),
        "late_hit_rate_among_success": safe_div(late_count, len(offsets)),
        "hit_offset_distribution": {
            "early": early_count,
            "on_time": on_time_count,
            "late": late_count,
        },
    }


def evaluate(rows: List[Dict[str, Any]]) -> Dict[str, Any]:
    return {
        "basic_counts": {
            "turn_count": len(rows),
            "dialog_count": len(
                {str(row.get("dialog_id") or "") for row in rows}
            ),
            "case_turn_count": sum(
                bool(get_gold_case_ids(row)) for row in rows
            ),
        },
        "case_turn_performance": evaluate_case_turns(rows),
        "recommendation_timing": evaluate_recommendation_timing(rows),
        "dialogue_level_performance": evaluate_dialogues(rows),
    }


def load_json(path: str) -> Any:
    with open(path, "r", encoding="utf-8") as file:
        return json.load(file)


def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--pred_path", required=True)
    parser.add_argument("--output_path", default="")
    args = parser.parse_args()

    data = load_json(args.pred_path)
    rows = flatten_predictions(data)
    result = evaluate(rows)

    # if args.output_path:
    #     write_json(args.output_path, result)

    print(json.dumps(result, ensure_ascii=False, indent=2))


if __name__ == "__main__":
    main()
