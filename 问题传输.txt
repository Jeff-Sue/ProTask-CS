import re

def extract_case_id(turn):
    pattern2 = r'\$\$案例链接:\s*(KT\d+)\$\$'
    match = re.search(pattern2, turn)
    if not match:
        print("current wrong turn: ", turn)
        return None
    case_id = match.group(1)

    return case_id


turn = "[案例链接: KT00370519]。\n您好目前还不支持的。"
print(extract_case_id(turn))
