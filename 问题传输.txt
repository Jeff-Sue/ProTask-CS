class ModelRunner:

    def __init__(
        self,
        base_model: str,
        lora_path: str,
        gpu_id: int,
        batch_size: int = 8,
    ):
        self.device = f"cuda:{gpu_id}"
        self.batch_size = batch_size

        torch.cuda.set_device(gpu_id)

        self.tokenizer = AutoTokenizer.from_pretrained(
            base_model,
            trust_remote_code=True,
            padding_side="left",
        )

        if self.tokenizer.pad_token_id is None:
            self.tokenizer.pad_token_id = (
                self.tokenizer.eos_token_id
            )

        base = AutoModelForCausalLM.from_pretrained(
            base_model,
            trust_remote_code=True,
            torch_dtype=torch.float16,
            low_cpu_mem_usage=True,
            device_map={
                "": self.device,
            },
        )

        self.model = PeftModel.from_pretrained(
            base,
            lora_path,
            device_map={
                "": self.device,
            },
            torch_dtype=torch.float16,
            is_trainable=False,
        )

        self.model.eval()

    @torch.inference_mode()
    def generate(
        self,
        messages_batch: list[list[dict[str, str]]],
        max_new_tokens: int,
    ) -> list[str]:

        results = []

        for start in range(0, len(messages_batch), self.batch_size):
            messages_chunk = messages_batch[
                start:start + self.batch_size
            ]

            texts = [
                self.tokenizer.apply_chat_template(
                    messages,
                    tokenize=False,
                    add_generation_prompt=True,
                    enable_thinking=False,
                )
                for messages in messages_chunk
            ]

            inputs = self.tokenizer(
                texts,
                return_tensors="pt",
                padding=True,
                truncation=True,
                max_length=8192,
            ).to(self.device)

            outputs = self.model.generate(
                **inputs,
                max_new_tokens=max_new_tokens,
                do_sample=False,
                pad_token_id=self.tokenizer.pad_token_id,
                eos_token_id=self.tokenizer.eos_token_id,
            )

            prompt_length = inputs["input_ids"].shape[1]

            generated = outputs[:, prompt_length:]

            decoded = self.tokenizer.batch_decode(
                generated,
                skip_special_tokens=True,
            )

            results.extend(text.strip() for text in decoded)

        return results


def log_progress(
    wave: int,
    module: str,
    active_num: int,
    message: str = "",
) -> None:

    current_time = datetime.now().strftime("%Y-%m-%d %H:%M:%S")

    text = (
        f"[{current_time}] "
        f"[Wave {wave}] "
        f"[Active {active_num}] "
        f"[{module}]"
    )

    if message:
        text += f" {message}"

    print(text, flush=True)

def reformat_dialog(dialog: dict) -> list:
    new_dialogue = []
    for dialogue_id, dialogue_content in dialog.items():
        sample = {}
        sample['dialog_id'] = dialogue_id
        sample['turns'] = []
        for idx, turn in enumerate(dialogue_content['text']):
            if "用户" in list(turn.keys()):
                turn_sample = {}
                turn_sample['turn_id'] = int(idx / 2)
                turn_sample['current_user'] = turn['用户']
                turn_sample['golden_response'] = dialogue_content['text'][idx+1]['客服']
                turn_sample['case_ids'] = []
                for case in dialogue_content['caseId']:
                    if case in turn_sample['golden_response']:
                        turn_sample['case_ids'].append(case)
                sample['turns'].append(turn_sample)
        new_dialogue.append(sample)

    return new_dialogue



def make_messages(
    prompt: str,
    data: dict[str, Any],
) -> list[dict[str, str]]:

    return [
        {
            "role": "system",
            "content": prompt,
        },
        {
            "role": "user",
            "content": json.dumps(
                data,
                ensure_ascii=False,
            ),
        },
    ]


def parse_json(text: str) -> dict[str, Any]:
    try:
        return json.loads(repair_json(text))
    except Exception:
        return {
            "raw_output": text,
            "parse_error": True,
        }


def compact_evidence(
    evidence: list[dict[str, Any]],
) -> list[dict[str, Any]]:

    return [
        {
            "case_id": case.get("case_id", ""),
            "title": case.get("title", ""),
            "keywords": case.get("keywords", []),
        }
        for case in evidence
    ]


def should_retrieve(trigger: dict[str, Any]) -> bool:
    return (
        trigger.get("trigger") is True
        or trigger.get("evidence_mode")
        == "retrieve_new_evidence"
    )


def format_dialogue_history(
    history_state: list[dict[str, Any]],
    current_user: str,
) -> str:

    messages = []

    for turn in history_state:
        messages.append(f"用户：{turn['u']}")
        messages.append(f"客服：{turn['r']}")

    messages.append(f"用户：{current_user}")

    return "\n".join(messages)

def load_runner(
    name: str,
    base_model: str,
    lora_path: str,
    gpu_id: int,
    batch_size: int = 4,
):
    print(
        f"[Model Loading] [{name}] 开始加载到 cuda:{gpu_id}",
        flush=True,
    )

    start_time = time.time()

    runner = ModelRunner(
        base_model=base_model,
        lora_path=lora_path,
        gpu_id=gpu_id,
        batch_size=batch_size,
    )

    print(
        f"[Model Loading] [{name}] 加载完成，"
        f"设备 cuda:{gpu_id}，"
        f"耗时 {time.time() - start_time:.2f}s",
        flush=True,
    )

    return runner


def run_pipeline(
    dialogues: list[dict[str, Any]],
    gpu_ids: list[int],
    history_response_source: str,
) -> list[dict[str, Any]]:

    query_prompt = "根据历史状态、active_evidence和当前用户输入生成query，只输出合法JSON。"
    trigger_prompt = "判断当前轮是否需要重新调用RAG，只输出合法JSON。"
    policy_prompt = "根据当前状态和active_evidence选择客服策略，只输出合法JSON。"
    response_prompt = "根据当前状态、active_evidence和policy生成客服回复，只输出回复内容。"

    print(
        f"[Model Loading] 开始并行加载四个模型，GPU={gpu_ids}",
        flush=True,
    )

    load_start_time = time.time()

    with ThreadPoolExecutor(max_workers=4) as executor:
        futures = [
            executor.submit(
                ModelRunner,
                BASE_MODEL,
                lora_path,
                gpu_id,
                8,
            )
            for lora_path, gpu_id in zip(
                [Q_LORA, T_LORA, P_LORA, R_LORA],
                gpu_ids,
            )
        ]

        q_model, t_model, p_model, r_model = [
            future.result()
            for future in futures
        ]

    print(
        f"[Model Loading] 四个模型全部加载完成，"
        f"总耗时 {time.time() - load_start_time:.2f}s",
        flush=True,
    )

    states = [
        {
            "dialogue": dialogue,
            "current_index": 0,
            "history_state": [],
            "active_evidence": [],
            "outputs": [],
        }
        for dialogue in dialogues
    ]

    wave = 0

    while True:
        active_states = [
            state
            for state in states
            if state["current_index"]
            < len(state["dialogue"]["turns"])
        ]
        if not active_states:
            break
            
        wave += 1
        active_num = len(active_states)

        current_turns = [
            state["dialogue"]["turns"][
                state["current_index"]
            ]
            for state in active_states
        ]

        log_progress(
            wave,
            "START",
            active_num,
            f"开始处理第 {wave} 轮",
        )

        # =====================================================
        # Query Generator
        # =====================================================

        log_progress(
            wave,
            "QUERY",
            active_num,
            "开始 Query Generator",
        )

        start_time = time.time()

        q_messages = []

        for state, turn in zip(active_states, current_turns):
            q_messages.append(
                make_messages(
                    query_prompt,
                    {
                        "history_state": state["history_state"],
                        "active_evidence": state["active_evidence"],
                        "current_user": turn["current_user"],
                    },
                )
            )

        q_outputs = q_model.generate(
            q_messages,
            max_new_tokens=128,
        )

        q_states = [
            parse_json(output)
            for output in q_outputs
        ]

        log_progress(
            wave,
            "QUERY",
            active_num,
            f"完成，耗时 {time.time() - start_time:.2f}s",
        )

        # =====================================================
        # RAG Trigger
        # =====================================================

        log_progress(
            wave,
            "TRIGGER",
            active_num,
            "开始 RAG Trigger",
        )

        start_time = time.time()

        t_messages = []

        for state, turn, query in zip(
            active_states,
            current_turns,
            q_states,
        ):
            t_messages.append(
                make_messages(
                    trigger_prompt,
                    {
                        "history_state": state["history_state"],
                        "active_evidence": state["active_evidence"],
                        "current_user": turn["current_user"],
                        "current_query": query,
                    },
                )
            )

        t_outputs = t_model.generate(
            t_messages,
            max_new_tokens=128,
        )

        t_states = [
            parse_json(output)
            for output in t_outputs
        ]

        log_progress(
            wave,
            "TRIGGER",
            active_num,
            f"完成，耗时 {time.time() - start_time:.2f}s",
        )

        # =====================================================
        # RAG
        # =====================================================

        rag_indices = [
            index
            for index, trigger in enumerate(t_states)
            if should_retrieve(trigger)
        ]

        rag_num = len(rag_indices)

        log_progress(
            wave,
            "RAG",
            active_num,
            f"需要检索 {rag_num}/{active_num} 个样本",
        )

        if rag_indices:

            start_time = time.time()

            rag_histories = [
                format_dialogue_history(
                    active_states[index]["history_state"],
                    current_turns[index]["current_user"],
                )
                for index in rag_indices
            ]

            rag_queries = [
                q_states[index].get("query_text", "")
                for index in rag_indices
            ]

            rag_results = batch_zefeng_rag(
                rag_histories,
                rag_queries,
            )

            for index, evidence in zip(
                rag_indices,
                rag_results,
            ):
                active_states[index]["active_evidence"] = evidence

            log_progress(
                wave,
                "RAG",
                active_num,
                f"全部检索完成，耗时 {time.time() - start_time:.2f}s",
            )

        else:
            log_progress(
                wave,
                "RAG",
                active_num,
                "本轮无需调用 RAG",
            )

        # =====================================================
        # Policy Predictor
        # =====================================================

        log_progress(
            wave,
            "POLICY",
            active_num,
            "开始 Policy Predictor",
        )

        start_time = time.time()

        p_messages = []

        for state, turn, query, trigger in zip(
            active_states,
            current_turns,
            q_states,
            t_states,
        ):
            p_messages.append(
                make_messages(
                    policy_prompt,
                    {
                        "history_state": state["history_state"],
                        "active_evidence": state["active_evidence"],
                        "current_user": turn["current_user"],
                        "current_query": query,
                        "current_trigger": trigger,
                    },
                )
            )

        p_outputs = p_model.generate(
            p_messages,
            max_new_tokens=128,
        )

        p_states = [
            parse_json(output)
            for output in p_outputs
        ]

        log_progress(
            wave,
            "POLICY",
            active_num,
            f"完成，耗时 {time.time() - start_time:.2f}s",
        )

        # =====================================================
        # Response Generator
        # =====================================================

        log_progress(
            wave,
            "RESPONSE",
            active_num,
            "开始 Response Generator",
        )

        start_time = time.time()

        r_messages = []

        for state, turn, query, trigger, policy in zip(
            active_states,
            current_turns,
            q_states,
            t_states,
            p_states,
        ):
            r_messages.append(
                make_messages(
                    response_prompt,
                    {
                        "history_state": state["history_state"],
                        "active_evidence": state["active_evidence"],
                        "current_user": turn["current_user"],
                        "current_query": query,
                        "current_trigger": trigger,
                        "current_policy": policy,
                    },
                )
            )

        predicted_responses = r_model.generate(
            r_messages,
            max_new_tokens=512,
        )

        log_progress(
            wave,
            "RESPONSE",
            active_num,
            f"完成，耗时 {time.time() - start_time:.2f}s",
        )

        # =====================================================
        # 更新对话状态
        # =====================================================

        log_progress(
            wave,
            "UPDATE",
            active_num,
            "开始更新 history_state 和 active_evidence",
        )

        for (
            state,
            turn,
            query,
            trigger,
            policy,
            predicted_response,
        ) in zip(
            active_states,
            current_turns,
            q_states,
            t_states,
            p_states,
            predicted_responses,
        ):

            result = {
                **turn,
                "predicted_q": query,
                "predicted_t": trigger,
                "predicted_evidence": state["active_evidence"],
                "predicted_p": policy,
                "predicted_response": predicted_response,
            }

            state["outputs"].append(result)

            if history_response_source == "golden":
                history_response = turn.get(
                    "golden_response",
                    predicted_response,
                )
            else:
                history_response = predicted_response

            state["history_state"].append(
                {
                    "turn_id": turn["turn_id"],
                    "u": turn["current_user"],
                    "q": query,
                    "t": trigger,
                    "e": compact_evidence(
                        state["active_evidence"]
                    ),
                    "p": policy,
                    "r": history_response,
                }
            )

            state["current_index"] += 1

        completed_num = sum(
            state["current_index"]
            >= len(state["dialogue"]["turns"])
            for state in states
        )

        log_progress(
            wave,
            "FINISH",
            active_num,
            (
                f"第 {wave} 轮完成；"
                f"已完成对话 {completed_num}/{len(states)}"
            ),
        )

    return [
        {
            "dialog_id": state["dialogue"]["dialog_id"],
            "turns": state["outputs"],
        }
        for state in states
    ]


def main():

    parser = argparse.ArgumentParser()

    parser.add_argument(
        "--input_path",
        required=True,
    )

    parser.add_argument(
        "--output_path",
        required=True,
    )

    parser.add_argument(
        "--gpus",
        required=True,
        help="例如 0,1,2,3",
    )

    parser.add_argument(
        "--history_response_source",
        choices=["golden", "predicted"],
        default="golden",
    )

    args = parser.parse_args()

    with open(args.input_path, "r", encoding="utf-8") as file:
        dialogues = json.load(file)

    groups = reformat_dialog(dialogues)
    print(groups[0])
    current_dialogues = groups[:150]

    gpu_ids = [
        int(item)
        for item in args.gpus.split(",")
    ]

    results = run_pipeline(
        dialogues=current_dialogues,
        gpu_ids=gpu_ids,
        history_response_source=args.history_response_source,
    )

    with open(args.output_path, "w", encoding="utf-8") as file:
        json.dump(
            results,
            file,
            ensure_ascii=False,
            indent=2,
        )


if __name__ == "__main__":
    main()
