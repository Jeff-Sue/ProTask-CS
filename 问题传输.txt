import json
import re
from typing import List, Dict, Any

class RAGEvaluator:
    def __init__(self, predictions_file: str, ground_truth_file: str):
        """
        初始化评估器
        
        Args:
            predictions_file: 预测结果文件路径
            ground_truth_file: 真实答案文件路径（包含正确的case_id）
        """
        self.predictions = self._load_json(predictions_file)
        self.ground_truth = self._load_json(ground_truth_file)
        
    def _load_json(self, file_path: str) -> List[Dict]:
        """加载JSON文件"""
        with open(file_path, 'r', encoding='utf-8') as f:
            content = f.read()
        try:
            return json.loads(content)
        except json.JSONDecodeError:
            # 如果是每行一个JSON
            data = []
            for line in content.strip().split('\n'):
                if line:
                    data.append(json.loads(line))
            return data
    
    def _extract_case_ids(self, cases_str: str) -> List[str]:
        """从cases字符串中提取所有case_id"""
        # 匹配 'case_id': 'KTxxxxxxx' 格式
        pattern = r"'case_id':\s*'([^']+)'"
        return re.findall(pattern, cases_str)
    
    def _get_correct_case_id(self, idx: int) -> str:
        """获取第idx个样本的正确case_id"""
        if idx < len(self.ground_truth):
            return self.ground_truth[idx]
        return ''
    
    def evaluate(self) -> Dict[str, float]:
        """
        执行完整的评估流程
        
        Returns:
            dict: 包含所有评估指标的字典
        """
        total_samples = len(self.predictions)
        
        # 统计变量
        trigger_count = 0
        top5_recall_count = 0
        top3_recall_count = 0
        top1_recall_count = 0
        strategy_accurate_count = 0
        case_accurate_count = 0
        
        for idx, pred in enumerate(self.predictions):
            # 获取正确case_id
            correct_case_id = self._get_correct_case_id(idx)
            if not correct_case_id:
                continue
            
            # 1. 检查是否触发检索
            trigger = pred.get('trigger', '')
            is_triggered = '检索+常识' in trigger
            
            if is_triggered:
                trigger_count += 1
                
                # 2. 提取cases中的case_id
                cases_str = pred.get('cases', '')
                case_ids = self._extract_case_ids(cases_str)
                
                # 计算top-5召回
                top5_cases = case_ids[:5]
                if correct_case_id in top5_cases:
                    top5_recall_count += 1
                    
                    # 3. 检查策略推荐
                    policy = pred.get('policy', {})
                    if isinstance(policy, str):
                        try:
                            policy = eval(policy)
                        except:
                            policy = {}
                    
                    label = policy.get('label', '')
                    if label == 'CaseRecommendation':
                        strategy_accurate_count += 1
                        
                        # 4. 检查案例选择
                        predicted_case_id = policy.get('case_id', '')
                        if predicted_case_id == correct_case_id:
                            case_accurate_count += 1
                
                # 计算top-3召回
                top3_cases = case_ids[:3]
                if correct_case_id in top3_cases:
                    top3_recall_count += 1
                
                # 计算top-1召回
                top1_cases = case_ids[:1]
                if correct_case_id in top1_cases:
                    top1_recall_count += 1
        
        # 计算最终指标
        results = {}
        
        # 第一：知识检索触发率
        results['trigger_rate'] = trigger_count / total_samples if total_samples > 0 else 0
        
        # 第二：有效案例召回率（top-5）
        results['recall_top5'] = top5_recall_count / trigger_count if trigger_count > 0 else 0
        
        # 第三：有效案例召回率（top-3）
        results['recall_top3'] = top3_recall_count / trigger_count if trigger_count > 0 else 0
        
        # 第四：有效案例召回率（top-1）
        results['recall_top1'] = top1_recall_count / trigger_count if trigger_count > 0 else 0
        
        # 第五：策略推荐准确率
        results['strategy_accuracy'] = strategy_accurate_count / top5_recall_count if top5_recall_count > 0 else 0
        
        # 第六：案例选择准确率
        results['case_accuracy'] = case_accurate_count / strategy_accurate_count if strategy_accurate_count > 0 else 0
        
        # 第七：综合得分
        results['composite_score'] = (
            results['trigger_rate'] * 
            results['recall_top5'] * 
            results['strategy_accuracy'] * 
            results['case_accuracy']
        )
        
        # 添加原始计数
        results['total_samples'] = total_samples
        results['trigger_count'] = trigger_count
        results['top5_recall_count'] = top5_recall_count
        results['top3_recall_count'] = top3_recall_count
        results['top1_recall_count'] = top1_recall_count
        results['strategy_accurate_count'] = strategy_accurate_count
        results['case_accurate_count'] = case_accurate_count
        
        return results
    
    def print_results(self):
        """打印评估结果"""
        results = self.evaluate()
        
        print("=" * 60)
        print("RAG系统评估结果")
        print("=" * 60)
        print(f"\n样本总数: {results['total_samples']}")
        print(f"触发检索数: {results['trigger_count']}")
        print(f"\n1. 知识检索触发率: {results['trigger_rate']:.2%}")
        print(f"2. 有效案例召回率 (top-5): {results['recall_top5']:.2%}")
        print(f"3. 有效案例召回率 (top-3): {results['recall_top3']:.2%}")
        print(f"4. 有效案例召回率 (top-1): {results['recall_top1']:.2%}")
        print(f"5. 策略推荐准确率: {results['strategy_accuracy']:.2%}")
        print(f"6. 案例选择准确率: {results['case_accuracy']:.2%}")
        print(f"\n7. 综合得分: {results['composite_score']:.4f}")
        print("=" * 60)

# 使用示例
if __name__ == "__main__":
    # 假设预测结果文件为 predictions.json
    # 真实答案文件为 ground_truth.json（包含 correct_case_id 字段）
    
    evaluator = RAGEvaluator(
        predictions_file="result.json",
        ground_truth_file="golden.json"
    )
    evaluator.print_results()
