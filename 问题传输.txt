from fastapi import FastAPI
from pydantic import BaseModel

from pipeline import Pipeline

# =========================
# 1. 初始化服务（模型常驻）
# =========================
app = FastAPI(title="Huawei IT LLM Service")

pipeline = Pipeline()   # ⭐只加载一次（关键）


# =========================
# 2. 请求格式
# =========================
class Request(BaseModel):
    dialogue: str


# =========================
# 3. 返回格式
# =========================
class Response(BaseModel):
    query: str
    trigger: str
    cases: str
    policy: str
    response: str


# =========================
# 4. 推理接口
# =========================
@app.post("/chat", response_model=Response)
def chat(req: Request):

    result = pipeline.run(req.dialogue)

    return {
        "query": result["query"],
        "trigger": result["trigger"],
        "cases": result["cases"],
        "policy": result["policy"],
        "response": result["response"]
    }


# =========================
# 5. health check
# =========================
@app.get("/health")
def health():
    return {"status": "ok"}
