if __name__ == "__main__":

        if i["rag_trigger"]["knowledge_support"] == "仅常识":
            useful_evidence = "无，目前用户需求仅需要常识解决"
        elif i["rag_trigger"]["knowledge_support"] == "知识不足":
            useful_evidence = "无，目前知识不足以解决用户需求"
        else:
            useful_evidence = str(i["retrieval_cases"])
        policy_user = f"对话历史：{messages_to_string(i['dialogue_history'])}\n案例：{useful_evidence}\n\n请预测下一步动作："

            select_case_content = "无"

        response_sample["conversations"] = [{"from":"human", "value":response_user}, {"from":"gpt", "value":i['response']}]

        query_generator_data.append(query_sample)
        rag_trigger_data.append(trigger_sample)
        policy_predictor_data.append(policy_sample)
