#!/usr/bin/env bash

set -euo pipefail

echo "[Patch] 检查 vLLM tokenizer 兼容性..."

python - <<'PY'
from pathlib import Path
import vllm.transformers_utils.tokenizer as tokenizer_module

path = Path(tokenizer_module.__file__)
text = path.read_text(encoding="utf-8")

old = """    tokenizer_all_special_tokens_extended = (
        tokenizer.all_special_tokens_extended)"""

new = """    tokenizer_all_special_tokens_extended = getattr(
        tokenizer,
        "all_special_tokens_extended",
        tokenizer_all_special_tokens,
    )"""

if old in text:
    backup_path = Path(str(path) + ".bak")

    if not backup_path.exists():
        backup_path.write_text(text, encoding="utf-8")

    path.write_text(
        text.replace(old, new, 1),
        encoding="utf-8",
    )

    print(f"[Patch] 已修复：{path}")

elif '"all_special_tokens_extended",' in text and "getattr(" in text:
    print(f"[Patch] 已经修复，无需重复修改：{path}")

else:
    print(f"[Patch] 未找到原始代码，请检查：{path}")
    raise SystemExit(1)
PY

BASE_MODEL="/data/MODEL_ZOO/Qwen/Qwen3-8B"

Q_LORA="/home/ma-user/work/jiahui/静态多轮微调/LlamaFactory/saves/static_multi/query_generator"
T_LORA="/home/ma-user/work/jiahui/静态多轮微调/LlamaFactory/saves/static_multi/rag_trigger"
P_LORA="/home/ma-user/work/jiahui/静态多轮微调/LlamaFactory/saves/static_multi/policy_predictor"
R_LORA="/home/ma-user/work/jiahui/静态多轮微调/LlamaFactory/saves/static_multi/response_generator"

LOG_DIR="./vllm_logs"
PID_DIR="./vllm_pids"

mkdir -p "${LOG_DIR}"
mkdir -p "${PID_DIR}"

# 避免多个服务共享同一个临时编译目录
export VLLM_WORKER_MULTIPROC_METHOD=spawn
export TOKENIZERS_PARALLELISM=false


start_service() {
    local gpu_id=$1
    local port=$2
    local model_name=$3
    local lora_path=$4
    local log_path="${LOG_DIR}/${model_name}.log"
    local pid_path="${PID_DIR}/${model_name}.pid"

    echo "启动 ${model_name}: GPU=${gpu_id}, PORT=${port}"

    CUDA_VISIBLE_DEVICES="${gpu_id}" nohup \
    vllm serve "${BASE_MODEL}" \
        --host 0.0.0.0 \
        --port "${port}" \
        --served-model-name "${model_name}" \
        --dtype float16 \
        --trust-remote-code \
        --tensor-parallel-size 1 \
        --max-model-len 8192 \
        --gpu-memory-utilization 0.90 \
        --max-num-seqs 16 \
        --enable-lora \
        --max-loras 1 \
        --max-lora-rank 64 \
        --lora-modules "${model_name}=${lora_path}" \
        --default-chat-template-kwargs '{"enable_thinking": false}' \
        > "${log_path}" 2>&1 &

    echo $! > "${pid_path}"

    echo "${model_name} PID=$(cat "${pid_path}")"
}


start_service 0 8001 "query_generator"    "${Q_LORA}"
start_service 1 8002 "rag_trigger"        "${T_LORA}"
start_service 2 8003 "policy_predictor"   "${P_LORA}"
start_service 3 8004 "response_generator" "${R_LORA}"

echo
echo "四个服务已启动。"
echo "日志目录：${LOG_DIR}"
echo "PID 目录：${PID_DIR}"
