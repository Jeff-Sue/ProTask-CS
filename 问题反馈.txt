from __future__ import annotations

import argparse
import json
from pathlib import Path
from typing import Any

import torch
from json_repair import repair_json
from peft import PeftModel
from transformers import AutoModelForCausalLM, AutoTokenizer

# 改成你的实际导入路径
from src.rag import rag_zefeng_batch


BASE_MODEL = "/data/MODEL_ZOO/Qwen/Qwen3-8B"

Q_LORA = "saves/static_multi/query_generator"
T_LORA = "saves/static_multi/trigger"
P_LORA = "saves/static_multi/policy"
R_LORA = "saves/static_multi/response"

PROMPT_DIR = Path("prompts")


class ModelRunner:

    def __init__(
        self,
        base_model: str,
        lora_path: str,
        gpu_id: int,
        batch_size: int = 8,
    ):
        self.device = f"cuda:{gpu_id}"
        self.batch_size = batch_size

        self.tokenizer = AutoTokenizer.from_pretrained(
            base_model,
            trust_remote_code=True,
            padding_side="left",
        )

        if self.tokenizer.pad_token_id is None:
            self.tokenizer.pad_token_id = self.tokenizer.eos_token_id

        base = AutoModelForCausalLM.from_pretrained(
            base_model,
            trust_remote_code=True,
            torch_dtype=torch.float16,
            low_cpu_mem_usage=True,
        )

        self.model = PeftModel.from_pretrained(
            base,
            lora_path,
        ).to(self.device)

        self.model.eval()

    @torch.inference_mode()
    def generate(
        self,
        messages_batch: list[list[dict[str, str]]],
        max_new_tokens: int,
    ) -> list[str]:

        results = []

        for start in range(0, len(messages_batch), self.batch_size):
            messages_chunk = messages_batch[
                start:start + self.batch_size
            ]

            texts = [
                self.tokenizer.apply_chat_template(
                    messages,
                    tokenize=False,
                    add_generation_prompt=True,
                    enable_thinking=False,
                )
                for messages in messages_chunk
            ]

            inputs = self.tokenizer(
                texts,
                return_tensors="pt",
                padding=True,
                truncation=True,
                max_length=16384,
            ).to(self.device)

            outputs = self.model.generate(
                **inputs,
                max_new_tokens=max_new_tokens,
                do_sample=False,
                pad_token_id=self.tokenizer.pad_token_id,
                eos_token_id=self.tokenizer.eos_token_id,
            )

            prompt_length = inputs["input_ids"].shape[1]

            generated = outputs[:, prompt_length:]

            decoded = self.tokenizer.batch_decode(
                generated,
                skip_special_tokens=True,
            )

            results.extend(text.strip() for text in decoded)

        return results


def load_prompt(name: str) -> str:
    return (PROMPT_DIR / name).read_text(encoding="utf-8")


def make_messages(
    prompt: str,
    data: dict[str, Any],
) -> list[dict[str, str]]:

    return [
        {
            "role": "system",
            "content": prompt,
        },
        {
            "role": "user",
            "content": json.dumps(
                data,
                ensure_ascii=False,
            ),
        },
    ]


def parse_json(text: str) -> dict[str, Any]:
    try:
        return json.loads(repair_json(text))
    except Exception:
        return {
            "raw_output": text,
            "parse_error": True,
        }


def compact_evidence(
    evidence: list[dict[str, Any]],
) -> list[dict[str, Any]]:

    return [
        {
            "case_id": case.get("case_id", ""),
            "title": case.get("title", ""),
            "keywords": case.get("keywords", []),
        }
        for case in evidence
    ]


def should_retrieve(trigger: dict[str, Any]) -> bool:
    return (
        trigger.get("trigger") is True
        or trigger.get("evidence_mode")
        == "retrieve_new_evidence"
    )


def format_dialogue_history(
    history_state: list[dict[str, Any]],
    current_user: str,
) -> str:

    messages = []

    for turn in history_state:
        messages.append(f"用户：{turn['u']}")
        messages.append(f"客服：{turn['r']}")

    messages.append(f"用户：{current_user}")

    return "\n".join(messages)


def balance_dialogues(
    dialogues: list[dict[str, Any]],
) -> tuple[list[dict[str, Any]], list[dict[str, Any]]]:

    dialogues = sorted(
        dialogues,
        key=lambda x: len(x["turns"]),
        reverse=True,
    )

    group_0 = []
    group_1 = []

    load_0 = 0
    load_1 = 0

    for dialogue in dialogues:
        turn_num = len(dialogue["turns"])

        if load_0 <= load_1:
            group_0.append(dialogue)
            load_0 += turn_num
        else:
            group_1.append(dialogue)
            load_1 += turn_num

    return group_0, group_1


def run_pipeline(
    dialogues: list[dict[str, Any]],
    gpu_ids: list[int],
    history_response_source: str,
) -> list[dict[str, Any]]:

    query_prompt = load_prompt("query.prompt")
    trigger_prompt = load_prompt("trigger.prompt")
    policy_prompt = load_prompt("policy.prompt")
    response_prompt = load_prompt("response.prompt")

    q_model = ModelRunner(
        BASE_MODEL,
        Q_LORA,
        gpu_ids[0],
    )

    t_model = ModelRunner(
        BASE_MODEL,
        T_LORA,
        gpu_ids[1],
    )

    p_model = ModelRunner(
        BASE_MODEL,
        P_LORA,
        gpu_ids[2],
    )

    r_model = ModelRunner(
        BASE_MODEL,
        R_LORA,
        gpu_ids[3],
    )

    states = [
        {
            "dialogue": dialogue,
            "current_index": 0,
            "history_state": [],
            "active_evidence": [],
            "outputs": [],
        }
        for dialogue in dialogues
    ]

    while True:

        active_states = [
            state
            for state in states
            if state["current_index"]
            < len(state["dialogue"]["turns"])
        ]

        if not active_states:
            break

        current_turns = [
            state["dialogue"]["turns"][
                state["current_index"]
            ]
            for state in active_states
        ]

        # =====================================================
        # Query Generator
        # =====================================================

        q_messages = []

        for state, turn in zip(active_states, current_turns):
            q_messages.append(
                make_messages(
                    query_prompt,
                    {
                        "history_state": state["history_state"],
                        "active_evidence": state["active_evidence"],
                        "current_user": turn["current_user"],
                    },
                )
            )

        q_outputs = q_model.generate(
            q_messages,
            max_new_tokens=128,
        )

        q_states = [
            parse_json(output)
            for output in q_outputs
        ]

        # =====================================================
        # RAG Trigger
        # =====================================================

        t_messages = []

        for state, turn, query in zip(
            active_states,
            current_turns,
            q_states,
        ):
            t_messages.append(
                make_messages(
                    trigger_prompt,
                    {
                        "history_state": state["history_state"],
                        "active_evidence": state["active_evidence"],
                        "current_user": turn["current_user"],
                        "current_query": query,
                    },
                )
            )

        t_outputs = t_model.generate(
            t_messages,
            max_new_tokens=128,
        )

        t_states = [
            parse_json(output)
            for output in t_outputs
        ]

        # =====================================================
        # RAG：统一等待所有需要检索的样本完成
        # =====================================================

        rag_indices = [
            index
            for index, trigger in enumerate(t_states)
            if should_retrieve(trigger)
        ]

        if rag_indices:

            rag_histories = [
                format_dialogue_history(
                    active_states[index]["history_state"],
                    current_turns[index]["current_user"],
                )
                for index in rag_indices
            ]

            rag_queries = [
                q_states[index].get("query_text", "")
                for index in rag_indices
            ]

            rag_results = rag_zefeng_batch(
                rag_histories,
                rag_queries,
            )

            for index, evidence in zip(
                rag_indices,
                rag_results,
            ):
                active_states[index]["active_evidence"] = evidence

        # 不检索的样本保持原有 active_evidence

        # =====================================================
        # Policy Predictor
        # =====================================================

        p_messages = []

        for state, turn, query, trigger in zip(
            active_states,
            current_turns,
            q_states,
            t_states,
        ):
            p_messages.append(
                make_messages(
                    policy_prompt,
                    {
                        "history_state": state["history_state"],
                        "active_evidence": state["active_evidence"],
                        "current_user": turn["current_user"],
                        "current_query": query,
                        "current_trigger": trigger,
                    },
                )
            )

        p_outputs = p_model.generate(
            p_messages,
            max_new_tokens=128,
        )

        p_states = [
            parse_json(output)
            for output in p_outputs
        ]

        # =====================================================
        # Response Generator
        # =====================================================

        r_messages = []

        for state, turn, query, trigger, policy in zip(
            active_states,
            current_turns,
            q_states,
            t_states,
            p_states,
        ):
            r_messages.append(
                make_messages(
                    response_prompt,
                    {
                        "history_state": state["history_state"],
                        "active_evidence": state["active_evidence"],
                        "current_user": turn["current_user"],
                        "current_query": query,
                        "current_trigger": trigger,
                        "current_policy": policy,
                    },
                )
            )

        predicted_responses = r_model.generate(
            r_messages,
            max_new_tokens=512,
        )

        # =====================================================
        # 更新每个对话
        # =====================================================

        for (
            state,
            turn,
            query,
            trigger,
            policy,
            predicted_response,
        ) in zip(
            active_states,
            current_turns,
            q_states,
            t_states,
            p_states,
            predicted_responses,
        ):

            result = {
                **turn,
                "predicted_q": query,
                "predicted_t": trigger,
                "predicted_evidence": state["active_evidence"],
                "predicted_p": policy,
                "predicted_response": predicted_response,
            }

            state["outputs"].append(result)

            if history_response_source == "golden":
                history_response = turn.get(
                    "golden_response",
                    predicted_response,
                )
            else:
                history_response = predicted_response

            state["history_state"].append(
                {
                    "turn_id": turn["turn_id"],
                    "u": turn["current_user"],
                    "q": query,
                    "t": trigger,
                    "e": compact_evidence(
                        state["active_evidence"]
                    ),
                    "p": policy,
                    "r": history_response,
                }
            )

            state["current_index"] += 1

    return [
        {
            "dialog_id": state["dialogue"]["dialog_id"],
            "turns": state["outputs"],
        }
        for state in states
    ]


def main():

    parser = argparse.ArgumentParser()

    parser.add_argument(
        "--input",
        required=True,
    )

    parser.add_argument(
        "--output",
        required=True,
    )

    parser.add_argument(
        "--shard",
        type=int,
        choices=[0, 1],
        required=True,
    )

    parser.add_argument(
        "--gpus",
        required=True,
        help="例如 0,1,2,3",
    )

    parser.add_argument(
        "--history_response_source",
        choices=["golden", "predicted"],
        default="golden",
    )

    args = parser.parse_args()

    with open(args.input, "r", encoding="utf-8") as file:
        dialogues = json.load(file)

    groups = balance_dialogues(dialogues)
    current_dialogues = groups[args.shard]

    gpu_ids = [
        int(item)
        for item in args.gpus.split(",")
    ]

    results = run_pipeline(
        dialogues=current_dialogues,
        gpu_ids=gpu_ids,
        history_response_source=args.history_response_source,
    )

    with open(args.output, "w", encoding="utf-8") as file:
        json.dump(
            results,
            file,
            ensure_ascii=False,
            indent=2,
        )


if __name__ == "__main__":
    main()
