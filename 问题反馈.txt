from utils.qwen_api_multi import parallel_inference
from utils.scu_rag import batch_zefeng_rag

def parse_json(value: Any) -> Dict[str, Any]:
    """
    Parse a value into a JSON object.

    Supports:
    - dict
    - JSON string
    - model wrapper such as {"response": "..."}
    """
    if isinstance(value, dict):
        return value

    if not isinstance(value, str):
        return {}

    value = value.strip()
    if not value:
        return {}

    try:
        result = json.loads(value)
        return result if isinstance(result, dict) else {}
    except (json.JSONDecodeError, TypeError):
        return {}


def unwrap_model_response(output: Any) -> Any:
    """
    Extract the actual model response from the API wrapper.

    Example:
        {"response": "{\"query_mode\": \"rewrite\", ...}"}
    """
    if isinstance(output, dict):
        for key in ("response", "text", "content", "output"):
            if key in output:
                return output[key]

    return output


def parse_model_json(output: Any) -> Dict[str, Any]:
    """
    Parse JSON returned by parallel_inference.
    """
    return parse_json(unwrap_model_response(output))


def parse_model_text(output: Any) -> str:
    """
    Parse plain-text response returned by parallel_inference.

    Response Generator may return either:
    - plain text
    - {"response": "..."}
    - {"text": "..."}
    """
    value = unwrap_model_response(output)

    if isinstance(value, str):
        value = value.strip()

        # The model may return a JSON object containing the final response.
        parsed = parse_json(value)
        if parsed:
            for key in ("response", "text", "content", "answer"):
                if key in parsed:
                    return str(parsed[key]).strip()

        return value

    if isinstance(value, dict):
        for key in ("response", "text", "content", "answer"):
            if key in value:
                return str(value[key]).strip()

    return str(value) if value is not None else ""


def prompt_read(path: Path) -> str:
    if not path.exists():
        raise FileNotFoundError(f"Prompt file not found: {path}")

    return path.read_text(encoding="utf-8").strip()


def write_json(path: str | Path, data: Any) -> None:
    output_path = Path(path)
    output_path.parent.mkdir(parents=True, exist_ok=True)
    output_path.write_text(
        json.dumps(data, ensure_ascii=False, indent=2),
        encoding="utf-8",
    )


def build_messages(
    system_prompt: str,
    data: Dict[str, Any],
) -> List[Dict[str, str]]:
    return [
        {
            "role": "system",
            "content": system_prompt,
        },
        {
            "role": "user",
            "content": json.dumps(
                data,
                ensure_ascii=False,
            ),
        },
    ]

# Memory
# ============================================================

def init_memory() -> Dict[str, Any]:
    """
    Summarization-based memory.

    dialog_state no longer stores recent raw dialogue turns.
    It only stores a continuously updated natural-language summary.
    """
    return {
        "issue_state": {
            "current_issue": "",
            "query_history": [],
        },
        "dialog_state": {
            "dialogue_summary": "",
        },
        "evidence_state": {
            "active_cases": [],
        },
        "interaction_state": {
            "stage": "Start",
            "last_action": "",
            "action_history": [],
        },
    }


# ============================================================
# Deterministic Memory Update
# ============================================================

def update_issue_memory(
    memory: Dict[str, Any],
    query_state: Dict[str, Any],
) -> Dict[str, Any]:
    mode = query_state.get("query_mode", "none")
    query = str(query_state.get("query_text", "")).strip()

    if mode in {"rewrite", "refine"} and query:
        memory["issue_state"]["current_issue"] = query

        history = memory["issue_state"]["query_history"]
        if not history or history[-1] != query:
            history.append(query)

    # reuse / none: keep current issue unchanged
    return memory


def update_evidence_memory(
    memory: Dict[str, Any],
    trigger_state: Dict[str, Any],
    evidence: Any,
) -> Dict[str, Any]:
    mode = trigger_state.get(
        "evidence_mode",
        "no_evidence_needed",
    )

    if mode == "retrieve_new_evidence":
        memory["evidence_state"]["active_cases"] = evidence or []

    # reuse_existing / no_evidence_needed:
    # keep previously active evidence
    return memory


def update_interaction_memory(
    memory: Dict[str, Any],
    policy_state: Dict[str, Any],
) -> Dict[str, Any]:
    label = str(policy_state.get("label", "")).strip()

    if not label:
        return memory

    interaction_state = memory["interaction_state"]
    interaction_state["last_action"] = label
    interaction_state["action_history"].append(label)

    if label in {
        "AskMissingSlot",
        "AskClarification",
    }:
        interaction_state["stage"] = "Clarification"

    elif label in {
        "ExplainedResponse",
        "CaseRecommendation",
    }:
        interaction_state["stage"] = "Resolution"

    elif label == "Handoff":
        interaction_state["stage"] = "End"

    # ProcessAcknowledgement does not change stage.
    return memory


def update_summary_memory(
    memory: Dict[str, Any],
    summary_state: Dict[str, Any],
) -> Dict[str, Any]:
    """
    Store the summary produced by the Summary Generator.
    """
    summary = summary_state.get("dialogue_summary", "")

    if isinstance(summary, str) and summary.strip():
        memory["dialog_state"]["dialogue_summary"] = summary.strip()

    return memory


# ============================================================
# Prompt Input Builders
# ============================================================

def build_query_input(
    memory: Dict[str, Any],
    user: str,
) -> Dict[str, Any]:
    return {
        "issue_state": memory["issue_state"],
        "dialog_state": memory["dialog_state"],
        "current": {
            "u": user,
        },
    }


def build_trigger_input(
    memory: Dict[str, Any],
) -> Dict[str, Any]:
    return {
        "issue_state": memory["issue_state"],
        "evidence_state": memory["evidence_state"],
        "dialog_state": memory["dialog_state"],
    }


def build_policy_input(
    memory: Dict[str, Any],
    user: str,
) -> Dict[str, Any]:
    return {
        "issue_state": memory["issue_state"],
        "evidence_state": memory["evidence_state"],
        "interaction_state": memory["interaction_state"],
        "dialog_state": memory["dialog_state"],
        "current": {
            "u": user,
        },
    }


def build_response_input(
    memory: Dict[str, Any],
    user: str,
    policy_state: Dict[str, Any],
) -> Dict[str, Any]:
    return {
        "issue_state": memory["issue_state"],
        "evidence_state": memory["evidence_state"],
        "interaction_state": memory["interaction_state"],
        "dialog_state": memory["dialog_state"],
        "current": {
            "u": user,
            # Keep the complete policy object, especially case_id.
            "policy": policy_state,
        },
    }


def build_summary_input(
    memory: Dict[str, Any],
    user: str,
    assistant: str,
) -> Dict[str, Any]:
    return {
        "previous_summary": (
            memory["dialog_state"]["dialogue_summary"]
        ),
        "current_turn": {
            "user": user,
            "assistant": assistant,
        },
    }


# ============================================================
# Dialogue Utilities
# ============================================================

def build_user_turn_batch(
    data: Dict[str, Any],
    t_id: int,
    dialogue_histories: Dict[str, str],
) -> List[Dict[str, Any]]:
    """
    Collect all dialogues that contain a user message at position t_id.

    Assumption:
        item["text"] alternates between:
        {"用户": "..."}
        {"客服": "..."}
    """
    current_batch: List[Dict[str, Any]] = []

    for dialog_id, item in data.items():
        messages = item.get("text", [])

        if t_id >= len(messages):
            continue

        msg = messages[t_id]

        if "用户" in msg:
            user = str(msg["用户"])
            dialogue_histories[dialog_id] += f"用户: {user}\n"
        elif "客服" in msg:
            assistant = str(msg["客服"])
            dialogue_histories[dialog_id] += f"客服: {assistant}\n"
            continue
        else:
            continue

        golden_response = ""

        if (
            t_id + 1 < len(messages)
            and "客服" in messages[t_id + 1]
        ):
            golden_response = str(
                messages[t_id + 1]["客服"]
            )

        current_batch.append(
            {
                "dialog_id": dialog_id,
                "message_index": t_id,
                "turn_id": t_id // 2,
                "user": user,
                "golden_response": golden_response,
                "dialogue_history": (
                    dialogue_histories[dialog_id]
                ),
            }
        )

    return current_batch


# ============================================================
# Batch Pipeline
# ============================================================

def run_summarization_agent_batch(
    data: Dict[str, Any],
    summary_response_source: str = "golden",
) -> Dict[str, Any]:
    """
    Run the Summarization-based memory baseline.

    summary_response_source:
        golden:
            Use the original customer-service response to update memory.
            Recommended for representation comparison and teacher-forced
            evaluation.

        generated:
            Use the generated response to update memory.
            Recommended for rollout/error-accumulation evaluation.
    """
    if summary_response_source not in {
        "golden",
        "generated",
    }:
        raise ValueError(
            "summary_response_source must be "
            "'golden' or 'generated'"
        )

    query_prompt = prompt_read(
        Path("prompts/query.prompt")
    )
    trigger_prompt = prompt_read(
        Path("prompts/trigger.prompt")
    )
    policy_prompt = prompt_read(
        Path("prompts/policy.prompt")
    )
    response_prompt = prompt_read(
        Path("prompts/response.prompt")
    )
    summary_prompt = prompt_read(
        Path("prompts/summary.prompt")
    )

    memories: Dict[str, Dict[str, Any]] = {
        dialog_id: init_memory()
        for dialog_id in data
    }

    dialogue_histories: Dict[str, str] = {
        dialog_id: ""
        for dialog_id in data
    }

    results: Dict[str, Dict[str, Any]] = {
        dialog_id: {
            "turns": [],
        }
        for dialog_id in data
    }

    max_message_count = max(
        (
            len(item.get("text", []))
            for item in data.values()
        ),
        default=0,
    )

    for t_id in range(max_message_count):
        current_batch = build_user_turn_batch(
            data=data,
            t_id=t_id,
            dialogue_histories=dialogue_histories,
        )

        if not current_batch:
            continue

        print(
            f"\n========== MESSAGE {t_id} | "
            f"USER TURNS {len(current_batch)} =========="
        )

        # ==================================================
        # 1. Query Generator
        # ==================================================
        print("[1/6] Query Batch")

        query_messages = [
            build_messages(
                query_prompt,
                build_query_input(
                    memories[x["dialog_id"]],
                    x["user"],
                ),
            )
            for x in current_batch
        ]

        query_outputs = parallel_inference(
            query_messages
        )

        if len(query_outputs) != len(current_batch):
            raise RuntimeError(
                "Query output count does not match "
                "the input batch size."
            )

        for x, output in zip(
            current_batch,
            query_outputs,
        ):
            query_state = parse_model_json(output)
            x["query"] = query_state

            dialog_id = x["dialog_id"]
            memories[dialog_id] = update_issue_memory(
                memories[dialog_id],
                query_state,
            )

        # ==================================================
        # 2. RAG Trigger
        # ==================================================
        print("[2/6] Trigger Batch")

        trigger_messages = [
            build_messages(
                trigger_prompt,
                build_trigger_input(
                    memories[x["dialog_id"]]
                ),
            )
            for x in current_batch
        ]

        trigger_outputs = parallel_inference(
            trigger_messages
        )

        if len(trigger_outputs) != len(current_batch):
            raise RuntimeError(
                "Trigger output count does not match "
                "the input batch size."
            )

        for x, output in zip(
            current_batch,
            trigger_outputs,
        ):
            x["trigger"] = parse_model_json(output)

        # ==================================================
        # 3. RAG Retrieval
        # ==================================================
        print("[3/6] RAG Batch")

        rag_dialogues: List[str] = []
        rag_queries: List[str] = []
        rag_indices: List[int] = []

        for index, x in enumerate(current_batch):
            dialog_id = x["dialog_id"]
            trigger_state = x["trigger"]

            if (
                trigger_state.get("evidence_mode")
                == "retrieve_new_evidence"
            ):
                query = memories[dialog_id][
                    "issue_state"
                ]["current_issue"]

                # Do not call RAG with an empty query.
                if query:
                    rag_dialogues.append(
                        x["dialogue_history"]
                    )
                    rag_queries.append(query)
                    rag_indices.append(index)
                else:
                    x["evidence"] = []
            else:
                x["evidence"] = memories[dialog_id][
                    "evidence_state"
                ]["active_cases"]

        if rag_dialogues:
            rag_outputs = batch_zefeng_rag(
                rag_dialogues,
                rag_queries,
            )

            if len(rag_outputs) != len(rag_indices):
                raise RuntimeError(
                    "RAG output count does not match "
                    "the retrieval batch size."
                )

            for index, evidence in zip(
                rag_indices,
                rag_outputs,
            ):
                current_batch[index]["evidence"] = (
                    evidence
                )

        for x in current_batch:
            dialog_id = x["dialog_id"]

            # Ensure the field always exists.
            x.setdefault("evidence", [])

            memories[dialog_id] = (
                update_evidence_memory(
                    memories[dialog_id],
                    x["trigger"],
                    x["evidence"],
                )
            )

        # ==================================================
        # 4. Policy Predictor
        # ==================================================
        print("[4/6] Policy Batch")

        policy_messages = [
            build_messages(
                policy_prompt,
                build_policy_input(
                    memories[x["dialog_id"]],
                    x["user"],
                ),
            )
            for x in current_batch
        ]

        policy_outputs = parallel_inference(
            policy_messages
        )

        if len(policy_outputs) != len(current_batch):
            raise RuntimeError(
                "Policy output count does not match "
                "the input batch size."
            )

        for x, output in zip(
            current_batch,
            policy_outputs,
        ):
            policy_state = parse_model_json(output)
            x["policy"] = policy_state

            dialog_id = x["dialog_id"]
            memories[dialog_id] = (
                update_interaction_memory(
                    memories[dialog_id],
                    policy_state,
                )
            )

        # ==================================================
        # 5. Response Generator
        # ==================================================
        print("[5/6] Response Batch")

        response_messages = [
            build_messages(
                response_prompt,
                build_response_input(
                    memories[x["dialog_id"]],
                    x["user"],
                    x["policy"],
                ),
            )
            for x in current_batch
        ]

        response_outputs = parallel_inference(
            response_messages
        )

        if len(response_outputs) != len(current_batch):
            raise RuntimeError(
                "Response output count does not match "
                "the input batch size."
            )

        for x, output in zip(
            current_batch,
            response_outputs,
        ):
            x["response"] = parse_model_text(output)

        # ==================================================
        # 6. Summary Memory Update
        # ==================================================
        print("[6/6] Summary Update Batch")

        summary_messages = []

        for x in current_batch:
            dialog_id = x["dialog_id"]

            if summary_response_source == "golden":
                assistant_for_summary = (
                    x["golden_response"]
                )
            else:
                assistant_for_summary = x["response"]

            x["summary_response_source"] = (
                summary_response_source
            )
            x["assistant_for_summary"] = (
                assistant_for_summary
            )

            summary_messages.append(
                build_messages(
                    summary_prompt,
                    build_summary_input(
                        memory=memories[dialog_id],
                        user=x["user"],
                        assistant=assistant_for_summary,
                    ),
                )
            )

        summary_outputs = parallel_inference(
            summary_messages
        )

        if len(summary_outputs) != len(current_batch):
            raise RuntimeError(
                "Summary output count does not match "
                "the input batch size."
            )

        for x, output in zip(
            current_batch,
            summary_outputs,
        ):
            dialog_id = x["dialog_id"]

            previous_summary = memories[dialog_id][
                "dialog_state"
            ]["dialogue_summary"]

            summary_state = parse_model_json(output)

            memories[dialog_id] = update_summary_memory(
                memories[dialog_id],
                summary_state,
            )

            updated_summary = memories[dialog_id][
                "dialog_state"
            ]["dialogue_summary"]

            x["previous_summary"] = previous_summary
            x["updated_summary"] = updated_summary

            results[dialog_id]["turns"].append(
                {
                    "turn_id": x["turn_id"],
                    "message_index": x[
                        "message_index"
                    ],
                    "user": x["user"],
                    "query": x["query"],
                    "trigger": x["trigger"],
                    "evidence": x["evidence"],
                    "policy": x["policy"],
                    "response": x["response"],
                    "golden_response": (
                        x["golden_response"]
                    ),
                    "summary_response_source": (
                        x["summary_response_source"]
                    ),
                    "previous_summary": (
                        x["previous_summary"]
                    ),
                    "updated_summary": (
                        x["updated_summary"]
                    ),
                }
            )

    for dialog_id in data:
        results[dialog_id]["final_memory"] = (
            memories[dialog_id]
        )

    return results


# ============================================================
# Main
# ============================================================

def str_to_bool(value: str) -> bool:
    value = value.lower().strip()

    if value in {"true", "1", "yes", "y"}:
        return True

    if value in {"false", "0", "no", "n"}:
        return False

    raise argparse.ArgumentTypeError(
        f"Invalid boolean value: {value}"
    )


def main() -> None:
    parser = argparse.ArgumentParser()

    parser.add_argument(
        "--data_path",
        required=True,
        help="Input JSON dialogue file.",
    )
    parser.add_argument(
        "--output_path",
        required=True,
        help="Output JSON path.",
    )
    parser.add_argument(
        "--max_dialogues",
        type=int,
        default=300,
        help="Maximum number of dialogues to run.",
    )
    parser.add_argument(
        "--summary_response_source",
        choices=["golden", "generated"],
        default="golden",
        help=(
            "Use golden or generated assistant responses "
            "to update the dialogue summary."
        ),
    )

    args = parser.parse_args()

    with open(
        args.data_path,
        "r",
        encoding="utf-8",
    ) as file:
        data = json.load(file)

    if not isinstance(data, dict):
        raise ValueError(
            "The input data must be a dictionary "
            "mapping dialog_id to dialogue objects."
        )

    if args.max_dialogues > 0:
        data = dict(
            list(data.items())[
                :args.max_dialogues
            ]
        )

    outputs = run_summarization_agent_batch(
        data=data,
        summary_response_source=(
            args.summary_response_source
        ),
    )

    write_json(
        args.output_path,
        outputs,
    )

    print(
        f"Saved {len(outputs)} dialogues "
        f"to {args.output_path}"
    )


if __name__ == "__main__":
    main()
