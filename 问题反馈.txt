# =========================

# Data schema

# =========================

@dataclass

class Pair:

    user: str

    assistant: str

@dataclass

class TargetTurn:

    dialog_id: str

    turn_id: int

    history_pairs: List[Pair]

    current_user: str

    golden_response: str

    case_ids: List[str]

# =========================

# Prompt bank

# =========================

class PromptBank:

    def __init__(self, prompt_dir: str | Path):

        self.prompt_dir = Path(prompt_dir)

        self.query = self._read("predict_query.prompt")

        self.trigger = self._read("predict_trigger.prompt")

        self.policy = self._read("predict_policy.prompt")

        self.response = self._read("predict_response.prompt")

    def _read(self, name: str) -> str:

        path = self.prompt_dir / name

        if not path.exists():

            raise FileNotFoundError(f"Prompt file not found: {path}")

        return path.read_text(encoding="utf-8").strip()

def normalize_dialogues(data: Any) -> List[Dict[str, Any]]:

    """

    Support:

    1. list of dialogue samples

    2. dict keyed by dialog_id:

       {

         "220599461": {

           "text": [{"用户": "..."}, {"客服": "..."}],

           "caseId": [...]

         }

       }

    """

    if isinstance(data, list):

        return data

    if isinstance(data, dict):

        for key in ("data", "dialogues", "items", "records"):

            if isinstance(data.get(key), list):

                return data[key]

        rows = []

        for dialog_id, sample in data.items():

            if isinstance(sample, dict):

                x = dict(sample)

                x.setdefault("dialog_id", str(dialog_id))

                rows.append(x)

        return rows

    raise ValueError(f"Unsupported data type: {type(data)}")

def load_dialogues(path: str | Path) -> List[Dict[str, Any]]:

    data = load_json_or_jsonl(str(path))

    return normalize_dialogues(data)

def parse_message(item: Any) -> Optional[Dict[str, str]]:

    if isinstance(item, str):

        text = item.strip()

        if not text:

            return None

        if text.startswith(("用户:", "用户：")):

            return {"role": "user", "text": text.split("：", 1)[-1].split(":", 1)[-1].strip()}

        if text.startswith(("客服:", "客服：")):

            return {"role": "assistant", "text": text.split("：", 1)[-1].split(":", 1)[-1].strip()}

        return {"role": "user", "text": text}

    if not isinstance(item, dict):

        return None

    for k in ("用户", "user", "human"):

        if k in item and str(item[k]).strip():

            return {"role": "user", "text": str(item[k]).strip()}

    for k in ("客服", "assistant", "agent", "system"):

        if k in item and str(item[k]).strip():

            return {"role": "assistant", "text": str(item[k]).strip()}

    role = str(item.get("role") or item.get("speaker") or "").lower()

    text = str(item.get("text") or item.get("content") or item.get("utterance") or "").strip()

    if not text:

        return None

    if role in {"user", "human", "客户", "用户"}:

        return {"role": "user", "text": text}

    if role in {"assistant", "agent", "客服", "system"}:

        return {"role": "assistant", "text": text}

    return {"role": "user", "text": text}

def get_messages(raw: Dict[str, Any]) -> List[Dict[str, str]]:

    for key in ("text", "messages", "turns", "dialogue", "conversation"):

        if isinstance(raw.get(key), list):

            msgs = []

            for item in raw[key]:

                msg = parse_message(item)

                if msg:

                    msgs.append(msg)

            return msgs

    raise ValueError("Cannot find dialogue messages in sample.")

def build_target_turns(raw: Dict[str, Any]) -> List[TargetTurn]:

    dialog_id = str(raw.get("dialog_id") or raw.get("id") or raw.get("dialogue_id") or "")

    case_ids = [

        str(x).strip()

        for x in raw.get("caseId", raw.get("case_ids", []))

        if str(x).strip()

    ]

    messages = get_messages(raw)

    completed: List[Pair] = []

    user_buffer: List[str] = []

    targets: List[TargetTurn] = []

    turn_id = 0

    for msg in messages:

        if msg["role"] == "user":

            user_buffer.append(msg["text"])

        elif msg["role"] == "assistant":

            if not user_buffer:

                continue

            current_user = "\n".join(user_buffer).strip()

            turn_id += 1

            targets.append(

                TargetTurn(

                    dialog_id=dialog_id,

                    turn_id=turn_id,

                    history_pairs=list(completed),

                    current_user=current_user,

                    golden_response=msg["text"],

                    case_ids=case_ids,

                )

            )

            completed.append(Pair(user=current_user, assistant=msg["text"]))

            user_buffer = []

    return targets

def remove_reason(obj: Any) -> Any:

    """

    Remove reason fields from q/t/p before putting them into history state.

    """

    if isinstance(obj, dict):

        return {

            k: remove_reason(v)

            for k, v in obj.items()

            if k != "reason"

        }

    if isinstance(obj, list):

        return [remove_reason(x) for x in obj]

    return obj

def compact_evidence(retrieval: Dict[str, Any]) -> Dict[str, Any]:

    """

    Keep only case_id and title in evidence history.

    Do not keep full case content.

    """

    cases = retrieval.get("retrieved_cases", []) or []

    compact_cases = []

    for c in cases:

        if not isinstance(c, dict):

            continue

        case_id = (

            c.get("case_id")

            or c.get("caseId")

            or c.get("id")

            or c.get("doc_id")

            or ""

        )

        title = (

            c.get("title")

            or c.get("case_title")

            or c.get("name")

            or c.get("summary")

            or ""

        )

        compact_cases.append(

            {

                "case_id": str(case_id),

                "title": str(title),

            }

        )

    return {

        "selected_case_ids": retrieval.get("selected_case_ids", []),

        "cases": compact_cases,

    }

def build_history_state(turn_states: List[Dict[str, Any]]) -> List[Dict[str, Any]]:

    """

    Each state item:

    {

      "turn_id": 1,

      "u": "...",

      "q": {... no reason ...},

      "t": {... no reason ...},

      "e": {"cases": [{"case_id": "...", "title": "..."}]},

      "p": {... no reason ...},

      "r": "..."

    }

    """

    history = []

    for s in turn_states:

        history.append(

            {

                "turn_id": s["turn_id"],

                "u": s["u"],

                "q": remove_reason(s["q"]),

                "t": remove_reason(s["t"]),

                "e": compact_evidence(s["e"]),

                "p": remove_reason(s["p"]),

                "r": s["r"],

            }

        )

    return history

# LLM / RAG helpers

def make_messages(system_prompt: str, payload: Dict[str, Any]) -> List[Dict[str, str]]:

    return [

        {

            "role": "system",

            "content": system_prompt,

        },

        {

            "role": "user",

            "content": json.dumps(payload, ensure_ascii=False, indent=2),

        },

    ]

def unwrap_llm_response(raw: Any) -> str:

    if isinstance(raw, dict):

        for k in ("response", "text", "content", "output"):

            if k in raw:

                return str(raw[k])

        return json.dumps(raw, ensure_ascii=False)

    return str(raw)

def call_llm_json(parallel_inference, messages: List[Dict[str, str]]) -> Dict[str, Any]:

    raw = run_llm_batch(parallel_inference, [messages], batch_size=1)

    if not isinstance(raw, list) or not raw:

        raise ValueError(f"Empty LLM output: {raw}")

    text = unwrap_llm_response(raw[0])

    obj = parse_json_obj(text)

    if not isinstance(obj, dict):

        raise ValueError(f"LLM output is not JSON object: {text}")

    return obj

def needs_rag(q: Dict[str, Any]) -> bool:

    return q.get("query_mode") in {"rewrite", "refine", "reuse"} and bool(q.get("query_text"))

def run_rag_once(batch_zefeng_rag, target: TargetTurn, q: Dict[str, Any], top_k: int) -> Dict[str, Any]:

    if not needs_rag(q):

        return {"selected_case_ids": [], "retrieved_cases": []}

    history_text = "\n".join(

        [f"用户：{p.user}\n客服：{p.assistant}" for p in target.history_pairs]

    )

    if history_text:

        history_text += "\n"

    history_text += f"用户：{target.current_user}"

    raw = run_rag_batch(batch_zefeng_rag, [history_text], [q["query_text"]])

    if isinstance(raw, list):

        raw = raw[0] if raw else None

    return normalize_retrieval(raw, top_k=top_k)

def normalize_response_obj(obj: Dict[str, Any]) -> str:

    return str(obj.get("response") or obj.get("text") or obj.get("content") or "").strip()

# Predict one turn

# =========================

def predict_query(

    bank: PromptBank,

    parallel_inference,

    history_state: List[Dict[str, Any]],

    target: TargetTurn,

) -> Dict[str, Any]:

    payload = {

        "turn_id": target.turn_id,

        "history_state": history_state,

        "current": {

            "u": target.current_user,

        },

    }

    return call_llm_json(parallel_inference, make_messages(bank.query, payload))

def predict_trigger(

    bank: PromptBank,

    parallel_inference,

    history_state: List[Dict[str, Any]],

    target: TargetTurn,

    q: Dict[str, Any],

    e: Dict[str, Any],

) -> Dict[str, Any]:

    payload = {

        "turn_id": target.turn_id,

        "history_state": history_state,

        "current": {

            "u": target.current_user,

            "q": remove_reason(q),

            "e": compact_evidence(e),

        },

    }

    return call_llm_json(parallel_inference, make_messages(bank.trigger, payload))

def predict_policy(

    bank: PromptBank,

    parallel_inference,

    history_state: List[Dict[str, Any]],

    target: TargetTurn,

    q: Dict[str, Any],

    e: Dict[str, Any],

    t: Dict[str, Any],

) -> Dict[str, Any]:

    payload = {

        "turn_id": target.turn_id,

        "history_state": history_state,

        "current": {

            "u": target.current_user,

            "q": remove_reason(q),

            "e": compact_evidence(e),

            "t": remove_reason(t),

        },

    }

    return call_llm_json(parallel_inference, make_messages(bank.policy, payload))

def predict_response(

    bank: PromptBank,

    parallel_inference,

    history_state: List[Dict[str, Any]],

    target: TargetTurn,

    q: Dict[str, Any],

    e: Dict[str, Any],

    t: Dict[str, Any],

    p: Dict[str, Any],

) -> str:

    payload = {

        "turn_id": target.turn_id,

        "history_state": history_state,

        "current": {

            "u": target.current_user,

            "q": remove_reason(q),

            "e": compact_evidence(e),

            "t": remove_reason(t),

            "p": remove_reason(p),

        },

    }

    obj = call_llm_json(parallel_inference, make_messages(bank.response, payload))

    return normalize_response_obj(obj)

def predict_one_turn(

    target: TargetTurn,

    turn_states: List[Dict[str, Any]],

    bank: PromptBank,

    parallel_inference,

    batch_zefeng_rag,

    top_k: int,

    no_response: bool = False,

) -> Dict[str, Any]:

    """

    Current turn order:

    q_n -> RAG/e_n -> t_n -> p_n -> r_n

    """

    start = time.time()

    latency: Dict[str, int] = {}

    history_state = build_history_state(turn_states)

    # q

    t0 = time.time()

    q = predict_query(bank, parallel_inference, history_state, target)

    latency["query"] = int((time.time() - t0) * 1000)

    # e

    t0 = time.time()

    e = run_rag_once(batch_zefeng_rag, target, q, top_k=top_k)

    latency["evidence"] = int((time.time() - t0) * 1000)

    # t

    t0 = time.time()

    trig = predict_trigger(bank, parallel_inference, history_state, target, q, e)

    latency["trigger"] = int((time.time() - t0) * 1000)

    # p

    t0 = time.time()

    p = predict_policy(bank, parallel_inference, history_state, target, q, e, trig)

    latency["policy"] = int((time.time() - t0) * 1000)

    # r

    if no_response:

        r = ""

        latency["response"] = 0

    else:

        t0 = time.time()

        r = predict_response(bank, parallel_inference, history_state, target, q, e, trig, p)

        latency["response"] = int((time.time() - t0) * 1000)

    latency["total"] = int((time.time() - start) * 1000)

    return {

        "turn_id": target.turn_id,

        "u": target.current_user,

        "q": q,

        "e": e,

        "t": trig,

        "p": p,

        "r": r,

        "golden_response": target.golden_response,

        "latency_ms": latency,

    }

# =========================

# Run pred-state incremental

# =========================

def run_predict_state_incremental(args) -> List[Dict[str, Any]]:

    raw_dialogues = load_dialogues(args.data_path)[: args.max_dialogues]

    bank = PromptBank(args.prompt_dir)

    parallel_inference = get_parallel_inference(args.project_root)

    batch_zefeng_rag = get_batch_rag(args.project_root)

    outputs: List[Dict[str, Any]] = []

    for raw in tqdm(raw_dialogues, desc="Predict-state dialogues"):

        targets = build_target_turns(raw)

        if not targets:

            continue

        dialog_id = targets[0].dialog_id

        turn_states: List[Dict[str, Any]] = []

        for target in targets:

            try:

                pred = predict_one_turn(

                    target=target,

                    turn_states=turn_states,

                    bank=bank,

                    parallel_inference=parallel_inference,

                    batch_zefeng_rag=batch_zefeng_rag,

                    top_k=args.top_k,

                    no_response=args.no_response,

                )

                turn_states.append(

                    {

                        "turn_id": pred["turn_id"],

                        "u": pred["u"],

                        "q": pred["q"],

                        "e": pred["e"],

                        "t": pred["t"],

                        "p": pred["p"],

                        "r": pred["r"],

                    }

                )

            except Exception as exc:

                if not args.continue_on_error:

                    raise

                turn_states.append(

                    {

                        "turn_id": target.turn_id,

                        "u": target.current_user,

                        "q": {},

                        "e": {"selected_case_ids": [], "retrieved_cases": []},

                        "t": {},

                        "p": {},

                        "r": "",

                        "error": repr(exc),

                    }

                )

        outputs.append(

            {

                "dialog_id": dialog_id,

                "case_ids": targets[0].case_ids,

                "num_turns": len(targets),

                "turns": turn_states,

            }

        )

    return outputs

def main() -> None:

    parser = argparse.ArgumentParser()

    parser.add_argument("--data_path", required=True)

    parser.add_argument("--output_path", required=True)

    parser.add_argument("--prompt_dir", default="prompts/predict_state")

    parser.add_argument("--project_root", default=".")

    parser.add_argument("--top_k", type=int, default=5)

    parser.add_argument("--max_dialogues", type=int, default=300)

    parser.add_argument("--no_response", action="store_true")

    parser.add_argument("--continue_on_error", action="store_true")

    args = parser.parse_args()

    rows = run_predict_state_incremental(args)

    write_json(args.output_path, rows)

    print(f"Saved {len(rows)} dialogues to {args.output_path}")

if __name__ == "__main__":

    main()
