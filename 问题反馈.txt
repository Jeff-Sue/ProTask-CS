from concurrent.futures import ThreadPoolExecutor

def load_runner(
    name: str,
    base_model: str,
    lora_path: str,
    gpu_id: int,
    batch_size: int = 8,
):
    print(
        f"[Model Loading] [{name}] 开始加载到 cuda:{gpu_id}",
        flush=True,
    )

    start_time = time.time()

    runner = ModelRunner(
        base_model=base_model,
        lora_path=lora_path,
        gpu_id=gpu_id,
        batch_size=batch_size,
    )

    print(
        f"[Model Loading] [{name}] 加载完成，"
        f"设备 cuda:{gpu_id}，"
        f"耗时 {time.time() - start_time:.2f}s",
        flush=True,
    )

    return runner

    f"[Model Loading] 开始并行加载四个模型，GPU={gpu_ids}",
