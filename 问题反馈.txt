from __future__ import annotations

import sys
from pathlib import Path
from typing import Any, List


def prepare_imports(project_root: str | None = None) -> None:
    if project_root:
        root = str(Path(project_root).resolve())
        if root not in sys.path:
            sys.path.insert(0, root)


def get_parallel_inference(project_root: str | None = None):
    prepare_imports(project_root)
    try:
        from utils.qwen_api import parallel_inference
    except Exception as e:
        raise ImportError(
            "Cannot import utils.qwen_api.parallel_inference. Edit pe/external.py to match your project."
        ) from e
    return parallel_inference


def get_batch_rag(project_root: str | None = None):
    prepare_imports(project_root)
    try:
        from utils.scu_rag import batch_zefeng_rag
    except Exception as e:
        raise ImportError(
            "Cannot import utils.scu_rag.batch_zefeng_rag. Edit pe/external.py to match your project."
        ) from e
    return batch_zefeng_rag


def run_llm_batch(parallel_inference, messages_list: List[List[dict]], batch_size: int = 8) -> List[str]:
    # Assumes your parallel_inference accepts a list of chat-message lists.
    return parallel_inference(messages_list, batch_size=batch_size)


def run_rag_batch(batch_zefeng_rag, histories: List[str], queries: List[str]) -> List[Any]:
    # Assumes current RAG API: batch_zefeng_rag(dialog_history_list, query_list)
    return batch_zefeng_rag(histories, queries)
