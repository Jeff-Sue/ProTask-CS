from __future__ import annotations

from dataclasses import asdict, dataclass, field
from typing import Any, Dict, List, Literal, Optional


# =========================================================
# Basic Types
# =========================================================

Role = Literal["user", "assistant"]

QueryMode = Literal["none", "reuse", "refine", "rewrite"]

EvidenceMode = Literal[
    "no_evidence_needed",
    "retrieve_new_evidence",
    "reuse_existing_evidence",
]

PolicyLabel = Literal[
    "AskMissingSlot",
    "AskClarification",
    "ExplainedResponse",
    "CaseRecommendation",
    "Handoff",
    "ProcessAcknowledgement",
]


ALLOWED_QUERY_MODES = {"none", "reuse", "refine", "rewrite"}

ALLOWED_EVIDENCE_MODES = {
    "no_evidence_needed",
    "retrieve_new_evidence",
    "reuse_existing_evidence",
}

ALLOWED_POLICY_LABELS = {
    "AskMissingSlot",
    "AskClarification",
    "ExplainedResponse",
    "CaseRecommendation",
    "Handoff",
    "ProcessAcknowledgement",
}


# =========================================================
# Dialogue Schemas
# =========================================================

@dataclass
class DialogueTurn:
    """
    A normalized dialogue turn.

    role:
        - "user"
        - "assistant"

    text:
        Raw utterance text.
    """

    role: Role
    text: str

    def to_dict(self) -> Dict[str, Any]:
        return asdict(self)


@dataclass
class Dialogue:
    """
    A full human-human dialogue.

    turns should already be normalized into DialogueTurn.
    """

    dialog_id: str
    turns: List[DialogueTurn]

    def assistant_turn_count(self) -> int:
        return sum(1 for t in self.turns if t.role == "assistant")

    def to_dict(self) -> Dict[str, Any]:
        return asdict(self)


@dataclass
class PrefixSample:
    """
    A static multi-turn prefix sample.

    For assistant target turn n:

        history =
            user_1, assistant_1, ..., user_{n-1}, assistant_{n-1}, user_n

        gold_response =
            assistant_n

    The current gold_response must NOT be included in history.
    """

    dialog_id: str
    turn_id: int
    history: List[DialogueTurn]
    target_role: Role
    gold_response: str
    target_turn_index: int

    def to_dict(self) -> Dict[str, Any]:
        return asdict(self)


@dataclass
class StaticPrefixInput:
    """
    A clearer split of PrefixSample for static prefix PE.

    completed_history:
        Fully completed golden history before the current user input:
        user_1, assistant_1, ..., user_{n-1}, assistant_{n-1}

    current_user_input:
        The user utterance before current assistant target turn n.

    golden_response:
        The current assistant_n response, used only for evaluation.
    """

    dialog_id: str
    turn_id: int
    completed_history: List[DialogueTurn]
    current_user_input: str
    golden_response: str

    def to_dict(self) -> Dict[str, Any]:
        return asdict(self)


# =========================================================
# Query Schemas
# =========================================================

@dataclass
class QueryTrajectoryStep:
    """
    One query decision step for a specific assistant target turn.
    """

    turn_id: int
    query_mode: str
    query_text: str = ""
    reason: str = ""

    def is_active(self) -> bool:
        return self.query_mode != "none" and bool(self.query_text.strip())

    def to_dict(self) -> Dict[str, Any]:
        return asdict(self)


@dataclass
class QueryTrajectoryOutput:
    """
    A query trajectory.

    In full-dialogue annotation:
        trajectory = turn 1...T

    In static prefix PE:
        trajectory = turn 1...n
        where turn 1...n-1 are reconstructed history states,
        and turn n is current prediction.

    In prediction-only mode:
        trajectory may contain only one step, i.e. turn n.
    """

    trajectory: List[QueryTrajectoryStep] = field(default_factory=list)

    def get_step(self, turn_id: int) -> Optional[QueryTrajectoryStep]:
        for step in self.trajectory:
            if step.turn_id == turn_id:
                return step
        return None

    def last_step(self) -> Optional[QueryTrajectoryStep]:
        if not self.trajectory:
            return None
        return self.trajectory[-1]

    def current_step(self, turn_id: int) -> Optional[QueryTrajectoryStep]:
        return self.get_step(turn_id)

    def to_dict(self) -> Dict[str, Any]:
        return asdict(self)


# =========================================================
# Retrieval Schemas
# =========================================================

@dataclass
class RetrievalCase:
    """
    A normalized retrieved case.
    """

    case_id: str
    title: str = ""
    content: str = ""
    keywords: List[str] = field(default_factory=list)
    score: Optional[float] = None
    metadata: Dict[str, Any] = field(default_factory=dict)

    def to_dict(self) -> Dict[str, Any]:
        return asdict(self)


@dataclass
class RetrievalOutput:
    """
    Retrieval result for one target turn.
    """

    retrieved_cases: List[RetrievalCase] = field(default_factory=list)

    raw_evidence: str = ""
    selected_case_ids: List[str] = field(default_factory=list)
    dropped_case_ids: List[str] = field(default_factory=list)
    selection_relevances: Dict[str, str] = field(default_factory=dict)
    selection_failed: bool = False
    diagnostics: List[Dict[str, Any]] = field(default_factory=list)

    def case_ids(self) -> List[str]:
        if self.selected_case_ids:
            return self.selected_case_ids
        return [case.case_id for case in self.retrieved_cases if case.case_id]

    def contains_case(self, case_id: str) -> bool:
        return case_id in set(self.case_ids())

    def to_dict(self) -> Dict[str, Any]:
        return asdict(self)


@dataclass
class RetrievalByTurn:
    """
    Retrieval results indexed by assistant target turn id.

    Used mainly for full-dialogue annotation, where each historical
    query step may have its own retrieval evidence.
    """

    by_turn: Dict[int, RetrievalOutput] = field(default_factory=dict)

    def get(self, turn_id: int) -> RetrievalOutput:
        return self.by_turn.get(turn_id, RetrievalOutput())

    def set(self, turn_id: int, output: RetrievalOutput) -> None:
        self.by_turn[turn_id] = output

    def to_dict(self) -> Dict[str, Any]:
        return {
            str(turn_id): output.to_dict()
            for turn_id, output in self.by_turn.items()
        }

    One policy decision step.

# =========================================================
# Latency Schema
# =========================================================

@dataclass
class Latency:
    """
    Latency record.

    In global batch mode, these values may be amortized average latency.
    In single-sample mode, they can be real per-sample latency.
    """

    query_ms: int = 0
    retrieval_ms: int = 0
    trigger_ms: int = 0
    policy_ms: int = 0
    response_ms: int = 0
    total_ms: int = 0

    selection_ms: int = 0
    compress_ms: int = 0
    keywords_ms: int = 0

    def to_dict(self) -> Dict[str, Any]:
        return asdict(self)


# =========================================================
# Prompt Bundle Schemas
# =========================================================

@dataclass
class AnnotationPromptBundle:
    """
    Prompts for annotation / reconstruction.

    Used for:
        1. full-dialogue annotation
        2. prefix history reconstruction in static PE

    Input:
        already happened dialogue segment

    Output:
        trajectory for all completed assistant turns in that segment.
    """

    query: str
    trigger: str
    policy: str


@dataclass
class PredictionPromptBundle:
    """
    Prompts for next-step prediction.

    Used for:
        1. static prefix PE current turn prediction
        2. later dynamic multi-turn prediction
        3. SFT data generation format

    Input:
        completed history dialogue
        historical trajectories
        current user input
        current retrieval evidence

    Output:
        current target turn query / trigger / policy / response.
    """

    query: str
    trigger: str
    policy: str
    response: str


@dataclass
class PromptBundle:
    """
    Combined prompt bundle.
    """

    annotation: AnnotationPromptBundle
    prediction: PredictionPromptBundle

# =========================================================
# Static Prefix PE Output
# =========================================================

@dataclass
class HistoryState:
    """
    Reconstructed state memory from completed golden history.

    For predicting turn n+1, this contains trajectories for turn 1...n.
    """

    query_trajectory: QueryTrajectoryOutput = field(default_factory=QueryTrajectoryOutput)
    retrieval_by_turn: RetrievalByTurn = field(default_factory=RetrievalByTurn)
    trigger_trajectory: TriggerTrajectoryOutput = field(default_factory=TriggerTrajectoryOutput)
    policy_trajectory: PolicyTrajectoryOutput = field(default_factory=PolicyTrajectoryOutput)

    def to_dict(self) -> Dict[str, Any]:
        return asdict(self)


@dataclass
class CurrentTurnPrediction:
    """
    Prediction for one current target assistant turn.

    This should contain only the current target turn decision.
    """

    turn_id: int

    query: Optional[QueryTrajectoryStep] = None
    retrieval: RetrievalOutput = field(default_factory=RetrievalOutput)
    trigger: Optional[TriggerTrajectoryStep] = None
    policy: Optional[PolicyTrajectoryStep] = None
    response: ResponseOutput = field(default_factory=ResponseOutput)

    def to_dict(self) -> Dict[str, Any]:
        return asdict(self)


@dataclass
class StaticPrefixPEResult:
    """
    Static prefix-level PE result.

    For target assistant turn n:

        history_state:
            reconstructed from completed golden history before current user input.

        current_prediction:
            predicted for target turn n.

        golden_response:
            assistant_n, used only for evaluation.
    """

    dialog_id: str
    turn_id: int

    dialogue_input: StaticPrefixInput
    history_state: HistoryState
    current_prediction: CurrentTurnPrediction

    golden_response: str
    latency: Latency = field(default_factory=Latency)

    metadata: Dict[str, Any] = field(default_factory=dict)

    def to_dict(self) -> Dict[str, Any]:
        return asdict(self)


# =========================================================
# Legacy-compatible Prefix Result
# =========================================================

@dataclass
class PrefixResult:
    """
    Legacy-compatible prefix result.

    This matches the older style:

        query_output: trajectory 1...n
        retrieval_output: retrieval for current turn n
        trigger_output: trajectory 1...n
        policy_output: trajectory 1...n
        response_output: response for current turn n

    Keep this for compatibility with older evaluators.
    """

    dialog_id: str
    turn_id: int
    prefix_sample: PrefixSample

    query_output: QueryTrajectoryOutput
    retrieval_output: RetrievalOutput
    trigger_output: TriggerTrajectoryOutput
    policy_output: PolicyTrajectoryOutput
    response_output: ResponseOutput

    latency: Latency = field(default_factory=Latency)
    metadata: Dict[str, Any] = field(default_factory=dict)

    def current_query_step(self) -> Optional[QueryTrajectoryStep]:
        return self.query_output.get_step(self.turn_id)

    def current_trigger_step(self) -> Optional[TriggerTrajectoryStep]:
        return self.trigger_output.get_step(self.turn_id)

    def current_policy_step(self) -> Optional[PolicyTrajectoryStep]:
        return self.policy_output.get_step(self.turn_id)

    def to_static_prefix_pe_result(self) -> StaticPrefixPEResult:
        """
        Convert legacy PrefixResult into the clearer StaticPrefixPEResult format.

        Since legacy format does not explicitly split completed_history and
        current_user_input, this method uses best-effort reconstruction.
        The caller should prefer building StaticPrefixInput directly in prefix.py.
        """

        current_query = self.current_query_step()
        current_trigger = self.current_trigger_step()
        current_policy = self.current_policy_step()

        history_state = HistoryState(
            query_trajectory=QueryTrajectoryOutput(
                trajectory=[
                    step
                    for step in self.query_output.trajectory
                    if step.turn_id < self.turn_id
                ]
            ),
            trigger_trajectory=TriggerTrajectoryOutput(
                trajectory=[
                    step
                    for step in self.trigger_output.trajectory
                    if step.turn_id < self.turn_id
                ]
            ),
            policy_trajectory=PolicyTrajectoryOutput(
                trajectory=[
                    step
                    for step in self.policy_output.trajectory
                    if step.turn_id < self.turn_id
                ]
            ),
        )

        current_prediction = CurrentTurnPrediction(
            turn_id=self.turn_id,
            query=current_query,
            retrieval=self.retrieval_output,
            trigger=current_trigger,
            policy=current_policy,
            response=self.response_output,
        )

        dialogue_input = StaticPrefixInput(
            dialog_id=self.dialog_id,
            turn_id=self.turn_id,
            completed_history=[],
            current_user_input="",
            golden_response=self.prefix_sample.gold_response,
        )

        return StaticPrefixPEResult(
            dialog_id=self.dialog_id,
            turn_id=self.turn_id,
            dialogue_input=dialogue_input,
            history_state=history_state,
            current_prediction=current_prediction,
            golden_response=self.prefix_sample.gold_response,
            latency=self.latency,
            metadata=self.metadata,
        )

    def to_dict(self) -> Dict[str, Any]:
        return asdict(self)

# =========================================================
# Utility Serialization
# =========================================================

def dataclass_to_dict(obj: Any) -> Dict[str, Any]:
    """
    Convert dataclass object to dict.

    This is a thin wrapper so other modules do not need to import asdict.
    """

    return asdict(obj)


def list_to_dict(items: List[Any]) -> List[Dict[str, Any]]:
    """
    Convert a list of dataclass objects to list of dicts.
    """

    return [asdict(item) for item in items]
