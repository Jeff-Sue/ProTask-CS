# ============================================================
# Data schema
# ============================================================

@dataclass
class Pair:
    user: str
    assistant: str


@dataclass
class TargetTurn:
    dialog_id: str
    turn_id: int
    history_pairs: List[Pair]
    current_user: str
    golden_response: str
    case_ids: List[str]


# ============================================================
# Prompt bank
# ============================================================

class PromptBank:
    """
    Expected prompt files:

    prompt_dir/
    ├── predict_query.prompt
    ├── predict_trigger.prompt
    ├── predict_policy.prompt
    └── predict_response.prompt
    """

    def __init__(self, prompt_dir: str | Path):
        self.prompt_dir = Path(prompt_dir)
        self.query = self._read("predict_query.prompt")
        self.trigger = self._read("predict_trigger.prompt")
        self.policy = self._read("predict_policy.prompt")
        self.response = self._read("predict_response.prompt")

    def _read(self, filename: str) -> str:
        path = self.prompt_dir / filename
        if not path.exists():
            raise FileNotFoundError(f"Prompt file not found: {path}")
        return path.read_text(encoding="utf-8").strip()

def normalize_dialogues(data: Any) -> List[Dict[str, Any]]:
    """
    Support:
    1. list-of-dialogues
    2. dict-of-dialogues:
       {
         "220599461": {
           "text": [{"用户": "..."}, {"客服": "..."}],
           "caseId": [...]
         }
       }
    """
    if isinstance(data, list):
        return data

    if isinstance(data, dict):
        for key in ("data", "dialogues", "items", "records"):
            if isinstance(data.get(key), list):
                return data[key]

        rows: List[Dict[str, Any]] = []
        for dialog_id, sample in data.items():
            if isinstance(sample, dict):
                x = dict(sample)
                x.setdefault("dialog_id", str(dialog_id))
                rows.append(x)
        return rows

    raise ValueError(f"Unsupported data type: {type(data)}")


def load_dialogues(path: str | Path) -> List[Dict[str, Any]]:
    data = load_json_or_jsonl(str(path))
    return normalize_dialogues(data)


def parse_message(item: Any) -> Optional[Dict[str, str]]:
    if isinstance(item, str):
        text = item.strip()
        if not text:
            return None

        if text.startswith(("用户:", "用户：")):
            content = text.replace("用户:", "", 1).replace("用户：", "", 1).strip()
            return {"role": "user", "text": content}

        if text.startswith(("客服:", "客服：")):
            content = text.replace("客服:", "", 1).replace("客服：", "", 1).strip()
            return {"role": "assistant", "text": content}

        return {"role": "user", "text": text}

    if not isinstance(item, dict):
        return None

    for key in ("用户", "user", "human", "客户"):
        if key in item and str(item[key]).strip():
            return {"role": "user", "text": str(item[key]).strip()}

    for key in ("客服", "assistant", "agent", "system"):
        if key in item and str(item[key]).strip():
            return {"role": "assistant", "text": str(item[key]).strip()}

    role = str(item.get("role") or item.get("speaker") or item.get("from") or "").lower()
    text = str(
        item.get("text")
        or item.get("content")
        or item.get("utterance")
        or item.get("value")
        or ""
    ).strip()

    if not text:
        return None

    if role in {"user", "human", "客户", "用户"}:
        return {"role": "user", "text": text}

    if role in {"assistant", "agent", "客服", "system", "gpt"}:
        return {"role": "assistant", "text": text}

    return {"role": "user", "text": text}


def get_messages(raw: Dict[str, Any]) -> List[Dict[str, str]]:
    for key in ("text", "messages", "turns", "dialogue", "conversation", "dialogue_history"):
        if isinstance(raw.get(key), list):
            msgs: List[Dict[str, str]] = []
            for item in raw[key]:
                msg = parse_message(item)
                if msg:
                    msgs.append(msg)
            return msgs

    raise ValueError("Cannot find dialogue messages in sample.")


def build_target_turns(raw: Dict[str, Any]) -> List[TargetTurn]:
    dialog_id = str(raw.get("dialog_id") or raw.get("id") or raw.get("dialogue_id") or "")
    case_ids = [
        str(x).strip()
        for x in raw.get("caseId", raw.get("case_ids", []))
        if str(x).strip()
    ]

    messages = get_messages(raw)

    completed: List[Pair] = []
    user_buffer: List[str] = []
    targets: List[TargetTurn] = []
    turn_id = 0

    for msg in messages:
        if msg["role"] == "user":
            user_buffer.append(msg["text"])

        elif msg["role"] == "assistant":
            if not user_buffer:
                continue

            current_user = "\n".join(user_buffer).strip()
            turn_id += 1

            targets.append(
                TargetTurn(
                    dialog_id=dialog_id,
                    turn_id=turn_id,
                    history_pairs=list(completed),
                    current_user=current_user,
                    golden_response=msg["text"],
                    case_ids=case_ids,
                )
            )

            completed.append(Pair(user=current_user, assistant=msg["text"]))
            user_buffer = []

    return targets

def make_messages(system_prompt: str, payload: Dict[str, Any]) -> List[Dict[str, str]]:
    return [
        {"role": "system", "content": system_prompt},
        {"role": "user", "content": json.dumps(payload, ensure_ascii=False, indent=2)},
    ]


def unwrap_llm_response(raw: Any) -> str:
    if isinstance(raw, dict):
        for key in ("response", "text", "content", "output"):
            if key in raw:
                return str(raw[key])
        return json.dumps(raw, ensure_ascii=False)

    return str(raw)


def parse_llm_json(raw: Any) -> Dict[str, Any]:
    text = unwrap_llm_response(raw)
    obj = parse_json_obj(text)
    if not isinstance(obj, dict):
        raise ValueError(f"LLM output is not a JSON object: {text}")
    return obj


def run_llm_json_batch(
    parallel_inference,
    messages: List[List[Dict[str, str]]],
    batch_size: int,
) -> List[Dict[str, Any]]:
    if not messages:
        return []

    raw_outputs: List[Any] = []
    for start in range(0, len(messages), batch_size):
        batch = messages[start:start + batch_size]
        raw = run_llm_batch(parallel_inference, batch, batch_size=len(batch))
        if not isinstance(raw, list):
            raw = [raw]
        if len(raw) != len(batch):
            raise ValueError(f"LLM batch length mismatch: expected {len(batch)}, got {len(raw)}")
        raw_outputs.extend(raw)

    return [parse_llm_json(x) for x in raw_outputs]


def run_llm_response_batch(
    parallel_inference,
    messages: List[List[Dict[str, str]]],
    batch_size: int,
) -> List[str]:
    objs = run_llm_json_batch(parallel_inference, messages, batch_size=batch_size)
    return [
        str(obj.get("response") or obj.get("text") or obj.get("content") or "").strip()
        for obj in objs
    ]

def remove_reason(obj: Any) -> Any:
    if isinstance(obj, dict):
        return {
            k: remove_reason(v)
            for k, v in obj.items()
            if k != "reason"
        }

    if isinstance(obj, list):
        return [remove_reason(x) for x in obj]

    return obj


def get_case_id(case: Dict[str, Any]) -> str:
    return str(
        case.get("case_id")
        or case.get("caseId")
        or case.get("id")
        or case.get("doc_id")
        or case.get("knowledge_id")
        or ""
    ).strip()


def get_case_title(case: Dict[str, Any]) -> str:
    return str(
        case.get("title")
        or case.get("case_title")
        or case.get("name")
        or case.get("subject")
        or case.get("summary")
        or ""
    ).strip()


def has_evidence(evidence: Dict[str, Any]) -> bool:
    return bool(
        evidence.get("cases")
        or evidence.get("retrieved_cases")
        or evidence.get("selected_case_ids")
    )


def empty_evidence_metadata() -> Dict[str, Any]:
    return {
        "selected_case_ids": [],
        "cases": [],
    }


def empty_active_evidence() -> Dict[str, Any]:
    return {
        "scope": "none",
        "source": "empty",
        "selected_case_ids": [],
        "cases": [],
    }


def compact_evidence_metadata(evidence: Any) -> Dict[str, Any]:
    """
    e_i 写入 history_state 时只保留 case_id/title，不保留内容。
    """
    if not isinstance(evidence, dict):
        return empty_evidence_metadata()

    raw_cases = (
        evidence.get("retrieved_cases")
        or evidence.get("cases")
        or evidence.get("evidence")
        or []
    )

    cases: List[Dict[str, str]] = []

    if isinstance(raw_cases, list):
        for case in raw_cases:
            if not isinstance(case, dict):
                continue

            case_id = get_case_id(case)
            title = get_case_title(case)

            if case_id or title:
                cases.append({"case_id": case_id, "title": title})

    selected_case_ids = evidence.get("selected_case_ids") or evidence.get("case_ids") or []

    return {
        "selected_case_ids": selected_case_ids,
        "cases": cases,
    }


def keep_full_active_evidence(evidence: Any) -> Dict[str, Any]:
    """
    active_evidence = a。

    外部证据空间用于给 policy / response 提供完整案例知识。
    不截断内容，不删除 case 内字段。
    """
    if not isinstance(evidence, dict):
        return empty_active_evidence()

    raw_cases = (
        evidence.get("retrieved_cases")
        or evidence.get("cases")
        or evidence.get("evidence")
        or []
    )

    cases: List[Dict[str, Any]] = []

    if isinstance(raw_cases, list):
        for case in raw_cases:
            if isinstance(case, dict):
                cases.append(dict(case))

    selected_case_ids = evidence.get("selected_case_ids") or evidence.get("case_ids") or []

    scope = evidence.get("scope") or evidence.get("evidence_scope") or ""
    if not scope:
        if len(cases) == 0:
            scope = "none"
        elif len(cases) == 1:
            scope = "selected_case"
        else:
            scope = "candidate_set"

    source = evidence.get("source") or evidence.get("evidence_source") or "unknown"

    return {
        "scope": scope,
        "source": source,
        "selected_case_ids": selected_case_ids,
        "cases": cases,
    }


def build_history_state(turn_states: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    """
    history_state 保存前 n-1 轮：
    (u, q, t, e, p, r)

    不保存 active_evidence。
    不保存 golden_response。
    q/t/p 去掉 reason。
    e 只保留 case_id/title。
    """
    history: List[Dict[str, Any]] = []

    for s in turn_states:
        history.append(
            {
                "turn_id": s.get("turn_id"),
                "u": s.get("u", ""),
                "q": remove_reason(s.get("q", {})),
                "t": remove_reason(s.get("t", {})),
                "e": compact_evidence_metadata(s.get("e", {})),
                "p": remove_reason(s.get("p", {})),
                "r": s.get("r", ""),
            }
        )

    return history


# ============================================================
# Evidence decision / RAG
# ============================================================

def needs_query(q: Dict[str, Any]) -> bool:
    return q.get("query_mode") in {"rewrite", "refine", "reuse"} and bool(q.get("query_text"))


def format_history_for_rag(target: TargetTurn) -> str:
    text = "\n".join(
        [f"用户：{p.user}\n客服：{p.assistant}" for p in target.history_pairs]
    )
    if text:
        text += "\n"
    text += f"用户：{target.current_user}"
    return text


def run_rag_sparse_batch(
    batch_zefeng_rag,
    targets: List[TargetTurn],
    q_steps: List[Dict[str, Any]],
    t_steps: List[Dict[str, Any]],
    batch_size: int,
    top_k: int,
) -> List[Optional[Dict[str, Any]]]:
    """
    只对 evidence_mode=retrieve_new_evidence 的样本调用 RAG。
    返回长度与 targets 相同。
    不需要 RAG 的位置为 None。
    """
    results: List[Optional[Dict[str, Any]]] = [None for _ in targets]

    rag_indices: List[int] = []
    rag_histories: List[str] = []
    rag_queries: List[str] = []

    for i, (target, q, t) in enumerate(zip(targets, q_steps, t_steps)):
        evidence_mode = t.get("evidence_mode")
        trigger = bool(t.get("trigger"))

        if (trigger is True or evidence_mode == "retrieve_new_evidence") and needs_query(q):
            rag_indices.append(i)
            rag_histories.append(format_history_for_rag(target))
            rag_queries.append(q["query_text"])

    if not rag_indices:
        return results

    for start in range(0, len(rag_indices), batch_size):
        sub_indices = rag_indices[start:start + batch_size]
        h_batch = rag_histories[start:start + batch_size]
        q_batch = rag_queries[start:start + batch_size]

        raw_batch = run_rag_batch(batch_zefeng_rag, h_batch, q_batch)
        if not isinstance(raw_batch, list):
            raw_batch = [raw_batch]

        if len(raw_batch) != len(sub_indices):
            raise ValueError(f"RAG batch length mismatch: expected {len(sub_indices)}, got {len(raw_batch)}")

        for idx, raw in zip(sub_indices, raw_batch):
            retrieval = normalize_retrieval(raw, top_k=top_k)
            retrieval["source"] = "new_retrieval"
            retrieval["scope"] = "candidate_set"
            results[idx] = retrieval

    return results


def decide_current_evidence_batch(
    batch_zefeng_rag,
    targets: List[TargetTurn],
    q_steps: List[Dict[str, Any]],
    t_steps: List[Dict[str, Any]],
    a_befores: List[Dict[str, Any]],
    rag_batch_size: int,
    top_k: int,
) -> List[Dict[str, Any]]:
    """
    根据 t_n 批量决定当前轮证据。

    返回列表中每个元素：
    {
      "e_short": ...,
      "e_content": ...,
      "a_current": ...
    }

    规则：
    - retrieve_new_evidence:
        sparse batch RAG，a_current = 新证据
    - reuse_existing_evidence:
        e_content = a_before，a_current = a_before
    - no_evidence_needed:
        e_content = empty，a_current = a_before
    """
    rag_results = run_rag_sparse_batch(
        batch_zefeng_rag=batch_zefeng_rag,
        targets=targets,
        q_steps=q_steps,
        t_steps=t_steps,
        batch_size=rag_batch_size,
        top_k=top_k,
    )

    outputs: List[Dict[str, Any]] = []

    for i, (t, a_before) in enumerate(zip(t_steps, a_befores)):
        evidence_mode = t.get("evidence_mode")
        trigger = bool(t.get("trigger"))

        if trigger is True or evidence_mode == "retrieve_new_evidence":
            retrieved = rag_results[i] if rag_results[i] is not None else empty_active_evidence()
            e_content = keep_full_active_evidence(retrieved)
            e_short = compact_evidence_metadata(e_content)
            a_current = keep_full_active_evidence(e_content)

        elif evidence_mode == "reuse_existing_evidence":
            e_content = keep_full_active_evidence(a_before)
            e_short = compact_evidence_metadata(e_content)
            a_current = keep_full_active_evidence(a_before)

        else:
            e_content = empty_active_evidence()
            e_short = empty_evidence_metadata()
            a_current = keep_full_active_evidence(a_before)

        outputs.append(
            {
                "e_short": e_short,
                "e_content": e_content,
                "a_current": a_current,
            }
        )

    return outputs


def select_case_from_evidence(evidence: Dict[str, Any], case_id: str) -> Dict[str, Any]:
    if not case_id:
        return empty_active_evidence()

    raw_cases = (
        evidence.get("retrieved_cases")
        or evidence.get("cases")
        or evidence.get("evidence")
        or []
    )

    selected_cases: List[Dict[str, Any]] = []

    for case in raw_cases:
        if not isinstance(case, dict):
            continue

        cid = get_case_id(case)
        if cid == case_id:
            selected_cases.append(dict(case))

    if not selected_cases:
        return empty_active_evidence()

    return {
        "scope": "selected_case",
        "source": "selected_by_policy",
        "selected_case_ids": [case_id],
        "cases": selected_cases,
    }


def update_active_evidence_after_policy(
    a_current: Dict[str, Any],
    e_content: Dict[str, Any],
    current_trigger: Dict[str, Any],
    current_policy: Dict[str, Any],
) -> Dict[str, Any]:
    label = str(current_policy.get("label") or "")
    case_id = str(current_policy.get("case_id") or "").strip()
    evidence_mode = str(current_trigger.get("evidence_mode") or "")

    base = e_content if has_evidence(e_content) else a_current

    if case_id:
        selected = select_case_from_evidence(base, case_id)
        if has_evidence(selected):
            return keep_full_active_evidence(selected)

    if evidence_mode == "retrieve_new_evidence":
        if has_evidence(e_content):
            updated = keep_full_active_evidence(e_content)
            if label in {"CaseRecommendation", "ExplainedResponse"} and len(updated.get("cases", [])) == 1:
                updated["scope"] = "selected_case"
                updated["source"] = "used_for_response"
            else:
                updated["scope"] = "candidate_set" if len(updated.get("cases", [])) > 1 else updated.get("scope", "candidate_set")
                updated["source"] = "new_retrieval"
            return updated

    if evidence_mode == "reuse_existing_evidence":
        return keep_full_active_evidence(a_current)

    if label in {"AskMissingSlot", "AskClarification"}:
        if has_evidence(base):
            updated = keep_full_active_evidence(base)
            updated["scope"] = "candidate_set"
            updated["source"] = "candidate_for_missing_info"
            return updated
        return keep_full_active_evidence(a_current)

    if label in {"CaseRecommendation", "ExplainedResponse"}:
        if has_evidence(base):
            updated = keep_full_active_evidence(base)
            updated["source"] = "used_for_response"
            return updated
        return keep_full_active_evidence(a_current)

    return keep_full_active_evidence(a_current)

def build_query_messages(
    bank: PromptBank,
    turn_id: int,
    history_state: List[Dict[str, Any]],
    a_before: Dict[str, Any],
    current_user: str,
) -> List[Dict[str, str]]:
    payload = {
        "turn_id": turn_id,
        "history_state": history_state,
        "active_evidence": keep_full_active_evidence(a_before),
        "current": {"u": current_user},
    }
    return make_messages(bank.query, payload)

    payload = {
        "turn_id": turn_id,
        "history_state": history_state,
        "active_evidence": keep_full_active_evidence(a_before),
        "current": {
            "u": current_user,
            "q": remove_reason(current_query),
        },
    }
    return make_messages(bank.trigger, payload)

    current_user: str,
    current_query: Dict[str, Any],

# ============================================================
# Runtime
# ============================================================

def build_runtimes(raw_dialogues: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    runtimes: List[Dict[str, Any]] = []

    for raw in raw_dialogues:
        targets = build_target_turns(raw)
        if not targets:
            continue

        runtimes.append(
            {
                "dialog_id": targets[0].dialog_id,
                "case_ids": targets[0].case_ids,
                "targets": targets,
                "turn_states": [],
                "active_evidence": empty_active_evidence(),
                "error": None,
            }
        )

    return runtimes


def active_runtimes_for_round(runtimes: List[Dict[str, Any]], round_idx: int) -> List[Dict[str, Any]]:
    return [
        rt for rt in runtimes
        if rt.get("error") is None and round_idx < len(rt["targets"])
    ]


# ============================================================
# Round-wise batch prediction
# ============================================================

def run_predict_state_round_batch(args) -> List[Dict[str, Any]]:
    raw_dialogues = load_dialogues(args.data_path)[: args.max_dialogues]

    bank = PromptBank(args.prompt_dir)
    parallel_inference = get_parallel_inference(args.project_root)
    batch_zefeng_rag = get_batch_rag(args.project_root)

    runtimes = build_runtimes(raw_dialogues)

    if not runtimes:
        return []

    max_turns = max(len(rt["targets"]) for rt in runtimes)

    for round_idx in tqdm(range(max_turns), desc="Round-wise turns"):
        active = active_runtimes_for_round(runtimes, round_idx)
        if not active:
            continue

        contexts: List[Dict[str, Any]] = []

        for rt in active:
            target = rt["targets"][round_idx]
            history_state = build_history_state(rt["turn_states"])
            a_before = keep_full_active_evidence(rt["active_evidence"])

            contexts.append(
                {
                    "runtime": rt,
                    "target": target,
                    "history_state": history_state,
                    "a_before": a_before,
                }
            )

        # ----------------------------------------------------
        # 1. Batch q
        # ----------------------------------------------------
        t0 = time.time()

        q_messages = [
            build_query_messages(
                bank=bank,
                turn_id=ctx["target"].turn_id,
                history_state=ctx["history_state"],
                a_before=ctx["a_before"],
                current_user=ctx["target"].current_user,
            )
            for ctx in contexts
        ]

        try:
            q_steps = run_llm_json_batch(
                parallel_inference=parallel_inference,
                messages=q_messages,
                batch_size=args.batch_size,
            )
        except Exception as exc:
            if not args.continue_on_error:
                raise
            for ctx in contexts:
                ctx["runtime"]["error"] = f"query round {round_idx + 1}: {repr(exc)}"
            continue

        query_ms = int((time.time() - t0) * 1000 / max(len(contexts), 1))

        # ----------------------------------------------------
        # 2. Batch t
        # ----------------------------------------------------
        t0 = time.time()

        t_messages = [
            build_trigger_messages(
                bank=bank,
                turn_id=ctx["target"].turn_id,
                history_state=ctx["history_state"],
                a_before=ctx["a_before"],
                current_user=ctx["target"].current_user,
                current_query=q_steps[i],
            )
            for i, ctx in enumerate(contexts)
        ]

        try:
            t_steps = run_llm_json_batch(
                parallel_inference=parallel_inference,
                messages=t_messages,
                batch_size=args.batch_size,
            )
        except Exception as exc:
            if not args.continue_on_error:
                raise
            for ctx in contexts:
                ctx["runtime"]["error"] = f"trigger round {round_idx + 1}: {repr(exc)}"
            continue

        trigger_ms = int((time.time() - t0) * 1000 / max(len(contexts), 1))

        # ----------------------------------------------------
        # 3. Sparse batch RAG / evidence decision
        # ----------------------------------------------------
        t0 = time.time()

        targets = [ctx["target"] for ctx in contexts]
        a_befores = [ctx["a_before"] for ctx in contexts]

        try:
            evidence_results = decide_current_evidence_batch(
                batch_zefeng_rag=batch_zefeng_rag,
                targets=targets,
                q_steps=q_steps,
                t_steps=t_steps,
                a_befores=a_befores,
                rag_batch_size=args.rag_batch_size,
                top_k=args.top_k,
            )
        except Exception as exc:
            if not args.continue_on_error:
                raise
            for ctx in contexts:
                ctx["runtime"]["error"] = f"evidence round {round_idx + 1}: {repr(exc)}"
            continue

        evidence_ms = int((time.time() - t0) * 1000 / max(len(contexts), 1))

        # ----------------------------------------------------
        # 4. Batch p
        # ----------------------------------------------------
        t0 = time.time()

        p_messages = [
            build_policy_messages(
                bank=bank,
                turn_id=ctx["target"].turn_id,
                history_state=ctx["history_state"],
                a_current=evidence_results[i]["a_current"],
                current_user=ctx["target"].current_user,
                current_query=q_steps[i],
                current_trigger=t_steps[i],
                current_evidence_content=evidence_results[i]["e_content"],
            )
            for i, ctx in enumerate(contexts)
        ]

        try:
            p_steps = run_llm_json_batch(
                parallel_inference=parallel_inference,
                messages=p_messages,
                batch_size=args.batch_size,
            )
        except Exception as exc:
            if not args.continue_on_error:
                raise
            for ctx in contexts:
                ctx["runtime"]["error"] = f"policy round {round_idx + 1}: {repr(exc)}"
            continue

        policy_ms = int((time.time() - t0) * 1000 / max(len(contexts), 1))

        # ----------------------------------------------------
        # 5. Batch r
        # ----------------------------------------------------
        t0 = time.time()

        if args.no_response:
            responses = ["" for _ in contexts]
            response_ms = 0
        else:
            r_messages = [
                build_response_messages(
                    bank=bank,
                    turn_id=ctx["target"].turn_id,
                    history_state=ctx["history_state"],
                    a_current=evidence_results[i]["a_current"],
                    current_user=ctx["target"].current_user,
                    current_query=q_steps[i],
                    current_trigger=t_steps[i],
                    current_evidence_content=evidence_results[i]["e_content"],
                    current_policy=p_steps[i],
                )
                for i, ctx in enumerate(contexts)
            ]

            try:
                responses = run_llm_response_batch(
                    parallel_inference=parallel_inference,
                    messages=r_messages,
                    batch_size=args.batch_size,
                )
            except Exception as exc:
                if not args.continue_on_error:
                    raise
                for ctx in contexts:
                    ctx["runtime"]["error"] = f"response round {round_idx + 1}: {repr(exc)}"
                continue

            response_ms = int((time.time() - t0) * 1000 / max(len(contexts), 1))

        # ----------------------------------------------------
        # 6. Update each dialogue runtime
        # ----------------------------------------------------
        for i, ctx in enumerate(contexts):
            rt = ctx["runtime"]
            target = ctx["target"]
            a_before = ctx["a_before"]
            e_result = evidence_results[i]

            a_after = update_active_evidence_after_policy(
                a_current=e_result["a_current"],
                e_content=e_result["e_content"],
                current_trigger=t_steps[i],
                current_policy=p_steps[i],
            )

            turn_state = {
                "turn_id": target.turn_id,
                "u": target.current_user,
                "q": q_steps[i],
                "t": t_steps[i],
                "e": e_result["e_short"],
                "p": p_steps[i],
                "r": responses[i],

                # 只用于输出对比，不进入 history_state
                "golden_response": target.golden_response,

                # 调试字段：不进入 history_state
                "active_evidence_before_turn": keep_full_active_evidence(a_before),
                "active_evidence_for_policy_response": keep_full_active_evidence(e_result["a_current"]),
                "active_evidence_after_turn": keep_full_active_evidence(a_after),

                "latency_ms": {
                    "query": query_ms,
                    "trigger": trigger_ms,
                    "evidence": evidence_ms,
                    "policy": policy_ms,
                    "response": response_ms,
                    "total": query_ms + trigger_ms + evidence_ms + policy_ms + response_ms,
                },
            }

            rt["turn_states"].append(turn_state)
            rt["active_evidence"] = keep_full_active_evidence(a_after)

    outputs: List[Dict[str, Any]] = []

    for rt in runtimes:
        outputs.append(
            {
                "dialog_id": rt["dialog_id"],
                "case_ids": rt["case_ids"],
                "num_turns": len(rt["targets"]),
                "turns": rt["turn_states"],
                "final_active_evidence": keep_full_active_evidence(rt["active_evidence"]),
                "error": rt.get("error"),
            }
        )

    return outputs


# ============================================================
# CLI
# ============================================================

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--data_path", required=True)
    parser.add_argument("--output_path", required=True)
    parser.add_argument("--prompt_dir", default="prompts/predict_state")
    parser.add_argument("--project_root", default=".")
    parser.add_argument("--top_k", type=int, default=5)
    parser.add_argument("--max_dialogues", type=int, default=300)
    parser.add_argument("--batch_size", type=int, default=8)
    parser.add_argument("--rag_batch_size", type=int, default=64)
    parser.add_argument("--no_response", action="store_true")
    parser.add_argument("--continue_on_error", action="store_true")
    args = parser.parse_args()

    rows = run_predict_state_round_batch(args)
    write_json(args.output_path, rows)
    print(f"Saved {len(rows)} dialogues to {args.output_path}")


if __name__ == "__main__":
    main()
