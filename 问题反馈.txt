def print_max_length(name, dataset):

    max_len = 0

    max_idx = -1

    for i, sample in enumerate(dataset):

        text = sample["instruction"] + sample["input"]

        if len(text) > max_len:

            max_len = len(text)

            max_idx = i

    print(f"{name}:")

    print(f"  max length = {max_len}")

    print(f"  sample id  = {max_idx}")
