#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
Evaluation script for the new static multi-turn prediction format.

Expected prediction sample format:
{
  "dialogue_id": 1,
  "dialogue_input": {
    "current_turn": "客服: ...",
    "dialogue_history": ["用户: ..."]
  },
  "golden_response": "客服: ...",
  "query": {"query": "..."},
  "retriever": {
    "selected_case_ids": ["KT...", ...],
    "evidence": "case_id: KT...\n..."
  },
  "trigger": {"should_trigger": true},
  "policy": {"dialogue_action": "ExplainedResponse", "case_id": "KT..."},
  "response": {"response": "..."},
  "latency": {"query_ms": 0, "retrieval_ms": 0, ...}
}

This script keeps the original metric set only:
1) Case recommendation turn metrics
2) All-turn recommendation timing metrics
3) Offset and dialogue-level success metrics
4) Internal diagnostic metrics
5) Latency metrics

It does not require external gold files because golden_response is embedded in each sample.
If turn_id is absent, it infers turn_id by the order of samples within each dialogue_id.
"""

from __future__ import annotations

import argparse
import json
import re
from collections import Counter, defaultdict
from pathlib import Path
from typing import Any, Dict, Iterable, List, Optional, Sequence, Set, Tuple


# =========================
# Constants
# =========================

KNOWLEDGE_POLICIES = {"CaseRecommendation", "ExplainedResponse"}
CASE_RECOMMEND_POLICY = "CaseRecommendation"

# The strict KT-pattern is the safest fallback for Huawei case ids.
KT_PATTERN = re.compile(r"\bKT\d{8}\b", re.IGNORECASE)
CASE_ID_FIELD_PATTERN = re.compile(
    r"(?:case[_\s-]*id|案例链接|案例ID|案例编号)\s*[:：]\s*\[?\s*([A-Za-z]{2}\d{8})\s*\]?",
    re.IGNORECASE,
)


# =========================
# Basic utilities
# =========================

def load_json_or_jsonl(path: str | Path) -> List[Dict[str, Any]]:
    """Load a JSON list, a single JSON object, or JSONL records."""
    path = Path(path)
    text = path.read_text(encoding="utf-8").strip()
    if not text:
        return []

    try:
        obj = json.loads(text)
        if isinstance(obj, list):
            return [x for x in obj if isinstance(x, dict)]
        if isinstance(obj, dict):
            # Support either a single sample or a wrapper like {"predictions": [...]}.
            for key in ("predictions", "data", "results", "items"):
                if isinstance(obj.get(key), list):
                    return [x for x in obj[key] if isinstance(x, dict)]
            return [obj]
    except json.JSONDecodeError:
        pass

    # JSONL fallback.
    records: List[Dict[str, Any]] = []
    for line in text.splitlines():
        line = line.strip()
        if not line:
            continue
        # tolerate trailing comma in copied JSON fragments
        if line.endswith(","):
            line = line[:-1]
        try:
            obj = json.loads(line)
            if isinstance(obj, dict):
                records.append(obj)
        except json.JSONDecodeError:
            continue
    if records:
        return records

    # Last-resort fallback for a copied list whose final item has a trailing comma.
    repaired = re.sub(r",\s*([}\]])", r"\1", text)
    try:
        obj = json.loads(repaired)
        if isinstance(obj, list):
            return [x for x in obj if isinstance(x, dict)]
        if isinstance(obj, dict):
            return [obj]
    except json.JSONDecodeError as exc:
        raise ValueError(f"Cannot parse prediction file as JSON/JSONL: {path}") from exc

    return []


def safe_div(num: float, den: float) -> float:
    return float(num) / float(den) if den else 0.0


def mean(values: Sequence[float]) -> float:
    return sum(values) / len(values) if values else 0.0


def prf(tp: int, fp: int, fn: int) -> Tuple[float, float, float]:
    p = safe_div(tp, tp + fp)
    r = safe_div(tp, tp + fn)
    f1 = safe_div(2 * p * r, p + r)
    return p, r, f1


def normalize_case_id(case_id: Any) -> str:
    if case_id is None:
        return ""
    s = str(case_id).strip().strip("[](){}'\"")
    return s.upper()


def normalize_dialogue_id(dialogue_id: Any) -> str:
    if dialogue_id is None:
        return ""
    return str(dialogue_id).strip()


def to_bool(value: Any) -> bool:
    if isinstance(value, bool):
        return value
    if value is None:
        return False
    if isinstance(value, (int, float)):
        return value != 0
    s = str(value).strip().lower()
    return s in {"true", "1", "yes", "y", "是", "需要", "trigger"}


def extract_case_ids(text: Any) -> Set[str]:
    """Extract case ids from golden response or free text."""
    if text is None:
        return set()
    s = str(text)
    ids: Set[str] = set()
    for m in CASE_ID_FIELD_PATTERN.finditer(s):
        cid = normalize_case_id(m.group(1))
        if cid:
            ids.add(cid)
    for m in KT_PATTERN.finditer(s):
        cid = normalize_case_id(m.group(0))
        if cid:
            ids.add(cid)
    return ids


def first_non_empty(*values: Any) -> Any:
    for v in values:
        if v is not None and v != "":
            return v
    return None


# =========================
# Field extraction for new format
# =========================

def get_dialogue_id(item: Dict[str, Any]) -> str:
    return normalize_dialogue_id(first_non_empty(item.get("dialogue_id"), item.get("dialog_id"), item.get("id")))


def get_turn_id(item: Dict[str, Any]) -> Optional[int]:
    for key in ("turn_id", "target_turn_id", "target_turn_index", "assistant_turn_id"):
        if key in item and item[key] is not None:
            try:
                return int(item[key])
            except (TypeError, ValueError):
                continue
    return None


def get_golden_response(item: Dict[str, Any]) -> str:
    # current_turn is a fallback only. golden_response is the true gold field.
    dialogue_input = item.get("dialogue_input") or {}
    return str(first_non_empty(item.get("golden_response"), dialogue_input.get("current_turn"), ""))


def get_gold_case_ids(item: Dict[str, Any]) -> Set[str]:
    return extract_case_ids(get_golden_response(item))


def get_query_text(item: Dict[str, Any]) -> str:
    q = item.get("query") or {}
    if isinstance(q, dict):
        return str(first_non_empty(q.get("query"), q.get("query_text"), "")).strip()
    return str(q).strip()


def get_selected_case_ids(item: Dict[str, Any]) -> List[str]:
    retriever = item.get("retriever") or {}
    if not isinstance(retriever, dict):
        return []

    selected = retriever.get("selected_case_ids")
    if isinstance(selected, list):
        ids = [normalize_case_id(x) for x in selected]
        return [x for x in ids if x]

    # Fallback 1: evidence text.
    evidence = retriever.get("evidence", "")
    ids = list(extract_case_ids(evidence))
    if ids:
        # Preserve evidence order if possible.
        ordered: List[str] = []
        seen: Set[str] = set()
        for m in KT_PATTERN.finditer(str(evidence)):
            cid = normalize_case_id(m.group(0))
            if cid and cid not in seen:
                ordered.append(cid)
                seen.add(cid)
        return ordered

    # Fallback 2: case_diagnostics.
    diagnostics = retriever.get("case_diagnostics")
    if isinstance(diagnostics, list):
        out: List[str] = []
        for c in diagnostics:
            if isinstance(c, dict):
                cid = normalize_case_id(c.get("case_id"))
                if cid:
                    out.append(cid)
        return out

    return []


def get_trigger_bool(item: Dict[str, Any]) -> bool:
    trigger = item.get("trigger") or {}
    if isinstance(trigger, dict) and "should_trigger" in trigger:
        return to_bool(trigger.get("should_trigger"))
    # If trigger is missing, use query existence as a weak fallback.
    return bool(get_query_text(item))


def get_policy_action(item: Dict[str, Any]) -> str:
    policy = item.get("policy") or {}
    if isinstance(policy, dict):
        return str(first_non_empty(policy.get("dialogue_action"), policy.get("label"), "")).strip()
    return ""


def get_policy_case_id(item: Dict[str, Any]) -> str:
    policy = item.get("policy") or {}
    if isinstance(policy, dict):
        return normalize_case_id(first_non_empty(policy.get("case_id"), policy.get("case"), ""))
    return ""


def get_pred_case_ids(item: Dict[str, Any]) -> Set[str]:
    cid = get_policy_case_id(item)
    return {cid} if cid else set()


def explicit_case_recommendation(item: Dict[str, Any]) -> bool:
    return get_policy_action(item) == CASE_RECOMMEND_POLICY and bool(get_policy_case_id(item))


def knowledge_use_prediction(item: Dict[str, Any]) -> bool:
    return get_policy_action(item) in KNOWLEDGE_POLICIES and bool(get_policy_case_id(item))


def knowledge_link_triggered(item: Dict[str, Any]) -> bool:
    return get_trigger_bool(item)


# =========================
# Preprocessing
# =========================

def assign_inferred_turn_ids(items: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    """
    Ensure each item has an internal _eval_turn_id.
    If turn_id is absent, infer it by input order within the same dialogue_id.
    """
    counters: Dict[str, int] = defaultdict(int)
    processed: List[Dict[str, Any]] = []

    for global_index, item in enumerate(items):
        copied = dict(item)
        did = get_dialogue_id(copied)
        if not did:
            did = f"__missing_dialogue_id_{global_index}"
            copied["dialogue_id"] = did

        explicit_turn_id = get_turn_id(copied)
        if explicit_turn_id is None:
            counters[did] += 1
            turn_id = counters[did]
        else:
            turn_id = explicit_turn_id
            counters[did] = max(counters[did], turn_id)

        copied["_eval_dialogue_id"] = did
        copied["_eval_turn_id"] = int(turn_id)
        copied["_eval_global_index"] = global_index
        processed.append(copied)

    return processed


def group_by_dialogue(items: Iterable[Dict[str, Any]]) -> Dict[str, List[Dict[str, Any]]]:
    grouped: Dict[str, List[Dict[str, Any]]] = defaultdict(list)
    for item in items:
        grouped[item["_eval_dialogue_id"]].append(item)
    for did in grouped:
        grouped[did].sort(key=lambda x: (x["_eval_turn_id"], x["_eval_global_index"]))
    return dict(grouped)


# =========================
# Metric group 1: case recommendation turn metrics
# =========================

def eval_case_recommendation_turns(items: List[Dict[str, Any]]) -> Dict[str, Any]:
    gold_case_items = [x for x in items if get_gold_case_ids(x)]
    gold_case_turn_count = len(gold_case_items)

    knowledge_trigger_hits = 0
    retrieval_eval_count = 0
    top1_hits = 0
    top3_hits = 0
    top5_hits = 0
    knowledge_policy_hits = 0
    case_selection_eval_count = 0
    case_selection_hits = 0
    final_case_hits = 0

    for item in gold_case_items:
        gold_ids = get_gold_case_ids(item)
        pred_ids = get_pred_case_ids(item)
        selected = get_selected_case_ids(item)
        action = get_policy_action(item)

        if knowledge_link_triggered(item):
            knowledge_trigger_hits += 1
            retrieval_eval_count += 1
            if gold_ids.intersection(selected[:1]):
                top1_hits += 1
            if gold_ids.intersection(selected[:3]):
                top3_hits += 1
            if gold_ids.intersection(selected[:5]):
                top5_hits += 1

        if action in KNOWLEDGE_POLICIES:
            knowledge_policy_hits += 1

        if pred_ids:
            case_selection_eval_count += 1
            if gold_ids.intersection(pred_ids):
                case_selection_hits += 1

        if gold_ids.intersection(pred_ids):
            final_case_hits += 1

    return {
        "gold_case_turn_count": gold_case_turn_count,
        "knowledge_link_trigger_rate": safe_div(knowledge_trigger_hits, gold_case_turn_count),
        "retrieval_eval_count": retrieval_eval_count,
        "top1_effective_case_recall": safe_div(top1_hits, retrieval_eval_count),
        "top3_effective_case_recall": safe_div(top3_hits, retrieval_eval_count),
        "top5_effective_case_recall": safe_div(top5_hits, retrieval_eval_count),
        "knowledge_policy_hit_rate": safe_div(knowledge_policy_hits, gold_case_turn_count),
        "case_selection_eval_count": case_selection_eval_count,
        "case_selection_accuracy": safe_div(case_selection_hits, case_selection_eval_count),
        "final_case_recommendation_accuracy": safe_div(final_case_hits, gold_case_turn_count),
    }


# =========================
# Metric group 2: all-turn timing metrics
# =========================

def eval_all_turn_timing(items: List[Dict[str, Any]]) -> Dict[str, Any]:
    case_rec_tp = case_rec_fp = case_rec_fn = 0
    knowledge_tp = knowledge_fp = knowledge_fn = 0
    early_count = 0
    missed_count = 0
    gold_positive_count = 0
    gold_negative_count = 0

    for item in items:
        gold_positive = bool(get_gold_case_ids(item))
        pred_has_case = bool(get_pred_case_ids(item))
        pred_case_rec = explicit_case_recommendation(item)
        pred_knowledge_use = knowledge_use_prediction(item)

        if gold_positive:
            gold_positive_count += 1
        else:
            gold_negative_count += 1

        # Explicit case recommendation timing: CaseRecommendation + case_id.
        if pred_case_rec and gold_positive:
            case_rec_tp += 1
        elif pred_case_rec and not gold_positive:
            case_rec_fp += 1
        elif (not pred_case_rec) and gold_positive:
            case_rec_fn += 1

        # Knowledge-use timing: {CaseRecommendation, ExplainedResponse} + case_id.
        if pred_knowledge_use and gold_positive:
            knowledge_tp += 1
        elif pred_knowledge_use and not gold_positive:
            knowledge_fp += 1
        elif (not pred_knowledge_use) and gold_positive:
            knowledge_fn += 1

        # Original early/missed recommendation definitions use case_id existence.
        if (not gold_positive) and pred_has_case:
            early_count += 1
        if gold_positive and (not pred_has_case):
            missed_count += 1

    case_p, case_r, case_f1 = prf(case_rec_tp, case_rec_fp, case_rec_fn)
    knowledge_p, knowledge_r, knowledge_f1 = prf(knowledge_tp, knowledge_fp, knowledge_fn)

    return {
        "all_turn_count": len(items),
        "gold_positive_turn_count": gold_positive_count,
        "gold_negative_turn_count": gold_negative_count,
        "case_recommend_timing_precision": case_p,
        "case_recommend_timing_recall": case_r,
        "case_recommend_timing_f1": case_f1,
        "case_recommend_timing_tp": case_rec_tp,
        "case_recommend_timing_fp": case_rec_fp,
        "case_recommend_timing_fn": case_rec_fn,
        "knowledge_use_timing_precision": knowledge_p,
        "knowledge_use_timing_recall": knowledge_r,
        "knowledge_use_timing_f1": knowledge_f1,
        "knowledge_use_timing_tp": knowledge_tp,
        "knowledge_use_timing_fp": knowledge_fp,
        "knowledge_use_timing_fn": knowledge_fn,
        "early_recommendation_rate": safe_div(early_count, gold_negative_count),
        "missed_recommendation_rate": safe_div(missed_count, gold_positive_count),
    }


# =========================
# Metric group 3: offset and dialogue success
# =========================

def eval_offset_and_dialog_success(grouped: Dict[str, List[Dict[str, Any]]]) -> Dict[str, Any]:
    gold_event_count = 0
    matched_offsets: List[int] = []
    dialog_success_eval_count = 0
    dialog_success_count = 0

    for did, turns in grouped.items():
        gold_events: List[Tuple[int, str]] = []
        pred_events: List[Tuple[int, str]] = []

        for item in turns:
            tid = int(item["_eval_turn_id"])
            for cid in get_gold_case_ids(item):
                gold_events.append((tid, cid))
            for cid in get_pred_case_ids(item):
                pred_events.append((tid, cid))

        if gold_events:
            dialog_success_eval_count += 1
            gold_case_set = {cid for _, cid in gold_events}
            pred_case_set = {cid for _, cid in pred_events}
            if gold_case_set.intersection(pred_case_set):
                dialog_success_count += 1

        for gold_tid, gold_cid in gold_events:
            gold_event_count += 1
            candidate_offsets = [pred_tid - gold_tid for pred_tid, pred_cid in pred_events if pred_cid == gold_cid]
            if candidate_offsets:
                best_offset = min(candidate_offsets, key=lambda x: (abs(x), x))
                matched_offsets.append(best_offset)

    matched_count = len(matched_offsets)
    exact = sum(1 for x in matched_offsets if x == 0)
    early = sum(1 for x in matched_offsets if x < 0)
    delayed = sum(1 for x in matched_offsets if x > 0)

    return {
        "recommendation_offset_eval_count": gold_event_count,
        "recommendation_offset_coverage": safe_div(matched_count, gold_event_count),
        "avg_abs_recommendation_offset": mean([abs(x) for x in matched_offsets]),
        "avg_signed_recommendation_offset": mean(matched_offsets),
        "exact_timing_ratio_among_matched": safe_div(exact, matched_count),
        "early_timing_ratio_among_matched": safe_div(early, matched_count),
        "delayed_timing_ratio_among_matched": safe_div(delayed, matched_count),
        "dialog_success_eval_count": dialog_success_eval_count,
        "dialog_level_final_success_rate": safe_div(dialog_success_count, dialog_success_eval_count),
    }


# =========================
# Metric group 4: internal diagnostics
# =========================

def eval_internal_diagnostics(items: List[Dict[str, Any]], grouped: Dict[str, List[Dict[str, Any]]]) -> Dict[str, Any]:
    pred_case_output_count = 0
    invalid_case_count = 0
    over_trigger_count = 0
    gold_negative_count = 0

    for item in items:
        gold_positive = bool(get_gold_case_ids(item))
        if not gold_positive:
            gold_negative_count += 1
            if knowledge_link_triggered(item):
                over_trigger_count += 1

        selected = set(get_selected_case_ids(item))
        for cid in get_pred_case_ids(item):
            pred_case_output_count += 1
            if cid not in selected:
                invalid_case_count += 1

    duplicate_pred_count = 0
    total_pred_case_events = 0
    state_switch_count = 0
    state_pair_count = 0

    for did, turns in grouped.items():
        case_counter: Counter[str] = Counter()
        states: List[bool] = []

        for item in turns:
            states.append(knowledge_link_triggered(item))
            for cid in get_pred_case_ids(item):
                case_counter[cid] += 1
                total_pred_case_events += 1

        duplicate_pred_count += sum(max(0, count - 1) for count in case_counter.values())

        for prev, cur in zip(states, states[1:]):
            state_pair_count += 1
            if prev != cur:
                state_switch_count += 1

    return {
        "pred_case_output_count": pred_case_output_count,
        "invalid_case_id_rate": safe_div(invalid_case_count, pred_case_output_count),
        "over_knowledge_trigger_rate": safe_div(over_trigger_count, gold_negative_count),
        "repeated_recommendation_rate": safe_div(duplicate_pred_count, total_pred_case_events),
        "state_jitter_rate": safe_div(state_switch_count, state_pair_count),
    }


# =========================
# Metric group 5: latency
# =========================

def eval_latency(items: List[Dict[str, Any]]) -> Dict[str, Any]:
    fields = ["query_ms", "retrieval_ms", "trigger_ms", "policy_ms", "response_ms", "total_ms"]
    values: Dict[str, List[float]] = {field: [] for field in fields}

    for item in items:
        latency = item.get("latency") or {}
        if not isinstance(latency, dict):
            continue
        for field in fields:
            v = latency.get(field)
            if isinstance(v, (int, float)):
                values[field].append(float(v))
            else:
                try:
                    values[field].append(float(v))
                except (TypeError, ValueError):
                    pass

    return {f"avg_{field}": mean(values[field]) for field in fields}


# =========================
# Main evaluation
# =========================

def evaluate(pred_path: str | Path, max_dialogues: Optional[int] = None) -> Dict[str, Any]:
    raw_items = load_json_or_jsonl(pred_path)
    items = assign_inferred_turn_ids(raw_items)

    if max_dialogues is not None and max_dialogues > 0:
        allowed_dialogues: Set[str] = set()
        filtered: List[Dict[str, Any]] = []
        for item in items:
            did = item["_eval_dialogue_id"]
            if did not in allowed_dialogues:
                if len(allowed_dialogues) >= max_dialogues:
                    continue
                allowed_dialogues.add(did)
            filtered.append(item)
        items = filtered

    grouped = group_by_dialogue(items)

    gold_turn_count = len(items)
    gold_event_count = sum(len(get_gold_case_ids(x)) for x in items)

    return {
        "basic_counts": {
            "pred_prefix_count": len(items),
            "dialogue_count": len(grouped),
            "gold_event_count": gold_event_count,
            "gold_turn_count": gold_turn_count,
        },
        "case_recommendation_turn_metrics": eval_case_recommendation_turns(items),
        "all_turn_timing_metrics": eval_all_turn_timing(items),
        "offset_and_dialog_success_metrics": eval_offset_and_dialog_success(grouped),
        "internal_diagnostic_metrics": eval_internal_diagnostics(items, grouped),
        "latency_metrics": eval_latency(items),
    }


def main() -> None:
    parser = argparse.ArgumentParser(description="Evaluate new static multi-turn prediction format.")
    parser.add_argument("--pred_path", required=True, help="Path to prediction JSON/JSONL file.")
    parser.add_argument("--output_path", default="", help="Optional path to save evaluation JSON.")
    parser.add_argument("--max_dialogues", type=int, default=0, help="Optional max dialogue count. 0 means all.")
    parser.add_argument("--indent", type=int, default=2, help="JSON output indent.")
    args = parser.parse_args()

    result = evaluate(
        pred_path=args.pred_path,
        max_dialogues=args.max_dialogues if args.max_dialogues > 0 else None,
    )

    text = json.dumps(result, ensure_ascii=False, indent=args.indent)
    print(text)

    if args.output_path:
        Path(args.output_path).write_text(text + "\n", encoding="utf-8")


if __name__ == "__main__":
    main()
