#!/bin/bash

mkdir -p logs

run_group_1() {
    export CUDA_VISIBLE_DEVICES=0,1,2,3
    export MASTER_ADDR=127.0.0.1
    export MASTER_PORT=29501

    echo "[$(date)] Start Q"
    llamafactory-cli train train_q.yaml \
        > logs/q.log 2>&1

    if [ $? -ne 0 ]; then
        echo "[$(date)] Q failed"
        return 1
    fi

    echo "[$(date)] Start T"
    llamafactory-cli train train_t.yaml \
        > logs/t.log 2>&1
}

run_group_2() {
    export CUDA_VISIBLE_DEVICES=4,5,6,7
    export MASTER_ADDR=127.0.0.1
    export MASTER_PORT=29502

    echo "[$(date)] Start P"
    llamafactory-cli train train_p.yaml \
        > logs/p.log 2>&1

    if [ $? -ne 0 ]; then
        echo "[$(date)] P failed"
        return 1
    fi

    echo "[$(date)] Start R"
    llamafactory-cli train train_r.yaml \
        > logs/r.log 2>&1
}

run_group_1 > logs/group1.log 2>&1 &
pid1=$!

run_group_2 > logs/group2.log 2>&1 &
pid2=$!

wait $pid1
status1=$?

wait $pid2
status2=$?

echo "Group 1 exit status: $status1"
echo "Group 2 exit status: $status2"

if [ $status1 -ne 0 ] || [ $status2 -ne 0 ]; then
    echo "Some training jobs failed."
    exit 1
fi

echo "All training jobs finished."
