from __future__ import annotations

import json
from pathlib import Path
from typing import Any, Dict, List

from .io_utils import read_prompt
from .schema import Pair, format_pairs


class PromptBank:
    def __init__(self, prompt_dir: str | Path):
        root = Path(prompt_dir)
        self.annotation_query = read_prompt(root / "annotation" / "query.prompt")
        self.annotation_trigger = read_prompt(root / "annotation" / "trigger.prompt")
        self.annotation_policy = read_prompt(root / "annotation" / "policy.prompt")
        self.prediction_query = read_prompt(root / "prediction" / "query.prompt")
        self.prediction_trigger = read_prompt(root / "prediction" / "trigger.prompt")
        self.prediction_policy = read_prompt(root / "prediction" / "policy.prompt")
        self.prediction_response = read_prompt(root / "prediction" / "response.prompt")


def _chat(system_prompt: str, user_content: str) -> List[Dict[str, str]]:
    return [
        {"role": "system", "content": system_prompt},
        {"role": "user", "content": user_content},
    ]


def build_annotation_query_messages(bank: PromptBank, history_pairs: List[Pair]) -> List[Dict[str, str]]:
    content = "【已完成历史对话】\n" + format_pairs(history_pairs)
    return _chat(bank.annotation_query, content)


def build_annotation_trigger_messages(
    bank: PromptBank,
    history_pairs: List[Pair],
    query_traj: List[Dict[str, Any]],
) -> List[Dict[str, str]]:
    content = (
        "【已完成历史对话】\n"
        + format_pairs(history_pairs)
        + "\n\n【Query Trajectory】\n"
        + json.dumps({"trajectory": query_traj}, ensure_ascii=False, indent=2)
    )
    return _chat(bank.annotation_trigger, content)


def build_annotation_policy_messages(
    bank: PromptBank,
    history_pairs: List[Pair],
    query_traj: List[Dict[str, Any]],
    trigger_traj: List[Dict[str, Any]],

        + json.dumps({"trajectory": trigger_traj}, ensure_ascii=False, indent=2)

    return _chat(bank.annotation_policy, content)


def build_prediction_query_messages(
    bank: PromptBank,
    history_pairs: List[Pair],
    historical_query: List[Dict[str, Any]],
    current_user: str,
    turn_id: int,
) -> List[Dict[str, str]]:
    content = (
        f"【当前 turn_id】\n{turn_id}\n\n"
        "【已完成历史对话（不包含当前用户输入）】\n"
        + format_pairs(history_pairs)
        + "\n\n【历史 Query Trajectory】\n"
        + json.dumps({"trajectory": historical_query}, ensure_ascii=False, indent=2)
        + "\n\n【当前用户输入】\n"
        + current_user
    )
    return _chat(bank.prediction_query, content)

    bank: PromptBank,

    historical_state: Dict[str, List[Dict[str, Any]]],

        + format_pairs(history_pairs)
        + "\n\n【历史状态轨迹】\n"
