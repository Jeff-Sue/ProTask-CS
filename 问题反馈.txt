#!/bin/bash

mkdir -p logs

############################################
# Group 1 (GPU 0-3)
############################################
(
export CUDA_VISIBLE_DEVICES=0,1,2,3

echo "Start Q..."
llamafactory-cli train train_q.yaml \
    > logs/q.log 2>&1

echo "Start T..."
llamafactory-cli train train_t.yaml \
    > logs/t.log 2>&1

echo "Group1 Finished."

) &


############################################
# Group 2 (GPU 4-7)
############################################
(
export CUDA_VISIBLE_DEVICES=4,5,6,7

echo "Start P..."
llamafactory-cli train train_p.yaml \
    > logs/p.log 2>&1

echo "Start R..."
llamafactory-cli train train_r.yaml \
    > logs/r.log 2>&1

echo "Group2 Finished."

) &


wait

echo "All Finished."
