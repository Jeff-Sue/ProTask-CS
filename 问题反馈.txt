from models.sft_model import SFTModel

MODEL_REGISTRY = {
    "query": SFTModel("/data/models/query_sft"),
    "trigger": SFTModel("/data/models/trigger_sft"),
    "policy": SFTModel("/data/models/policy_sft"),
    "response": SFTModel("/data/models/response_sft"),
}




from models.registry import MODEL_REGISTRY
from modules.query import run_query
from modules.trigger import run_trigger
from modules.policy import run_policy
from modules.response import run_response
from rag.retriever import retrieve_cases
from postprocess.case_filter import postprocess_cases


class Pipeline:

    def __init__(self, rag_fn):
        self.models = MODEL_REGISTRY
        self.rag_fn = rag_fn

    def run(self, dialogue):

        # 1. Query
        query = run_query(self.models["query"], dialogue)

        # 2. RAG
        cases = retrieve_cases(query, self.rag_fn)

        # 3. Postprocess
        cases = postprocess_cases(cases)

        # 4. Trigger
        trigger = run_trigger(self.models["trigger"], dialogue, cases)

        # 5. Policy
        policy = run_policy(self.models["policy"], dialogue, cases)

        # 6. Response
        response = run_response(
            self.models["response"],
            dialogue,
            policy,
            cases
        )

        return {
            "query": query,
            "cases": cases,
            "trigger": trigger,
            "policy": policy,
            "response": response
        }
