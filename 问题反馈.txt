def build_state_induction_input(
    memory: Dict[str, Any],
    user: str,
    recollection_state: Dict[str, Any],
) -> Dict[str, Any]:
    return {
        "current_user": user,
        "recollection": recollection_state,
        "previous_state": (
            memory["dialog_state"]["induced_state"]
        ),
        "current_issue": (
            memory["issue_state"]["current_issue"]
        ),
    }

def build_recollection_input(
    memory: Dict[str, Any],
    dialogue_history: str,
    user: str,
) -> Dict[str, Any]:
    return {
        "dialogue_history": dialogue_history,
        "current_user": user,
        "previous_state": (
            memory["dialog_state"]["induced_state"]
        ),
    }

def update_dialogue_induced_memory(
    memory: Dict[str, Any],
    state_output: Dict[str, Any],
) -> Dict[str, Any]:

    induced_state = state_output.get(
        "induced_state",
        {}
    )

    if isinstance(induced_state, dict):
        memory["dialog_state"]["induced_state"] = induced_state

    return memory
