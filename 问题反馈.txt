import json
import argparse
from pathlib import Path


def load_data(path):
    with open(path, "r", encoding="utf-8") as f:
        data = json.load(f)

    if isinstance(data, list):
        return data

    return data.get("data", data.get("dialogues", []))


def save_data(data, path):
    with open(path, "w", encoding="utf-8") as f:
        json.dump(data, f, ensure_ascii=False, indent=2)


def to_text(data):
    return json.dumps(data, ensure_ascii=False)


def empty_evidence():
    return {
        "selected_case_ids": [],
        "cases": []
    }


def get_response(turn, source):
    if source == "golden":
        return turn.get("golden_response", "")
    return turn.get("predicted_response", "")


def remove_reason(state):
    if not isinstance(state, dict):
        return state

    return {
        key: value
        for key, value in state.items()
        if key != "reason"
    }


def compact_evidence(evidence):
    return {
        "selected_case_ids": evidence.get("selected_case_ids", []),
        "cases": [
            {
                "case_id": case.get("case_id", ""),
                "title": case.get("title", ""),
                "keywords": case.get("keywords", "")
            }
            for case in evidence.get("cases", [])
        ]
    }


def build_history(previous_turns, response_source):
    history = []

    for turn in previous_turns:
        history.append({
            "turn_id": turn["turn_id"],
            "u": turn["u"],
            "q": remove_reason(turn["q"]),
            "t": remove_reason(turn["t"]),
            "e": compact_evidence(turn["e"]),
            "p": remove_reason(turn["p"]),
            "r": get_response(turn, response_source)
        })

    return history


def build_sample(instruction, model_input, output):
    return {
        "instruction": instruction,
        "input": to_text(model_input),
        "output": (
            to_text(output)
            if isinstance(output, (dict, list))
            else output
        )
    }


def build_datasets(
    dialogues,
    history_response_source="golden",
    response_target_source="predicted"
):
    q_data = []
    t_data = []
    p_data = []
    r_data = []

    for dialog in dialogues:
        previous_turns = []

        for turn in dialog["turns"]:
            history = build_history(
                previous_turns,
                history_response_source
            )

            active_evidence = (
                previous_turns[-1]["e"]
                if previous_turns
                else empty_evidence()
            )

            current_q = remove_reason(turn["q"])
            current_t = remove_reason(turn["t"])
            current_p = remove_reason(turn["p"])

            q_input = {
                "history_state": history,
                "active_evidence": active_evidence,
                "current_user": turn["u"]
            }

            q_data.append(
                build_sample(
                    "根据历史状态、active_evidence和当前用户输入生成query，只输出合法JSON。",
                    q_input,
                    current_q
                )
            )

            t_input = {
                "history_state": history,
                "active_evidence": active_evidence,
                "current_user": turn["u"],
                "current_query": current_q
            }

            t_data.append(
                build_sample(
                    "判断当前轮是否需要重新调用RAG，只输出合法JSON。",
                    t_input,
                    current_t
                )
            )

            p_input = {
                "history_state": history,
                "current_user": turn["u"],
                "current_query": current_q,
                "current_trigger": current_t,
                "current_evidence": turn["e"]
            }

            p_data.append(
                build_sample(
                    "根据当前状态和evidence选择客服策略，只输出合法JSON。",
                    p_input,
                    current_p
                )
            )

            r_input = {
                "history_state": history,
                "current_user": turn["u"],
                "current_query": current_q,
                "current_trigger": current_t,
                "current_evidence": turn["e"],
                "current_policy": current_p
            }

            response_target = get_response(
                turn,
                response_target_source
            )

            r_data.append(
                build_sample(
                    "根据当前状态、evidence和policy生成客服回复，只输出回复内容。",
                    r_input,
                    response_target
                )
            )

            previous_turns.append(turn)

    return q_data, t_data, p_data, r_data


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--input", required=True)
    parser.add_argument("--output_dir", default="data")
    parser.add_argument(
        "--history_response_source",
        choices=["golden", "predicted"],
        default="golden"
    )
    parser.add_argument(
        "--response_target_source",
        choices=["golden", "predicted"],
        default="predicted"
    )
    args = parser.parse_args()

    dialogues = load_data(args.input)

    q_data, t_data, p_data, r_data = build_datasets(
        dialogues,
        args.history_response_source,
        args.response_target_source
    )

    output_dir = Path(args.output_dir)
    output_dir.mkdir(parents=True, exist_ok=True)

    save_data(q_data, output_dir / "q_train.json")
    save_data(t_data, output_dir / "t_train.json")
    save_data(p_data, output_dir / "p_train.json")
    save_data(r_data, output_dir / "r_train.json")

    print(f"Q: {len(q_data)}")
    print(f"T: {len(t_data)}")
    print(f"P: {len(p_data)}")
    print(f"R: {len(r_data)}")


if __name__ == "__main__":
    main()
