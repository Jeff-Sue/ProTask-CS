def print_stage(round_idx: int, module: str) -> None:
    print(
        f"\n{'=' * 20} "
        f"第 {round_idx + 1} 轮｜{module} "
        f"{'=' * 20}",
        flush=True,
    )


for round_idx in tqdm(
    range(max_turns),
    desc="Round-wise turns",
):
    active = [
        runtime
        for runtime in runtimes
        if (
            runtime["error"] is None
            and round_idx < len(runtime["targets"])
        )
    ]

    if not active:
        continue

    print(
        f"\n\n{'#' * 60}\n"
        f"开始处理第 {round_idx + 1} 轮，"
        f"有效对话数：{len(active)}\n"
        f"{'#' * 60}",
        flush=True,
    )

    contexts = []

    for runtime in active:
        contexts.append(
            {
                "runtime": runtime,
                "target": runtime["targets"][round_idx],
                "history_state": build_history_state(
                    runtime["turn_states"],
                    args.history_response_source,
                ),
                "a_before": normalize_active_evidence(
                    runtime["active_evidence"]
                ),
            }
        )

    try:
        # Query Generator
        print_stage(round_idx, "Query Generator")

        q_steps = run_llm_json_batch(
            parallel_inference,
            [
                build_query_messages(
                    bank,
                    ctx["target"],
                    ctx["history_state"],
                    ctx["a_before"],
                )
                for ctx in contexts
            ],
            args.batch_size,
        )

        # RAG Trigger
        print_stage(round_idx, "RAG Trigger")

        t_steps = run_llm_json_batch(
            parallel_inference,
            [
                build_trigger_messages(
                    bank,
                    ctx["target"],
                    ctx["history_state"],
                    ctx["a_before"],
                    q_steps[i],
                )
                for i, ctx in enumerate(contexts)
            ],
            args.batch_size,
        )

        # Evidence Retrieval
        print_stage(round_idx, "Evidence Retrieval")

        evidence_results = decide_current_evidence_batch(
            batch_zefeng_rag=batch_zefeng_rag,
            targets=[
                ctx["target"]
                for ctx in contexts
            ],
            history_states=[
                ctx["history_state"]
                for ctx in contexts
            ],
            q_steps=q_steps,
            t_steps=t_steps,
            a_befores=[
                ctx["a_before"]
                for ctx in contexts
            ],
            rag_batch_size=args.rag_batch_size,
            top_k=args.top_k,
        )

        # Policy Predictor
        print_stage(round_idx, "Policy Predictor")

        p_steps = run_llm_json_batch(
            parallel_inference,
            [
                build_policy_messages(
                    bank,
                    ctx["target"],
                    ctx["history_state"],
                    evidence_results[i]["a_current"],
                    q_steps[i],
                    t_steps[i],
                    evidence_results[i]["e_content"],
                )
                for i, ctx in enumerate(contexts)
            ],
            args.batch_size,
        )

        # Response Generator
        print_stage(round_idx, "Response Generator")

        if args.no_response:
            responses = [""] * len(contexts)
            print(
                f"第 {round_idx + 1} 轮跳过回复生成。",
                flush=True,
            )
        else:
            responses = run_llm_response_batch(
                parallel_inference,
                [
                    build_response_messages(
                        bank,
                        ctx["target"],
                        ctx["history_state"],
                        evidence_results[i]["a_current"],
                        q_steps[i],
                        t_steps[i],
                        evidence_results[i]["e_content"],
                        p_steps[i],
                    )
                    for i, ctx in enumerate(contexts)
                ],
                args.batch_size,
            )

    except Exception as exc:
        if not args.continue_on_error:
            raise

        print(
            f"\n[第 {round_idx + 1} 轮运行失败] "
            f"{repr(exc)}",
            flush=True,
        )

        for ctx in contexts:
            ctx["runtime"]["error"] = repr(exc)

        continue

    # Runtime Update
    print_stage(round_idx, "State Update")

    for i, ctx in enumerate(contexts):
        runtime = ctx["runtime"]
        target = ctx["target"]
        evidence_result = evidence_results[i]

        runtime["turn_states"].append(
            {
                "turn_id": target.turn_id,
                "u": target.current_user,
                "q": q_steps[i],
                "t": t_steps[i],
                "e": evidence_result["e_short"],
                "p": p_steps[i],
                "predicted_response": responses[i],
                "golden_response": target.golden_response,
            }
        )

        runtime["active_evidence"] = (
            update_active_evidence_after_policy(
                a_current=evidence_result["a_current"],
                e_content=evidence_result["e_content"],
                trigger=t_steps[i],
                policy=p_steps[i],
            )
        )

    print(
        f"\n[第 {round_idx + 1} 轮完成] "
        f"已处理 {len(contexts)} 个对话。",
        flush=True,
    )
