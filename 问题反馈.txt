@dataclass
class TargetTurn:
    dialog_id: str
    turn_id: int
    history_pairs: List[Pair]
    current_user: str
    golden_response: str
    case_ids: List[str]


EMPTY_RETRIEVAL = {"selected_case_ids": [], "retrieved_cases": []}


def _empty_state() -> Dict[str, List[Dict[str, Any]]]:
    return {"query": [], "trigger": [], "policy": []}


def _state_slice(state: Dict[str, List[Dict[str, Any]]], n: int) -> Dict[str, List[Dict[str, Any]]]:
    return {
        "query": list(state.get("query", []))[:n],
        "trigger": list(state.get("trigger", []))[:n],
        "policy": list(state.get("policy", []))[:n],
    }


def _state_copy(state: Dict[str, List[Dict[str, Any]]]) -> Dict[str, List[Dict[str, Any]]]:
    return _state_slice(
        state,
        max(
            len(state.get("query", [])),
            len(state.get("trigger", [])),
            len(state.get("policy", [])),
        ),
    )


def _normalize_dialogues(data: Any) -> List[Dict[str, Any]]:
    """Support both list-of-dialogues and dict-of-dialogues.

    Example dict-of-dialogues:
    {
      "220599461": {"text": [{"用户": "..."}, {"客服": "..."}], "caseId": [...]}
    }
    """
    if isinstance(data, list):
        return data

    if isinstance(data, dict):
        for key in ("data", "dialogues", "items", "records"):
            if isinstance(data.get(key), list):
                return data[key]

        out: List[Dict[str, Any]] = []
        for dialog_id, sample in data.items():
            if isinstance(sample, dict):
                x = dict(sample)
                x.setdefault("dialog_id", str(dialog_id))
                out.append(x)
        return out

    raise ValueError(f"Unsupported dialogue data type: {type(data)}")


def load_dialogues(path: str | Path) -> List[Dict[str, Any]]:
    p = Path(path)
    text = p.read_text(encoding="utf-8").strip()
    if not text:
        return []

    if text[0] in "[{":
        return _normalize_dialogues(json.loads(text))

    rows = [json.loads(line) for line in text.splitlines() if line.strip()]
    return _normalize_dialogues(rows)


def _text_from_role_dict(item: Dict[str, Any]) -> Optional[Dict[str, str]]:
    # Native sample format: {"用户": "..."}, {"客服": "..."}
    for user_key in ("用户", "user", "human"):
        if user_key in item and str(item[user_key]).strip():
            return {"role": "user", "text": str(item[user_key]).strip()}

    for agent_key in ("客服", "assistant", "agent", "system"):
        if agent_key in item and str(item[agent_key]).strip():
            return {"role": "assistant", "text": str(item[agent_key]).strip()}

    role = str(item.get("role") or item.get("speaker") or item.get("from") or "").lower()
    text = str(
        item.get("text")
        or item.get("content")
        or item.get("value")
        or item.get("utterance")
        or ""
    ).strip()

    if not text:
        return None

    if role in {"user", "human", "客户", "用户"}:
        return {"role": "user", "text": text}

    if role in {"assistant", "客服", "agent", "system", "gpt"}:
        return {"role": "assistant", "text": text}

    if text.startswith(("用户:", "用户：")):
        return {"role": "user", "text": text}

    if text.startswith(("客服:", "客服：")):
        return {"role": "assistant", "text": text}

    return {"role": "user", "text": text}


def _messages(raw: Dict[str, Any]) -> List[Dict[str, str]]:
    source = None
    for key in ("text", "messages", "turns", "dialogue", "conversation", "dialogue_history"):
        if isinstance(raw.get(key), list):
            source = raw[key]
            break

    if source is None:
        raise ValueError(
            "Cannot find dialogue messages. Expected text/messages/turns/dialogue/conversation."
        )

    msgs: List[Dict[str, str]] = []

    for item in source:
        if isinstance(item, str):
            s = item.strip()
            if not s:
                continue

            if s.startswith(("用户:", "用户：")):
                msgs.append({"role": "user", "text": s})
            elif s.startswith(("客服:", "客服：")):
                msgs.append({"role": "assistant", "text": s})
            else:
                msgs.append({"role": "user", "text": s})

        elif isinstance(item, dict):
            msg = _text_from_role_dict(item)
            if msg is not None:
                msgs.append(msg)

    return msgs


def build_target_turns_no_batch(raw: Dict[str, Any]) -> List[TargetTurn]:
    dialog_id = str(raw.get("dialog_id") or raw.get("id") or raw.get("dialogue_id") or "")
    case_ids = [
        str(x).strip().upper()
        for x in raw.get("caseId", raw.get("case_ids", []))
        if str(x).strip()
    ]

    completed: List[Pair] = []
    user_buffer: List[str] = []
    targets: List[TargetTurn] = []
    turn_id = 0

    for msg in _messages(raw):
        if msg["role"] == "user":
            user_buffer.append(msg["text"])

        elif msg["role"] == "assistant":
            if not user_buffer:
                continue

            current_user = "\n".join(user_buffer).strip()
            turn_id += 1

            targets.append(
                TargetTurn(
                    dialog_id=dialog_id,
                    turn_id=turn_id,
                    history_pairs=list(completed),
                    current_user=current_user,
                    golden_response=msg["text"],
                    case_ids=case_ids,
                )
            )

            completed.append(Pair(user=current_user, assistant=msg["text"]))
            user_buffer = []

    return targets


def _unwrap_llm_response(raw: Any) -> str:
    """Accept parallel_inference outputs: str, {response: str}, or nested shapes."""
    if isinstance(raw, dict):
        for key in ("response", "text", "content", "output"):
            if key in raw:
                return str(raw[key])
        return json.dumps(raw, ensure_ascii=False)

    return str(raw)


def call_llm_once(parallel_inference, messages: List[Dict[str, str]]) -> str:
    raw = run_llm_batch(parallel_inference, [messages], batch_size=1)

    if not isinstance(raw, list) or not raw:
        raise ValueError(f"LLM returned empty output: {raw}")

    return _unwrap_llm_response(raw[0])


def _needs_retrieval(query_step: Dict[str, Any]) -> bool:
    return query_step.get("query_mode") in {"rewrite", "refine", "reuse"} and bool(
        query_step.get("query_text")
    )


def retrieve_once(
    batch_zefeng_rag,
    target: TargetTurn,
    query_step: Dict[str, Any],
    top_k: int,
) -> Dict[str, Any]:
    if not _needs_retrieval(query_step):
        return dict(EMPTY_RETRIEVAL)

    history_text = format_pairs(target.history_pairs) + "\n" + target.current_user
    raw_ret = run_rag_batch(batch_zefeng_rag, [history_text], [query_step["query_text"]])

    if isinstance(raw_ret, list):
        raw_ret = raw_ret[0] if raw_ret else None

    return normalize_retrieval(raw_ret, top_k=top_k)


def load_golden_annotations(path: str | Path) -> Dict[str, Dict[str, List[Dict[str, Any]]]]:
    """Load completed golden annotations and index by dialog_id.

    Supported row shapes:
    1) {dialog_id: {golden_state: {query/trigger/policy}}}
    2) {dialog_id: {query: [...], trigger: [...], policy: [...]}}
    3) JSONL rows with dialog_id and golden_state/state.
    """
    p = Path(path)
    text = p.read_text(encoding="utf-8").strip()

    if not text:
        return {}

    if text[0] in "[{":
        obj = json.loads(text)
    else:
        obj = [json.loads(line) for line in text.splitlines() if line.strip()]

    index: Dict[str, Dict[str, List[Dict[str, Any]]]] = {}

    def put(dialog_id: str, payload: Any) -> None:
        if not dialog_id or not isinstance(payload, dict):
            return

        state = (
            payload.get("golden_state")
            or payload.get("state")
            or payload.get("golden_annotation")
            or payload
        )

        if not isinstance(state, dict):
            return

        index[str(dialog_id)] = {
            "query": list(state.get("query", state.get("query_trajectory", [])) or []),
            "trigger": list(state.get("trigger", state.get("trigger_trajectory", [])) or []),
            "policy": list(state.get("policy", state.get("policy_trajectory", [])) or []),
        }

    if isinstance(obj, dict):
        # Dict keyed by dialog id.
        if "dialog_id" in obj:
            put(str(obj.get("dialog_id")), obj)
        else:
            for did, payload in obj.items():
                put(str(did), payload)

    elif isinstance(obj, list):
        for row in obj:
            if isinstance(row, dict):
                put(
                    str(row.get("dialog_id") or row.get("id") or row.get("dialogue_id") or ""),
                    row,
                )

    return index


def _gold_state_for_turn(
    golden_index: Dict[str, Dict[str, List[Dict[str, Any]]]],
    dialog_id: str,
    turn_id: int,
) -> Dict[str, List[Dict[str, Any]]]:
    full = golden_index.get(str(dialog_id), _empty_state())
    return _state_slice(full, turn_id - 1)


def _make_row(
    target: TargetTurn,
    state_source: str,
    historical_state: Dict[str, List[Dict[str, Any]]],
    query_step: Dict[str, Any],
    retrieval: Dict[str, Any],
    trigger_step: Dict[str, Any],
    policy_step: Dict[str, Any],
    response: str,
    latency_ms: Dict[str, int],
) -> Dict[str, Any]:
    return {
        "dialog_id": target.dialog_id,
        "turn_id": target.turn_id,
        "state_source": state_source,
        "current_user_input": target.current_user,
        "golden_response": target.golden_response,
        "gold_case_ids": extract_case_ids(target.golden_response) or target.case_ids,
        "historical_state": _state_copy(historical_state),
        "current": {
            "query": query_step,
            "retrieval": retrieval,
            "trigger": trigger_step,
            "policy": policy_step,
            "response": response,
        },
        "latency_ms": latency_ms,
    }


def predict_one_turn(
    target: TargetTurn,
    historical_state: Dict[str, List[Dict[str, Any]]],
    bank: PromptBank,
    parallel_inference,
    batch_zefeng_rag,
    top_k: int,
    generate_response: bool = True,
) -> Dict[str, Any]:
    """Run one current turn in the system forward order: Query -> RAG -> Trigger -> Policy -> Response.

    This is PE/evaluation inference, not golden annotation. Golden annotation may be produced in
    reverse explanatory order, but system prediction should still be forward.
    """
    start = time.time()
    latency: Dict[str, int] = {}

    # 1. Query
    t0 = time.time()
    q_msg = build_prediction_query_messages(
        bank,
        target.history_pairs,
        historical_state.get("query", []),
        target.current_user,
        target.turn_id,
    )
    q_raw = call_llm_once(parallel_inference, q_msg)
    q_step = parse_step(q_raw, "query", target.turn_id)
    latency["query"] = int((time.time() - t0) * 1000)

    # 2. RAG
    t0 = time.time()
    retrieval = retrieve_once(batch_zefeng_rag, target, q_step, top_k=top_k)
    latency["retrieval"] = int((time.time() - t0) * 1000)

    # 3. Trigger
    t0 = time.time()
    trig_msg = build_prediction_trigger_messages(
        bank,
        target.history_pairs,
        historical_state,
        target.current_user,
        q_step,
        retrieval,
        target.turn_id,
    )
    trig_raw = call_llm_once(parallel_inference, trig_msg)
    trig_step = parse_step(trig_raw, "trigger", target.turn_id)
    latency["trigger"] = int((time.time() - t0) * 1000)

    # 4. Policy
    t0 = time.time()
    pol_msg = build_prediction_policy_messages(
        bank,
        target.history_pairs,
        historical_state,
        target.current_user,
        q_step,
        retrieval,
        trig_step,
        target.turn_id,
    )
    pol_raw = call_llm_once(parallel_inference, pol_msg)
    pol_step = parse_step(pol_raw, "policy", target.turn_id)
    latency["policy"] = int((time.time() - t0) * 1000)

    # 5. Response
    response = ""
    if generate_response:
        t0 = time.time()
        resp_msg = build_response_messages(
            bank,
            target.history_pairs,
            target.current_user,
            retrieval,
            trig_step,
            pol_step,
        )
        resp_raw = call_llm_once(parallel_inference, resp_msg)

        try:
            response = normalize_response(parse_json_obj(resp_raw))
        except Exception:
            response = normalize_response(resp_raw)

        latency["response"] = int((time.time() - t0) * 1000)
    else:
        latency["response"] = 0

    latency["total"] = int((time.time() - start) * 1000)

    return {
        "query": q_step,
        "retrieval": retrieval,
        "trigger": trig_step,
        "policy": pol_step,
        "response": response,
        "latency_ms": latency,
    }


def run_gold_state_pe_no_batch(args) -> List[Dict[str, Any]]:
    if not args.golden_annotation_path:
        raise ValueError("--golden_annotation_path is required when --state_source gold_state")

    raw_dialogues = load_dialogues(args.data_path)[: args.max_dialogues]
    golden_index = load_golden_annotations(args.golden_annotation_path)

    bank = PromptBank(args.prompt_dir)
    parallel_inference = get_parallel_inference(args.project_root)
    batch_zefeng_rag = get_batch_rag(args.project_root)

    rows: List[Dict[str, Any]] = []

    for raw in tqdm(raw_dialogues, desc="Gold-state dialogues"):
        targets = build_target_turns_no_batch(raw)

        for target in targets:
            hist_state = _gold_state_for_turn(golden_index, target.dialog_id, target.turn_id)

            pred = predict_one_turn(
                target,
                hist_state,
                bank,
                parallel_inference,
                batch_zefeng_rag,
                top_k=args.top_k,
                generate_response=not args.no_response,
            )

            rows.append(
                _make_row(
                    target,
                    "gold_state",
                    hist_state,
                    pred["query"],
                    pred["retrieval"],
                    pred["trigger"],
                    pred["policy"],
                    pred["response"],
                    pred["latency_ms"],
                )
            )

    return rows


def run_pred_state_pe_no_batch(args) -> List[Dict[str, Any]]:
    raw_dialogues = load_dialogues(args.data_path)[: args.max_dialogues]

    bank = PromptBank(args.prompt_dir)
    parallel_inference = get_parallel_inference(args.project_root)
    batch_zefeng_rag = get_batch_rag(args.project_root)

    rows: List[Dict[str, Any]] = []

    for raw in tqdm(raw_dialogues, desc="Pred-state dialogues"):
        targets = build_target_turns_no_batch(raw)
        state = _empty_state()

        for target in targets:
            hist_state = _state_copy(state)

            pred = predict_one_turn(
                target,
                hist_state,
                bank,
                parallel_inference,
                batch_zefeng_rag,
                top_k=args.top_k,
                generate_response=not args.no_response,
            )

            rows.append(
                _make_row(
                    target,
                    "pred_state",
                    hist_state,
                    pred["query"],
                    pred["retrieval"],
                    pred["trigger"],
                    pred["policy"],
                    pred["response"],
                    pred["latency_ms"],
                )
            )

            # Pred-state: next turn sees the model's own predicted state.
            state["query"].append(pred["query"])
            state["trigger"].append(pred["trigger"])
            state["policy"].append(pred["policy"])

    return rows


def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--data_path", required=True)
    parser.add_argument("--output_path", required=True)
    parser.add_argument(
        "--state_source",
        choices=["gold_state", "pred_state"],
        default="gold_state",
    )
    parser.add_argument("--golden_annotation_path", default="")
    parser.add_argument("--prompt_dir", default="prompts")
    parser.add_argument("--project_root", default=".")
    parser.add_argument("--top_k", type=int, default=5)
    parser.add_argument("--max_dialogues", type=int, default=300)
    parser.add_argument(
        "--jsonl",
        action="store_true",
        help="Write output as JSONL instead of JSON list.",
    )
    parser.add_argument(
        "--no_response",
        action="store_true",
        help="Skip response generation; only output query/retrieval/trigger/policy.",
    )

    args = parser.parse_args()

    if args.state_source == "gold_state":
        rows = run_gold_state_pe_no_batch(args)
    else:
        rows = run_pred_state_pe_no_batch(args)

    if args.jsonl:
        write_jsonl(args.output_path, rows)
    else:
        write_json(args.output_path, rows)

    print(f"Saved {len(rows)} rows to {args.output_path}")


if __name__ == "__main__":
    main()
