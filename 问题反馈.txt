from pathlib import Path

import torch
from peft import PeftModel
from transformers import AutoModelForCausalLM, AutoTokenizer


BASE_MODEL = "/data/MODEL_ZOO/Qwen/Qwen3-8B"

LORAS = {
    "query_generator": "/path/to/query_generator",
    "rag_trigger": "/path/to/rag_trigger",
    "policy_predictor": "/path/to/policy_predictor",
    "response_generator": "/path/to/response_generator",
}

OUTPUT_ROOT = Path("/data/MODEL_ZOO/Qwen3-8B-Merged")


def merge_lora(name: str, lora_path: str) -> None:
    output_path = OUTPUT_ROOT / name
    output_path.mkdir(parents=True, exist_ok=True)

    print(f"[Merge] {name}")

    base_model = AutoModelForCausalLM.from_pretrained(
        BASE_MODEL,
        torch_dtype=torch.float16,
        device_map="cpu",
        low_cpu_mem_usage=True,
        trust_remote_code=True,
    )

    model = PeftModel.from_pretrained(
        base_model,
        lora_path,
        device_map="cpu",
    )

    model = model.merge_and_unload()

    model.save_pretrained(
        output_path,
        safe_serialization=True,
        max_shard_size="5GB",
    )

    tokenizer = AutoTokenizer.from_pretrained(
        BASE_MODEL,
        trust_remote_code=True,
    )

    tokenizer.save_pretrained(output_path)

    del model
    del base_model

    print(f"[Done] {output_path}")


if __name__ == "__main__":
    for model_name, adapter_path in LORAS.items():
        merge_lora(model_name, adapter_path)
