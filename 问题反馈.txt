from __future__ import annotations

import argparse
import json
import math
import re
from collections import Counter, defaultdict
from typing import Any, Dict, List, Tuple

from .io_utils import load_json_or_jsonl, write_json
from .schema import extract_case_ids

KNOWLEDGE_LABELS = {"ExplainedResponse", "CaseRecommendation"}
CLARIFY_LABELS = {"AskClarification", "AskMissingSlot"}
UNRESOLVED_LABELS = {"ProcessAcknowledgement", "Handoff"}


def safe_div(a: float, b: float) -> float:
    return float(a) / float(b) if b else 0.0


def f1(p: float, r: float) -> float:
    return safe_div(2 * p * r, p + r)


def get_current(row: Dict[str, Any]) -> Dict[str, Any]:
    return row.get("current", {}) if isinstance(row.get("current"), dict) else {}


def get_gold_case_ids(row: Dict[str, Any]) -> List[str]:
    ids = row.get("gold_case_ids")
    if isinstance(ids, list):
        return [str(x).upper() for x in ids if str(x).strip()]
    return extract_case_ids(row.get("golden_response", ""))


def get_retrieved_ids(row: Dict[str, Any]) -> List[str]:
    retrieval = get_current(row).get("retrieval", {}) or {}
    ids = retrieval.get("selected_case_ids")
    if isinstance(ids, list):
        return [str(x).upper() for x in ids if str(x).strip()]
    cases = retrieval.get("retrieved_cases", [])
    out = []
    if isinstance(cases, list):
        for c in cases:
            if isinstance(c, dict):
                cid = str(c.get("case_id") or "").upper().strip()
                if cid:
                    out.append(cid)
    return out


def get_query(row: Dict[str, Any]) -> Dict[str, Any]:
    return get_current(row).get("query", {}) or {}


def get_trigger(row: Dict[str, Any]) -> Dict[str, Any]:
    return get_current(row).get("trigger", {}) or {}


def get_policy(row: Dict[str, Any]) -> Dict[str, Any]:
    return get_current(row).get("policy", {}) or {}


def call_positive(row: Dict[str, Any]) -> bool:
    q = get_query(row)
    trig = get_trigger(row)
    mode = q.get("query_mode")
    trigger = bool(trig.get("trigger"))
    return trigger or mode in {"rewrite", "refine", "reuse"}


def use_positive(row: Dict[str, Any]) -> bool:
    p = get_policy(row)
    return p.get("label") in KNOWLEDGE_LABELS and bool(str(p.get("case_id") or "").strip())


def pred_case(row: Dict[str, Any]) -> str:
    return str(get_policy(row).get("case_id") or "").strip().upper()


def gold_is_clarify(text: str) -> bool:
    if not text:
        return False
    patterns = ["吗", "是否", "请问", "麻烦.*确认", "能否", "是不是", "哪", "什么", "提供", "补充", "截图", "报错", "具体"]
    return any(re.search(p, text) for p in patterns)


def gold_is_unresolved(text: str) -> bool:
    if not text:
        return False
    patterns = ["稍等", "处理中", "已反馈", "提交工单", "转.*处理", "联系.*处理", "人工", "后台", "后续", "确认后", "我这边.*看"]
    return any(re.search(p, text) for p in patterns)


def confusion(rows: List[Dict[str, Any]], pred_fn, gold_fn) -> Dict[str, int]:
    tp = fp = fn = tn = 0
    for r in rows:
        pred = bool(pred_fn(r))
        gold = bool(gold_fn(r))
        if pred and gold:
            tp += 1
        elif pred and not gold:
            fp += 1
        elif not pred and gold:
            fn += 1
        else:
            tn += 1
    return {"tp": tp, "fp": fp, "fn": fn, "tn": tn}


def greedy_soft_match(pred_events: List[Tuple[int, str]], gold_events: List[Tuple[int, str]], lam: float) -> float:
    candidates = []
    for pi, (tp, cp) in enumerate(pred_events):
        for gi, (tg, cg) in enumerate(gold_events):
            if cp == cg and cp:
                candidates.append((math.exp(-lam * abs(tp - tg)), pi, gi))
    candidates.sort(reverse=True)
    used_p, used_g = set(), set()
    score = 0.0
    for s, pi, gi in candidates:
        if pi not in used_p and gi not in used_g:
            score += s
            used_p.add(pi)
            used_g.add(gi)
    return score


def evaluate(rows: List[Dict[str, Any]], lam: float = 0.5) -> Dict[str, Any]:
    total = len(rows)
    gold_pos = [r for r in rows if get_gold_case_ids(r)]
    gold_neg = [r for r in rows if not get_gold_case_ids(r)]

    # A. Static single-turn case-grounded capability.
    a_den = len(gold_pos)
    call_on_gold = sum(call_positive(r) for r in gold_pos)
    retrieval_eval = [r for r in gold_pos if call_positive(r)]

    def topk_hit(r, k: int) -> bool:
        gold = set(get_gold_case_ids(r))
        got = set(get_retrieved_ids(r)[:k])
        return bool(gold & got)

    top1 = sum(topk_hit(r, 1) for r in retrieval_eval)
    top3 = sum(topk_hit(r, 3) for r in retrieval_eval)
    top5 = sum(topk_hit(r, 5) for r in retrieval_eval)
    top5_rows = [r for r in gold_pos if topk_hit(r, 5)]
    policy_hit = sum(use_positive(r) for r in top5_rows)
    selection_rows = [r for r in top5_rows if use_positive(r)]
    selection_hit = sum(pred_case(r) in set(get_gold_case_ids(r)) for r in selection_rows)
    final_hit = sum(pred_case(r) in set(get_gold_case_ids(r)) for r in gold_pos)

    # B1 call timing.
    call_cm = confusion(rows, call_positive, lambda r: bool(get_gold_case_ids(r)))
    call_p = safe_div(call_cm["tp"], call_cm["tp"] + call_cm["fp"])
    call_r = safe_div(call_cm["tp"], call_cm["tp"] + call_cm["fn"])

    # B2 use timing.
    use_cm = confusion(rows, use_positive, lambda r: bool(get_gold_case_ids(r)))
    use_p = safe_div(use_cm["tp"], use_cm["tp"] + use_cm["fp"])
    use_r = safe_div(use_cm["tp"], use_cm["tp"] + use_cm["fn"])

    # B3 post-retrieval policy distribution/correctness.
    call_rows = [r for r in rows if call_positive(r)]
    post_use = [r for r in call_rows if use_positive(r)]
    post_clarify = [r for r in call_rows if get_policy(r).get("label") in CLARIFY_LABELS]
    post_unresolved = [r for r in call_rows if get_policy(r).get("label") in UNRESOLVED_LABELS]
    post_use_correct = sum(pred_case(r) in set(get_gold_case_ids(r)) for r in post_use)
    post_clarify_correct = sum((not get_gold_case_ids(r)) and gold_is_clarify(r.get("golden_response", "")) for r in post_clarify)
    post_unresolved_correct = sum(gold_is_unresolved(r.get("golden_response", "")) for r in post_unresolved)

    # B4 events/dialog-level static event metrics.
    by_dialog: Dict[str, List[Dict[str, Any]]] = defaultdict(list)
    for r in rows:
        by_dialog[str(r.get("dialog_id", ""))].append(r)

    matched_offsets = []
    signed_offsets = []
    repeated_dialogs = 0
    pred_event_dialogs = 0
    soft_tp_total = 0.0
    pred_event_total = 0
    gold_event_total = 0
    dialog_success = 0
    dialog_with_gold = 0

    for did, ds in by_dialog.items():
        ds = sorted(ds, key=lambda x: int(x.get("turn_id") or 0))
        pred_events: List[Tuple[int, str]] = []
        gold_events: List[Tuple[int, str]] = []
        for r in ds:
            t = int(r.get("turn_id") or 0)
            for cid in get_gold_case_ids(r):
                gold_events.append((t, cid))
            if use_positive(r):
                cid = pred_case(r)
                if cid:
                    pred_events.append((t, cid))
        if gold_events:
            dialog_with_gold += 1
            if set(c for _, c in pred_events) & set(c for _, c in gold_events):
                dialog_success += 1
        if pred_events:
            pred_event_dialogs += 1
            counts = Counter(c for _, c in pred_events)
            if any(v > 1 for v in counts.values()):
                repeated_dialogs += 1

        for tg, cg in gold_events:
            same = [(abs(tp - tg), tp - tg) for tp, cp in pred_events if cp == cg]
            if same:
                abs_d, signed_d = min(same, key=lambda x: x[0])
                matched_offsets.append(abs_d)
                signed_offsets.append(signed_d)

        soft_tp_total += greedy_soft_match(pred_events, gold_events, lam)
        pred_event_total += len(pred_events)
        gold_event_total += len(gold_events)

    soft_p = safe_div(soft_tp_total, pred_event_total)
    soft_r = safe_div(soft_tp_total, gold_event_total)

    # B5 diagnostics.
    invalid_den = [r for r in rows if use_positive(r)]
    invalid_num = sum(pred_case(r) and pred_case(r) not in set(get_retrieved_ids(r)) for r in invalid_den)
    jitter_num = 0
    jitter_den = 0
    for ds in by_dialog.values():
        ds = sorted(ds, key=lambda x: int(x.get("turn_id") or 0))
        calls = [call_positive(r) for r in ds]
        for i in range(1, len(calls)):
            jitter_den += 1
            if calls[i] != calls[i - 1]:
                jitter_num += 1

    # Latency.
    latency_keys = ["query", "retrieval", "trigger", "policy", "response", "total"]
    latency = {}
    for k in latency_keys:
        vals = [r.get("latency_ms", {}).get(k) for r in rows if isinstance(r.get("latency_ms", {}).get(k), (int, float))]
        latency[f"avg_{k}_ms"] = safe_div(sum(vals), len(vals))

    return {
        "basic_counts": {
            "turn_count": total,
            "dialog_count": len(by_dialog),
            "gold_case_turn_count": len(gold_pos),
            "gold_non_case_turn_count": len(gold_neg),
        },
        "static_single_turn_case_capability": {
            "knowledge_link_trigger_rate": safe_div(call_on_gold, a_den),
            "retrieval_eval_count": len(retrieval_eval),
            "top1_effective_case_recall": safe_div(top1, len(retrieval_eval)),
            "top3_effective_case_recall": safe_div(top3, len(retrieval_eval)),
            "top5_effective_case_recall": safe_div(top5, len(retrieval_eval)),
            "knowledge_policy_hit_rate": safe_div(policy_hit, len(top5_rows)),
            "case_selection_accuracy": safe_div(selection_hit, len(selection_rows)),
            "final_case_hit_rate": safe_div(final_hit, a_den),
        },
        "knowledge_call_timing": {
            "precision": call_p,
            "recall": call_r,
            "f1": f1(call_p, call_r),
            **call_cm,
            "over_trigger_rate": safe_div(call_cm["fp"], call_cm["fp"] + call_cm["tn"]),
            "missed_trigger_rate": safe_div(call_cm["fn"], call_cm["tp"] + call_cm["fn"]),
        },
        "knowledge_use_timing": {
            "precision": use_p,
            "recall": use_r,
            "f1": f1(use_p, use_r),
            **use_cm,
            "early_use_rate": safe_div(use_cm["fp"], use_cm["fp"] + use_cm["tn"]),
            "missed_use_rate": safe_div(use_cm["fn"], use_cm["tp"] + use_cm["fn"]),
        },
        "post_retrieval_policy_distribution": {
            "post_retrieval_turn_count": len(call_rows),
            "post_retrieval_use_rate": safe_div(len(post_use), len(call_rows)),
            "post_retrieval_use_accuracy": safe_div(post_use_correct, len(post_use)),
            "post_retrieval_clarify_rate": safe_div(len(post_clarify), len(call_rows)),
            "post_retrieval_clarify_accuracy_heuristic": safe_div(post_clarify_correct, len(post_clarify)),
            "post_retrieval_unresolved_rate": safe_div(len(post_unresolved), len(call_rows)),
            "post_retrieval_unresolved_success_rate_heuristic": safe_div(post_unresolved_correct, len(post_unresolved)),
        },
        "case_event_metrics_static": {
            "avg_abs_recommend_offset": safe_div(sum(matched_offsets), len(matched_offsets)),
            "avg_signed_recommend_offset": safe_div(sum(signed_offsets), len(signed_offsets)),
            "matched_event_count": len(matched_offsets),
            "soft_precision": soft_p,
            "soft_recall": soft_r,
            "soft_f1": f1(soft_p, soft_r),
            "repeated_recommendation_rate": safe_div(repeated_dialogs, pred_event_dialogs),
            "oracle_history_dialog_success_rate": safe_div(dialog_success, dialog_with_gold),
        },
        "internal_diagnostics": {
            "invalid_case_id_rate": safe_div(invalid_num, len(invalid_den)),
            "state_jitter_rate": safe_div(jitter_num, jitter_den),
        },
        "latency_metrics": latency,
    }


def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--pred_path", required=True)
    parser.add_argument("--output_path", required=True)
    parser.add_argument("--lambda_value", type=float, default=0.5)
    args = parser.parse_args()
    rows = load_json_or_jsonl(args.pred_path)
    result = evaluate(rows, lam=args.lambda_value)
    write_json(args.output_path, result)
    print(json.dumps(result, ensure_ascii=False, indent=2))


if __name__ == "__main__":
    main()
