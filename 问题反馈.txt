from __future__ import annotations

import json
import re
from typing import Any, Dict, List

try:
    from json_repair import repair_json
except Exception:  # pragma: no cover
    repair_json = None


QUERY_MODES = {"none", "reuse", "refine", "rewrite"}
POLICY_LABELS = {
    "AskMissingSlot",
    "AskClarification",
    "ExplainedResponse",
    "CaseRecommendation",
    "Handoff",
    "ProcessAcknowledgement",
}
EVIDENCE_MODES = {"no_evidence_needed", "reuse_existing_evidence", "retrieve_new_evidence"}
EVIDENCE_STATUS = {"not_needed", "sufficient", "insufficient", "uncertain"}
EVIDENCE_BASIS = {"none", "candidate_case", "previous_evidence", "inferred_from_history"}


def _extract_json_text(text: str) -> str:
    text = str(text or "").strip()
    if text.startswith("```"):
        text = re.sub(r"^```(?:json)?", "", text, flags=re.I).strip()
        text = re.sub(r"```$", "", text).strip()
    start_obj, start_arr = text.find("{"), text.find("[")
    starts = [x for x in (start_obj, start_arr) if x >= 0]
    if not starts:
        return text
    start = min(starts)
    end_obj, end_arr = text.rfind("}"), text.rfind("]")
    end = max(end_obj, end_arr)
    if end >= start:
        return text[start:end + 1]
    return text[start:]


def parse_json_obj(text: str) -> Dict[str, Any]:
    raw = _extract_json_text(text)
    try:
        obj = json.loads(raw)
    except Exception:
        if repair_json is None:
            raise
        obj = json.loads(repair_json(raw))
    if isinstance(obj, list):
        return {"trajectory": obj}
    if not isinstance(obj, dict):
        raise ValueError(f"Expected JSON object, got {type(obj)}")
    return obj


def _as_list(x: Any) -> List[str]:
    if x is None:
        return []
    if isinstance(x, list):
        return [str(v).strip() for v in x if str(v).strip()]
    if isinstance(x, str):
        return [x.strip()] if x.strip() else []
    return [str(x).strip()] if str(x).strip() else []


def normalize_query_step(obj: Dict[str, Any], turn_id: int) -> Dict[str, Any]:
    mode = str(obj.get("query_mode") or "none").strip()
    if mode not in QUERY_MODES:
        mode = "none"
    q = str(obj.get("query_text") or "").strip()
    if mode == "none":
        q = ""
    return {
        "turn_id": int(obj.get("turn_id") or turn_id),
        "query_mode": mode,
        "query_text": q,
        "known_terms": _as_list(obj.get("known_terms")),
        "reason": str(obj.get("reason") or "").strip(),
    }


def normalize_trigger_step(obj: Dict[str, Any], turn_id: int) -> Dict[str, Any]:
    trig = obj.get("trigger", False)
    if isinstance(trig, str):
        trig = trig.strip().lower() in {"true", "yes", "1", "是"}
    trig = bool(trig)

    evidence_mode = str(obj.get("evidence_mode") or "").strip()
    # Backward compatibility with older prompt labels.
    if evidence_mode == "retrieved_evidence":
        evidence_mode = "retrieve_new_evidence"
    if evidence_mode == "insufficient_evidence":
        evidence_mode = "retrieve_new_evidence" if trig else "reuse_existing_evidence"

    if evidence_mode not in EVIDENCE_MODES:
        evidence_mode = "retrieve_new_evidence" if trig else "no_evidence_needed"

    # Enforce new semantics: trigger means new RAG call.
    if trig:
        evidence_mode = "retrieve_new_evidence"
    elif evidence_mode == "retrieve_new_evidence":
        evidence_mode = "reuse_existing_evidence"

    evidence_status = str(obj.get("evidence_status") or "").strip()
    if evidence_status not in EVIDENCE_STATUS:
        if evidence_mode == "no_evidence_needed":
            evidence_status = "not_needed"
        else:
            evidence_status = "uncertain"

    basis = str(obj.get("evidence_basis") or "").strip()
    if basis not in EVIDENCE_BASIS:
        if evidence_mode == "no_evidence_needed":
            basis = "none"
        elif str(obj.get("case_id") or "").strip():
            basis = "candidate_case"
        else:
            basis = "inferred_from_history"

    case_id = str(obj.get("case_id") or "").strip().upper()
    if evidence_mode == "no_evidence_needed":
        case_id = ""
        evidence_status = "not_needed"
        basis = "none"

    return {
        "turn_id": int(obj.get("turn_id") or turn_id),
        "trigger": trig,
        "evidence_mode": evidence_mode,
        "evidence_status": evidence_status,
        "case_id": case_id,
        "evidence_basis": basis,
        "reason": str(obj.get("reason") or "").strip(),
    }


def normalize_policy_step(obj: Dict[str, Any], turn_id: int) -> Dict[str, Any]:
    label = str(obj.get("label") or obj.get("policy") or "ProcessAcknowledgement").strip()
    if label not in POLICY_LABELS:
        label = "ProcessAcknowledgement"
    case_id = str(obj.get("case_id") or "").strip().upper()
    if label not in {"ExplainedResponse", "CaseRecommendation"}:
        case_id = ""
    return {
        "turn_id": int(obj.get("turn_id") or turn_id),
        "label": label,
        "case_id": case_id,
        "reason": str(obj.get("reason") or "").strip(),
    }


def normalize_response(obj: Dict[str, Any] | str) -> str:
    if isinstance(obj, dict):
        return str(obj.get("response") or obj.get("text") or "").strip()
    return str(obj or "").strip()


def parse_trajectory(text: str, kind: str, expected_len: int) -> List[Dict[str, Any]]:
    obj = parse_json_obj(text)
    traj = obj.get("trajectory", [])
    if not isinstance(traj, list):
        traj = []
    out: List[Dict[str, Any]] = []
    for i in range(expected_len):
        raw = traj[i] if i < len(traj) and isinstance(traj[i], dict) else {}
        turn_id = i + 1
        if kind == "query":
            out.append(normalize_query_step(raw, turn_id))
        elif kind == "trigger":
            out.append(normalize_trigger_step(raw, turn_id))
        elif kind == "policy":
            out.append(normalize_policy_step(raw, turn_id))
        else:
            raise ValueError(f"Unknown trajectory kind: {kind}")
    return out


def parse_step(text: str, kind: str, turn_id: int) -> Dict[str, Any]:
    obj = parse_json_obj(text)
    if "trajectory" in obj and isinstance(obj["trajectory"], list) and obj["trajectory"]:
        obj = obj["trajectory"][-1]
    if kind == "query":
        return normalize_query_step(obj, turn_id)
    if kind == "trigger":
        return normalize_trigger_step(obj, turn_id)
    if kind == "policy":
        return normalize_policy_step(obj, turn_id)
    raise ValueError(f"Unknown step kind: {kind}")


def normalize_retrieval(raw: Any, top_k: int = 5) -> Dict[str, Any]:
    cases: List[Dict[str, Any]] = []
    selected_case_ids: List[str] = []

    if raw is None:
        return {"selected_case_ids": [], "retrieved_cases": []}

    if isinstance(raw, dict):
        if isinstance(raw.get("selected_case_ids"), list):
            selected_case_ids = [str(x).strip().upper() for x in raw.get("selected_case_ids", []) if str(x).strip()]
        for key in ("retrieved_cases", "cases", "results", "data"):
            if isinstance(raw.get(key), list):
                cases = raw[key]
                break
        if not cases and isinstance(raw.get("case_diagnostics"), list):
            cases = raw["case_diagnostics"]
    elif isinstance(raw, list):
        cases = raw

    norm_cases: List[Dict[str, Any]] = []
    for c in cases[:top_k]:
        if isinstance(c, str):
            cid = c.strip().upper()
            norm_cases.append({"case_id": cid, "title": ""})
            continue
        if not isinstance(c, dict):
            continue
        cid = str(c.get("case_id") or c.get("id") or c.get("caseId") or c.get("caseID") or "").strip().upper()
        title = str(c.get("title") or c.get("name") or c.get("case_title") or "").strip()
        score = c.get("score", c.get("similarity", c.get("relevance")))
        item = {"case_id": cid, "title": title}
        if score is not None:
            item["score"] = score
        norm_cases.append(item)
        if cid and cid not in selected_case_ids:
            selected_case_ids.append(cid)

    if not norm_cases and selected_case_ids:
        norm_cases = [{"case_id": cid, "title": ""} for cid in selected_case_ids[:top_k]]
    if not selected_case_ids:
        selected_case_ids = [c["case_id"] for c in norm_cases if c.get("case_id")]

    return {
        "selected_case_ids": selected_case_ids[:top_k],
        "retrieved_cases": norm_cases[:top_k],
    }
