import json
from json_repair import repair_json
import sys
from pathlib import Path

sys.path.append(str(Path(__file__).parent.parent.parent))

from src.api import parallel_inference
from src.io import load_json, save_json, messages_to_string, get_resume_index, get_prompt


def parse_policy_result(result):
    """统一解析 policy 输出"""
    try:
        if isinstance(result, dict):
            response_str = result.get("response")
        elif isinstance(result, str):
            response_str = json.loads(repair_json(result)).get("response")
        else:
            print("Error Result Type:", type(result), result)
            return {"error": "invalid_type"}

        if not response_str:
            return {"error": "empty_response"}

        return json.loads(repair_json(response_str))

    except Exception as e:
        print("Parse Error:", e)
        return {"error": "parse_failed"}


if __name__ == "__main__":
    rag_trigger_path = "/home/aarc/CuhkszTeam/nas1/RUNTIME/Baselines/静态单轮SFT/大模型能力蒸馏/RAGTrigger/rag_trigger.json"
    rag_trigger_data = load_json(rag_trigger_path)

    prompt_path = "/home/aarc/CuhkszTeam/nas1/RUNTIME/Baselines/静态单轮SFT/大模型能力蒸馏/prompts/policy.prompt"
    prompt = get_prompt(prompt_path)

    policy_path = "policy.json"

    resume_idx, existing_data = get_resume_index(policy_path)
    print(f"已经有 {resume_idx} 条数据被标注过了")

    remaining_data = rag_trigger_data[resume_idx:]

    batch_size = 100
    policy_data = existing_data

    for start in range(0, len(remaining_data), batch_size):
        end = min(start + batch_size, len(remaining_data))
        batch_data = remaining_data[start:end]

        print(f"正在处理 {resume_idx + start} ~ {resume_idx + end}")

        sample_list = []

        for i in batch_data:
            sample = {}
            sample["id"] = f"{i['dialogue_id']}-{i['turn_id']}"

            dialogue_history_str = messages_to_string(i["dialogue_history"])

            if i["rag_trigger"]["knowledge_support"] == "仅常识":
                retrieval_cases_str = "无，目前用户需求仅需要常识解决"
            elif i["rag_trigger"]["knowledge_support"] == "知识不足":
                retrieval_cases_str = "无，目前知识不足以解决用户需求"
            else:
                retrieval_cases_str = str(i["retrieval_cases"])

            content = prompt.replace("{dialog_history}", dialogue_history_str)
            content = content.replace("{retrieved_cases}", retrieval_cases_str)

            sample["messages"] = [{"role": "user", "content": content}]
            sample_list.append(sample)

        results = parallel_inference(sample_list)

        new_data = []

        for idx, i in enumerate(batch_data):
            sample = {
                "dialogue_id": i["dialogue_id"],
                "turn_id": i["turn_id"],
                "dialogue_history": i["dialogue_history"],
                "query": i["query"],
                "retrieval_cases": i["retrieval_cases"],
                "rag_trigger": i["rag_trigger"],
                "current_response": i["current_response"],
            }

            parsed = parse_policy_result(results[idx])
            sample["policy"] = parsed

            new_data.append(sample)

        policy_data.extend(new_data)

        # ✅ 每100条保存一次 checkpoint
        save_json(policy_data, policy_path)
        print(f"已保存 checkpoint，总数={len(policy_data)}")

    print("全部处理完成")
